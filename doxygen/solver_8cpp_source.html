49template <typename T>
50double SolverT<T>::solve(Eigen::VectorX<T>& params) {
51 MT_PROFILE_FUNCTION();
52
53 if (storeHistory) {
54 auto& parameterHistory = iterationHistory_["parameters"];
55 auto& errorHistory = iterationHistory_["error"];
56 auto& iterationCount = iterationHistory_["iterations"];
57 if (parameterHistory.rows() != gsl::narrow_cast<Eigen::Index>(numParameters_) ||
58 parameterHistory.cols() != gsl::narrow_cast<Eigen::Index>(maxIterations_)) {
59 parameterHistory.resize(numParameters_, maxIterations_);
60 }
61 if (errorHistory.rows() != gsl::narrow_cast<Eigen::Index>(maxIterations_) ||
62 errorHistory.cols() != 1) {
63 errorHistory.resize(maxIterations_, 1);
64 }
65 if (iterationCount.rows() != 1 || iterationCount.cols() != 1) {
66 iterationCount.resize(1, 1);
67 }
68
69 parameterHistory.setZero();
70 errorHistory.setZero();
71 iterationCount.setZero();
72 }
73
74 errorHistory_.clear();
75 errorHistory_.reserve(maxIterations_);
76
77 MT_CHECK(params.size() == gsl::narrow<Eigen::DenseIndex>(numParameters_));
78 parameters_ = params;
79
80 static_assert(
81 std::is_same<decltype(error_), decltype(lastError_)>::value,
82 "error and lastError_ should be of the same type");
83 error_ = std::numeric_limits<decltype(error_)>::max();
84 lastError_ = std::numeric_limits<decltype(lastError_)>::max();
85
86 initializeSolver();
87
88 iteration_ = 0;
89 for (; iteration_ < maxIterations_; iteration_++) {
90 // do actual iteration (iterations should update the error value)
91 doIteration();
92 errorHistory_.push_back(error_);
93
94 MT_LOGI_IF(this->verbose_, "Iteration: {}, error: {}", this->iteration_, error_);
95
96 // check for convergence
97 bool converged = false;
98 if (std::fabs(lastError_ - error_) / (std::fabs(error_) + std::numeric_limits<float>::min()) <=
99 threshold_ * std::numeric_limits<float>::epsilon()) {
100 converged = true;
101 }
102
103 // set history
104 if (storeHistory) {
105 auto& parameterHistory = iterationHistory_["parameters"];
106 auto& errorHistory = iterationHistory_["error"];
107 parameterHistory.col(iteration_) = parameters_;
108 errorHistory(iteration_, 0) = gsl::narrow_cast<float>(error_);
109
110 solverFunction_->storeHistory(iterationHistory_, iteration_, maxIterations_);
111 }
112
113 if (iteration_ >= minIterations_ && converged) {
114 break;
115 }
116
117 // update last error
118 lastError_ = error_;
119 }
120
121 if (storeHistory) {
122 auto& iterationCount = iterationHistory_["iterations"];
123 iterationCount(0, 0) = gsl::narrow_cast<float>(iteration_);
124 }
125
126 params = parameters_;
127 return error_;
128}
