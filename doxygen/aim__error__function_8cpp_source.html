17template <typename T>
18void AimDistErrorFunctionT<T>::evalFunction(
19 const size_t constrIndex,
20 const JointStateT<T>& state,
21 Vector3<T>& f,
22 optional_ref<std::array<Vector3<T>, 2>> v,
23 optional_ref<std::array<Eigen::Matrix3<T>, 2>> dfdv) const {
24 MT_PROFILE_FUNCTION();
25
26 const AimDataT<T>& constr = this->constraints_[constrIndex];
27 const Vector3<T> point = state.transform * constr.localPoint;
28 const Vector3<T> srcDir = state.rotation() * constr.localDir;
29 const Vector3<T> tgtVec = constr.globalTarget - point;
30 const T projLength = srcDir.dot(tgtVec);
31
32 // f = (globalTarget - point).dot(srcDir) * srcDir - (globalTarget - point)
33 f = projLength * srcDir - tgtVec;
34 if (v) {
35 v->get().at(0) = point;
36 v->get().at(1) = srcDir;
37 }
38 if (dfdv) {
39 // df/d(point) = I - outterProd(srcDir, srcDir)
40 dfdv->get().at(0).setIdentity();
41 dfdv->get().at(0).noalias() -= srcDir * srcDir.transpose();
42 // df/d(dir) = projLength * I + outterProd(srcDir, tgtVec)
43 dfdv->get().at(1).noalias() = srcDir * tgtVec.transpose();
44 dfdv->get().at(1).diagonal().array() += projLength;
45 }
46}

48template <typename T>
49void AimDirErrorFunctionT<T>::evalFunction(
50 const size_t constrIndex,
51 const JointStateT<T>& state,
52 Vector3<T>& f,
53 optional_ref<std::array<Vector3<T>, 2>> v,
54 optional_ref<std::array<Eigen::Matrix3<T>, 2>> dfdv) const {
55 MT_PROFILE_FUNCTION();
56
57 const AimDataT<T>& constr = this->constraints_[constrIndex];
58 const Vector3<T> point = state.transform * constr.localPoint;
59 const Vector3<T> srcDir = state.rotation() * constr.localDir;
60 const Vector3<T> tgtVec = constr.globalTarget - point;
61 const T tgtNorm = tgtVec.norm();
62 Vector3<T> tgtDir = Vector3<T>::Zero();
63 if (tgtNorm > 1e-16) {
64 tgtDir = tgtVec / tgtNorm;
65 }
66
67 // f = srcDir - (globalTarget - point).normalize()
68 f = srcDir - tgtDir;
69 if (v) {
70 v->get().at(0) = point;
71 v->get().at(1) = srcDir;
72 }
73 if (dfdv) {
74 // df/d(point) = (I - outterProd(tgtDir, tgtDir)) / tgtNorm
75 dfdv->get().at(0).setZero();
76 if (tgtNorm > 1e-16) {
77 dfdv->get().at(0).noalias() -= (tgtDir * tgtDir.transpose()) / tgtNorm;
78 dfdv->get().at(0).diagonal().array() += T(1) / tgtNorm;
79 }
80 dfdv->get().at(1).setIdentity();
81 }
82}
