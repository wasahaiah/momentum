17template <typename T>
18VectorX<T> BlendShapeBase::computeDeltas(const BlendWeightsT<T>& blendWeights) const {
19 MT_CHECK(
20 blendWeights.size() <= shapeVectors_.cols(),
21 "{} vs {}",
22 blendWeights.size(),
23 shapeVectors_.cols());
24 return shapeVectors_.leftCols(blendWeights.size()).template cast<T>() * blendWeights.v;
25}

27template <typename T>
28void BlendShapeBase::applyDeltas(

31 const VectorX<T> deltas = computeDeltas(blendWeights);

36BlendShapeBase::BlendShapeBase(
37 const size_t modelSize,
38 const size_t numShapes,
39 std::span<const std::string> shapeNames)
40 : shapeVectors_(MatrixXf::Zero(modelSize * 3, numShapes)) {
41 shapeNames_.assign(shapeNames.begin(), shapeNames.end());
42 if (shapeNames_.size() != numShapes) {
43 MT_LOGW_IF(
44 !shapeNames.empty(),
45 "Provided some but not all blendshape names, all shape names will be auto-generated.");
46 shapeNames_.resize(numShapes);
47 for (size_t i = 0; i < numShapes; ++i) {
48 shapeNames_[i] = std::string("shape_") + std::to_string(i);
49 }
50 }
51}

71void BlendShapeBase::setShapeVector(
72 const size_t index,
73 std::span<const Vector3f> shape,
74 std::string_view name) {
75 MT_CHECK(modelSize() == shape.size(), "{} is not {}", modelSize(), shape.size());
76 MT_CHECK(
77 gsl::narrow<Eigen::Index>(shape.size()) * 3 == shapeVectors_.rows(),
78 "{} is not {}",
79 shape.size() * 3,
80 shapeVectors_.rows());
81 MT_CHECK(
82 index < static_cast<size_t>(shapeVectors_.cols()), "{} vs {}", index, shapeVectors_.cols());
83
84 shapeVectors_.col(index) = Map<const VectorXf>(&shape[0][0], shape.size() * 3);
85 if (name.empty()) {
86 shapeNames_[index] = std::string("shape_") + std::to_string(index);
87 } else {
88 shapeNames_[index] = name;
89 }
90}
