15template <typename T>
16void MppcaT<T>::set(
17 const VectorX<T>& inPi,
18 const MatrixX<T>& mmu,
19 std::span<const MatrixX<T>> W,
20 const VectorX<T>& sigma2) {
21 using CovarianceMatrix = LowRankCovarianceMatrixT<T>;
22
23 d = mmu.cols();
24 p = sigma2.rows();
25
26 // set the data
27 Rpre.setZero(p);
28 Cinv.resize(p);
29 L.resize(p);
30
31 CovarianceMatrix covariance;
32
33 const T half_d_log_2pi = 0.5 * static_cast<T>(d) * std::log(twopi<T>());
34
35 for (size_t c = 0; c < p; c++) {
36 covariance.reset(std::sqrt(sigma2(c)), W[c].transpose());
37
38 // Back-substitution of the identity matrix gives the inverse:
39 const Eigen::MatrixX<T> L_tmp = covariance.R()
40 .template triangularView<Eigen::Upper>()
41 .solve(Eigen::MatrixX<T>::Identity(d, d))
42 .transpose();
43 Cinv[c] = covariance.R().template triangularView<Eigen::Upper>().solve(L_tmp);
44 L[c] = L_tmp;
45
46 const T C_logDeterminant = covariance.logDeterminant();
47
48 // Rpre is the constant part of R that does not depend on the parameters, so we can precalculate
49 // it
50 Rpre(c) = std::log(inPi(c)) - 0.5 * C_logDeterminant - half_d_log_2pi;
51 }
52 mu = mmu;
53
54 names = std::vector<std::string>(d);
55}

57template <typename T, typename T2>
58std::vector<Eigen::MatrixX<T2>> castMatrices(std::span<const Eigen::MatrixX<T>> m_in) {
59 std::vector<Eigen::MatrixX<T2>> result;
60 result.reserve(m_in.size());
61 for (const auto& m : m_in) {
62 result.emplace_back(m.template cast<T2>());

83template <typename T>
84bool MppcaT<T>::isApprox(const MppcaT<T>& mppcaT) const {
85 auto vecIsApprox = [](auto& vL, auto& vR) {
86 if (vL.size() != vR.size()) {
87 return false;
88 }
89 for (size_t i = 0; i < vL.size(); i++) {
90 if (!vL[i].isApprox(vR[i])) {
91 return false;
92 }
93 }
94 return true;
95 };
96 return (
97 (d == mppcaT.d) && (p == mppcaT.p) && (names == mppcaT.names) && mu.isApprox(mppcaT.mu) &&
98 vecIsApprox(Cinv, mppcaT.Cinv) && vecIsApprox(L, mppcaT.L) && Rpre.isApprox(mppcaT.Rpre));
99}
