51[[nodiscard]] ::fbxsdk::FbxAxisSystem::EUpVector toFbx(const FbxUpVector upVector) {
52 switch (upVector) {
53 case FbxUpVector::XAxis:
54 return ::fbxsdk::FbxAxisSystem::EUpVector::eXAxis;
55 case FbxUpVector::YAxis:
56 return ::fbxsdk::FbxAxisSystem::EUpVector::eYAxis;
57 case FbxUpVector::ZAxis:
58 return ::fbxsdk::FbxAxisSystem::EUpVector::eZAxis;
59 default:
60 MT_THROW("Unsupported up vector");
61 }
62}
63
64[[nodiscard]] ::fbxsdk::FbxAxisSystem::EFrontVector toFbx(const FbxFrontVector frontVector) {
65 switch (frontVector) {
66 case FbxFrontVector::ParityEven:
67 return ::fbxsdk::FbxAxisSystem::EFrontVector::eParityEven;
68 case FbxFrontVector::ParityOdd:
69 return ::fbxsdk::FbxAxisSystem::EFrontVector::eParityOdd;
70 default:
71 MT_THROW("Unsupported front vector");
72 }
73}

76 switch (coordSystem) {
77 case FbxCoordSystem::RightHanded:

79 case FbxCoordSystem::LeftHanded:

93void createLocatorNodes(
94 const Character& character,
95 ::fbxsdk::FbxScene* scene,
96 const std::vector<::fbxsdk::FbxNode*>& skeletonNodes) {
97 for (const auto& loc : character.locators) {
98 ::fbxsdk::FbxMarker* markerAttribute = ::fbxsdk::FbxMarker::Create(scene, loc.name.c_str());
99 markerAttribute->Look.Set(::fbxsdk::FbxMarker::ELook::eHardCross);
100
101 // create the node
102 ::fbxsdk::FbxNode* locatorNode = ::fbxsdk::FbxNode::Create(scene, loc.name.c_str());
103 locatorNode->SetNodeAttribute(markerAttribute);
104
105 // set translation offset
106 locatorNode->LclTranslation.Set(FbxVector4(loc.offset[0], loc.offset[1], loc.offset[2]));
107
108 // set parent if it has one
109 if (loc.parent != kInvalidIndex) {
110 skeletonNodes[loc.parent]->AddChild(locatorNode);
111 }
112 }

115void createCollisionGeometryNodes(
116 const Character& character,
117 ::fbxsdk::FbxScene* scene,
118 const std::vector<::fbxsdk::FbxNode*>& skeletonNodes) {
119 if (!character.collision) {
120 MT_LOGD(
121 "No collision geometry found in character, skipping creation of collision geometry nodes");
122 return;
123 }
124
125 const auto& collisions = *character.collision;
126 for (auto i = 0u; i < collisions.size(); ++i) {
127 const TaperedCapsule& collision = collisions[i];
128
129 ::fbxsdk::FbxNode* collisionNode =
130 ::fbxsdk::FbxNode::Create(scene, ("Collision " + std::to_string(i)).c_str());
131 auto* nullNodeAttr =
132 ::fbxsdk::FbxNull::Create(scene, "Null"); // TODO: Find a good node attribute
133 collisionNode->SetNodeAttribute(nullNodeAttr);
134
135 ::fbxsdk::FbxProperty::Create(collisionNode, ::fbxsdk::FbxBoolDT, "Col_Type").Set(true);
136 ::fbxsdk::FbxProperty::Create(collisionNode, ::fbxsdk::FbxFloatDT, "Length")
137 .Set(collision.length);
138 ::fbxsdk::FbxProperty::Create(collisionNode, ::fbxsdk::FbxFloatDT, "Rad_A")
139 .Set(collision.radius[0]);
140 ::fbxsdk::FbxProperty::Create(collisionNode, ::fbxsdk::FbxFloatDT, "Rad_B")
141 .Set(collision.radius[1]);
142
143 collisionNode->LclTranslation.Set(FbxVector4(
144 collision.transformation.translation.x(),
145 collision.transformation.translation.y(),
146 collision.transformation.translation.z()));
147 const Vector3f rot = rotationMatrixToEulerXYZ<float>(
148 collision.transformation.rotation.toRotationMatrix(), EulerConvention::Extrinsic);
149 collisionNode->LclRotation.Set(FbxDouble3(toDeg(rot.x()), toDeg(rot.y()), toDeg(rot.z())));
150 collisionNode->LclScaling.Set(FbxDouble3(1));
151
152 if (collision.parent != kInvalidIndex) {
153 skeletonNodes[collision.parent]->AddChild(collisionNode);
154 } else {
155 MT_LOGE("Found a collision node with no parent");
156 }
157 }

160void setFrameRate(::fbxsdk::FbxScene* scene, const double framerate) {
161 // enumerate common frame rates first, then resort to custom framerate
162 if (std::abs(framerate - 30.0) < 1e-6) {
163 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames30);
164 } else if (std::abs(framerate - 24.0) < 1e-6) {
165 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames24);
166 } else if (std::abs(framerate - 48.0) < 1e-6) {
167 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames48);
168 } else if (std::abs(framerate - 50.0) < 1e-6) {
169 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames50);
170 } else if (std::abs(framerate - 60.0) < 1e-6) {
171 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames60);
172 } else if (std::abs(framerate - 72.0) < 1e-6) {
173 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames72);
174 } else if (std::abs(framerate - 96.0) < 1e-6) {
175 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames96);
176 } else if (std::abs(framerate - 100.0) < 1e-6) {
177 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames100);
178 } else if (std::abs(framerate - 120.0) < 1e-6) {
179 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames120);
180 } else if (std::abs(framerate - 1000.0) < 1e-6) {
181 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames1000);
182 } else {
183 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eCustom);
184 scene->GetGlobalSettings().SetCustomFrameRate(framerate);
185 }
186}

188// jointValues: (numJointParameters x numFrames) matrix of joint values
189void createAnimationCurves(
190 const Character& character,
191 ::fbxsdk::FbxScene* scene,
192 const std::vector<::fbxsdk::FbxNode*>& skeletonNodes,
193 const MatrixXf& jointValues,
194 const double framerate,
195 const bool skipActiveJointParamCheck) {
196 // set the framerate
197 setFrameRate(scene, framerate);
198
199 const auto& aj = character.parameterTransform.activeJointParams;
200
201 // create animation stack
202 ::fbxsdk::FbxAnimStack* animStack =
203 ::fbxsdk::FbxAnimStack::Create(scene, "Skeleton Animation Stack");
204 ::fbxsdk::FbxAnimLayer* animBaseLayer = ::fbxsdk::FbxAnimLayer::Create(scene, "Layer0");
205 animStack->AddMember(animBaseLayer);
206
207 // create anim curves for each joint and store them in an array
208 std::vector<::fbxsdk::FbxAnimCurve*> animCurves(character.skeleton.joints.size() * 9, nullptr);
209 std::vector<size_t> animCurvesIndex;
210 for (size_t i = 0; i < character.skeleton.joints.size(); i++) {
211 const size_t jointIndex = i * kParametersPerJoint;
212 const size_t index = i * 9;
213 skeletonNodes[i]->LclTranslation.GetCurveNode(true);
214 if (skipActiveJointParamCheck || aj[jointIndex + 0]) {
215 animCurves[index + 0] = skeletonNodes[i]->LclTranslation.GetCurve(
216 animBaseLayer, FBXSDK_CURVENODE_COMPONENT_X, true);
217 animCurvesIndex.push_back(index + 0);
218 }
219 if (skipActiveJointParamCheck || aj[jointIndex + 1]) {
220 animCurves[index + 1] = skeletonNodes[i]->LclTranslation.GetCurve(
221 animBaseLayer, FBXSDK_CURVENODE_COMPONENT_Y, true);
222 animCurvesIndex.push_back(index + 1);
223 }
224 if (skipActiveJointParamCheck || aj[jointIndex + 2]) {
225 animCurves[index + 2] = skeletonNodes[i]->LclTranslation.GetCurve(
226 animBaseLayer, FBXSDK_CURVENODE_COMPONENT_Z, true);
227 animCurvesIndex.push_back(index + 2);
228 }
229 skeletonNodes[i]->LclRotation.GetCurveNode(true);
230 if (skipActiveJointParamCheck || aj[jointIndex + 3]) {
231 animCurves[index + 3] =
232 skeletonNodes[i]->LclRotation.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_X, true);
233 animCurvesIndex.push_back(index + 3);
234 }
235 if (skipActiveJointParamCheck || aj[jointIndex + 4]) {
236 animCurves[index + 4] =
237 skeletonNodes[i]->LclRotation.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_Y, true);
238 animCurvesIndex.push_back(index + 4);
239 }
240 if (skipActiveJointParamCheck || aj[jointIndex + 5]) {
241 animCurves[index + 5] =
242 skeletonNodes[i]->LclRotation.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_Z, true);
243 animCurvesIndex.push_back(index + 5);
244 }
245 skeletonNodes[i]->LclScaling.GetCurveNode(true);
246 if (skipActiveJointParamCheck || aj[jointIndex + 6]) {
247 animCurves[index + 6] =
248 skeletonNodes[i]->LclScaling.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_X, true);
249 animCurvesIndex.push_back(index + 6);
250 animCurves[index + 7] =
251 skeletonNodes[i]->LclScaling.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_Y, true);
252 animCurvesIndex.push_back(index + 7);
253 animCurves[index + 8] =
254 skeletonNodes[i]->LclScaling.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_Z, true);
255 animCurvesIndex.push_back(index + 8);
256 }
257 }
258
259 // calculate the actual motion and set the keyframes
260 ::fbxsdk::FbxTime time;
261 // now go over each animCurveIndex and generate the curve
262 for (const auto ai : animCurvesIndex) {
263 const size_t jointIndex = ai / 9;
264 const size_t jointOffset = ai % 9;
265 const size_t parameterIndex =
266 jointIndex * kParametersPerJoint + std::min(jointOffset, size_t(6));
267 if (!skipActiveJointParamCheck && aj[parameterIndex] == 0) {
268 continue;
269 }
270
271 animCurves[ai]->KeyModifyBegin();
272 for (size_t f = 0; f < jointValues.cols(); f++) {
273 // set keyframe time
274 time.SetSecondDouble(static_cast<double>(f) / framerate);
275
276 // get joint value
277 float jointVal = jointValues(parameterIndex, f);
278
279 // add translation offset for tx values
280 if (jointOffset < 3 && jointIndex < character.skeleton.joints.size()) {
281 jointVal += character.skeleton.joints[jointIndex].translationOffset[jointOffset];
282 }
283 // convert to degrees
284 else if (jointOffset >= 3 && jointOffset <= 5) {
285 jointVal = toDeg(jointVal);
286 }
287 // convert to non-exponential scaling
288 else {
289 jointVal = std::pow(2.0f, jointVal);
290 }
291
292 const auto keyIndex = animCurves[ai]->KeyAdd(time);
293 animCurves[ai]->KeySet(keyIndex, time, jointVal);
294 }
295 animCurves[ai]->KeyModifyEnd();
296 }
297}

299// Post-process function to prepend namespace to all nodes in the scene
300void prependNamespaceToAllNodes(::fbxsdk::FbxNode* node, const std::string& namespacePrefix) {
301 if (namespacePrefix.empty() || node == nullptr) {
302 return;
303 }
304
305 // Prepend namespace to the node name
306 const std::string currentName = node->GetName();
307 const std::string newName = namespacePrefix + currentName;
308 node->SetName(newName.c_str());
309
310 // Recursively process all children
311 for (int i = 0; i < node->GetChildCount(); ++i) {
312 prependNamespaceToAllNodes(node->GetChild(i), namespacePrefix);
313 }
314}

316// Create marker nodes under a "Markers" hierarchy and add animation for their translations
317std::vector<::fbxsdk::FbxNode*> createMarkerNodes(
318 ::fbxsdk::FbxScene* scene,
319 std::span<const std::vector<Marker>> markerSequence,
320 const double framerate) {
321 std::vector<::fbxsdk::FbxNode*> markerNodes;
322
323 if (markerSequence.empty()) {
324 return markerNodes;
325 }
326
327 // Set the framerate for the scene
328 setFrameRate(scene, framerate);
329
330 // Create a root node for all markers
331 ::fbxsdk::FbxNode* markersRootNode = ::fbxsdk::FbxNode::Create(scene, "Markers");
332 ::fbxsdk::FbxNull* markersRootAttr = ::fbxsdk::FbxNull::Create(scene, "MarkersRootNull");
333 markersRootNode->SetNodeAttribute(markersRootAttr);
334
335 // Add custom property to identify this as a Momentum markers root
336 ::fbxsdk::FbxProperty::Create(markersRootNode, ::fbxsdk::FbxBoolDT, kMomentumMarkersRootProperty)
337 .Set(true);
338
339 // Collect unique marker names and organize data per marker
340 std::map<std::string, size_t> markerNameToIndex;
341 std::vector<std::string> markerNames;
342 std::vector<std::vector<float>> timestamps;
343 std::vector<std::vector<Vector3d>> markerPositions;
344
345 for (size_t frameIndex = 0; frameIndex < markerSequence.size(); ++frameIndex) {
346 const float timestamp = static_cast<float>(frameIndex) / static_cast<float>(framerate);
347 for (const auto& marker : markerSequence[frameIndex]) {
348 // Skip occluded markers
349 if (marker.occluded) {
350 continue;
351 }
352
353 // Create new arrays if marker is unknown
354 if (markerNameToIndex.count(marker.name) == 0) {
355 const auto index = timestamps.size();
356 timestamps.emplace_back();
357 markerPositions.emplace_back();
358 markerNameToIndex[marker.name] = index;
359 markerNames.emplace_back(marker.name);
360 }
361
362 // Add timestamp and position for this marker
363 const auto& index = markerNameToIndex.at(marker.name);
364 timestamps[index].push_back(timestamp);
365 markerPositions[index].push_back(marker.pos);
366 }
367 }
368
369 // Create a marker node for each unique marker name
370 for (const auto& markerName : markerNames) {
371 // Create FbxMarker attribute for visualization
372 ::fbxsdk::FbxMarker* markerAttribute = ::fbxsdk::FbxMarker::Create(scene, markerName.c_str());
373 markerAttribute->Look.Set(::fbxsdk::FbxMarker::ELook::eHardCross);
374
375 // Create the node
376 ::fbxsdk::FbxNode* markerNode = ::fbxsdk::FbxNode::Create(scene, markerName.c_str());
377 markerNode->SetNodeAttribute(markerAttribute);
378
379 // Add custom property to identify this as a Momentum marker
380 ::fbxsdk::FbxProperty::Create(markerNode, ::fbxsdk::FbxBoolDT, kMomentumMarkerProperty)
381 .Set(true);
382
383 // Initialize at origin
384 markerNode->LclTranslation.Set(FbxVector4(0.0, 0.0, 0.0));
385
386 // Add to markers root
387 markersRootNode->AddChild(markerNode);
388 markerNodes.push_back(markerNode);
389 }
390
391 // Add markers root to scene root
392 scene->GetRootNode()->AddChild(markersRootNode);
393
394 // Create animation stack if we have motion data
395 if (!timestamps.empty() && !timestamps[0].empty()) {
396 // Get or create animation stack
397 ::fbxsdk::FbxAnimStack* animStack = nullptr;
398 if (scene->GetSrcObjectCount<::fbxsdk::FbxAnimStack>() > 0) {
399 // Use existing animation stack (created for skeleton animation)
400 animStack = scene->GetSrcObject<::fbxsdk::FbxAnimStack>(0);
401 } else {
402 // Create new animation stack
403 animStack = ::fbxsdk::FbxAnimStack::Create(scene, "Marker Animation Stack");
404 }
405
406 // Get or create base layer
407 ::fbxsdk::FbxAnimLayer* animBaseLayer = nullptr;
408 if (animStack->GetMemberCount<::fbxsdk::FbxAnimLayer>() > 0) {
409 animBaseLayer = animStack->GetMember<::fbxsdk::FbxAnimLayer>(0);
410 } else {
411 animBaseLayer = ::fbxsdk::FbxAnimLayer::Create(scene, "Layer0");
412 animStack->AddMember(animBaseLayer);
413 }
414
415 // Create animation curves for each marker
416 for (size_t j = 0; j < markerNames.size(); ++j) {
417 if (timestamps[j].empty()) {
418 continue;
419 }
420
421 auto* markerNode = markerNodes.at(j);
422
423 // Create curves for X, Y, Z translation
424 markerNode->LclTranslation.GetCurveNode(animBaseLayer, true);
425 ::fbxsdk::FbxAnimCurve* curveX =
426 markerNode->LclTranslation.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_X, true);
427 ::fbxsdk::FbxAnimCurve* curveY =
428 markerNode->LclTranslation.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_Y, true);
429 ::fbxsdk::FbxAnimCurve* curveZ =
430 markerNode->LclTranslation.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_Z, true);
431
432 // Add keyframes for each timestamp
433 curveX->KeyModifyBegin();
434 curveY->KeyModifyBegin();
435 curveZ->KeyModifyBegin();
436
437 for (size_t k = 0; k < timestamps[j].size(); ++k) {
438 ::fbxsdk::FbxTime time;
439 time.SetSecondDouble(timestamps[j][k]);
440
441 const auto& pos = markerPositions[j][k];
442
443 const auto keyIndexX = curveX->KeyAdd(time);
444 curveX->KeySet(keyIndexX, time, static_cast<float>(pos.x()));
445
446 const auto keyIndexY = curveY->KeyAdd(time);
447 curveY->KeySet(keyIndexY, time, static_cast<float>(pos.y()));
448
449 const auto keyIndexZ = curveZ->KeyAdd(time);
450 curveZ->KeySet(keyIndexZ, time, static_cast<float>(pos.z()));
451 }
452
453 curveX->KeyModifyEnd();
454 curveY->KeyModifyEnd();
455 curveZ->KeyModifyEnd();
456 }
457 }
458
459 return markerNodes;
460}

462void saveSkinWeightsToFbx(
463 const Character& character,
464 ::fbxsdk::FbxScene* scene,
465 ::fbxsdk::FbxMesh* mesh,
466 const std::unordered_map<size_t, fbxsdk::FbxNode*>& jointToNodeMap) {
467 ::fbxsdk::FbxSkin* fbxskin = ::fbxsdk::FbxSkin::Create(scene, "meshskinning");
468 fbxskin->SetSkinningType(::fbxsdk::FbxSkin::eLinear);
469 fbxskin->SetGeometry(mesh);
470 FbxAMatrix meshTransform;
471 meshTransform.SetIdentity();
472 for (const auto& jointNode : jointToNodeMap) {
473 size_t jointIdx = jointNode.first;
474 auto* fbxJointNode = jointNode.second;
475
476 std::ostringstream s;
477 s << "skinningcluster_" << jointIdx;
478 FbxCluster* pCluster = ::fbxsdk::FbxCluster::Create(scene, s.str().c_str());
479 pCluster->SetLinkMode(::fbxsdk::FbxCluster::ELinkMode::eNormalize);
480 pCluster->SetLink(fbxJointNode);
481
482 ::fbxsdk::FbxAMatrix globalMatrix = fbxJointNode->EvaluateLocalTransform();
483 ::fbxsdk::FbxNode* pParent = fbxJointNode->GetParent();
484 // TODO: should use inverse bind transform from character instead.
485 while (pParent != nullptr) {
486 globalMatrix = pParent->EvaluateLocalTransform() * globalMatrix;
487 pParent = pParent->GetParent();
488 }
489 pCluster->SetTransformLinkMatrix(globalMatrix);
490 pCluster->SetTransformMatrix(meshTransform);
491
492 for (int i = 0; i < character.skinWeights->index.rows(); i++) {
493 for (int j = 0; j < character.skinWeights->index.cols(); j++) {
494 auto boneIndex = character.skinWeights->index(i, j);
495 if (boneIndex == jointNode.first && character.skinWeights->weight(i, j) > 0) {
496 pCluster->AddControlPointIndex(i, character.skinWeights->weight(i, j));
497 }
498 }
499 }
500 fbxskin->AddCluster(pCluster);
501 }
502 mesh->AddDeformer(fbxskin);
503}

505void saveBlendShapesToFbx(
506 const Character& character,
507 ::fbxsdk::FbxScene* scene,
508 ::fbxsdk::FbxMesh* mesh) {
509 // create blendshape deformer
510 ::fbxsdk::FbxBlendShape* blendShape = ::fbxsdk::FbxBlendShape::Create(scene, "blendshape");
511 blendShape->SetGeometry(mesh);
512
513 // add blendshape channels
514 const auto& shapes = character.blendShape->getShapeVectors();
515 const auto& names = character.blendShape->getShapeNames();
516 const auto& base = character.blendShape->getBaseShape();
517 const int numVertices = character.mesh.get()->vertices.size();
518
519 for (int j = 0; j < numVertices; j++) {
520 FbxVector4 point(base[j].x(), base[j].y(), base[j].z());
521 mesh->SetControlPointAt(point, j);
522 }
523
524 for (size_t i = 0; i < character.blendShape->shapeSize(); i++) {
525 ::fbxsdk::FbxBlendShapeChannel* blendShapeChannelPtr =
526 ::fbxsdk::FbxBlendShapeChannel::Create(scene, ("channel_" + std::to_string(i)).c_str());
527 blendShape->AddBlendShapeChannel(blendShapeChannelPtr);
528
529 // add blendshape targets
530 ::fbxsdk::FbxShape* shape = ::fbxsdk::FbxShape::Create(scene, names[i].c_str());
531 shape->SetControlPointCount(numVertices);
532 for (int j = 0; j < numVertices; j++) {
533 const Eigen::Vector3f delta = shapes.block<3, 1>(j * 3, i);
534 const Eigen::Vector3f p = delta + base[j];
535 shape->SetControlPointAt(FbxVector4(p.x(), p.y(), p.z()), j);
536 shape->AddControlPointIndex(j);
537 }
538 blendShapeChannelPtr->AddTargetShape(shape, 100.0);
539 }
540
541 // add blendshape to mesh
542 mesh->AddDeformer(blendShape);
543}

545void saveFbxCommon(
546 const filesystem::path& filename,
547 const Character& character,
548 const MatrixXf& jointValues,
549 const double framerate,
550 const bool saveMesh,
551 const bool skipActiveJointParamCheck,
552 const FbxCoordSystemInfo& coordSystemInfo,
553 Permissive permissive,
554 std::span<const std::vector<Marker>> markerSequence,
555 std::string_view fbxNamespace) {
556 // ---------------------------------------------
557 // initialize FBX SDK and prepare for export
558 // ---------------------------------------------
559 auto* manager = ::fbxsdk::FbxManager::Create();
560 auto* ios = ::fbxsdk::FbxIOSettings::Create(manager, IOSROOT);
561 manager->SetIOSettings(ios);
562
563 // Create an exporter.
564 ::fbxsdk::FbxExporter* lExporter = ::fbxsdk::FbxExporter::Create(manager, "");
565
566 // Declare the path and filename of the file containing the scene.
567 // In this case, we are assuming the file is in the same directory as the executable.
568 // Going through string() because on windows, wchar_t (native filesystem path) are different from
569 // char https://en.cppreference.com/w/cpp/language/types This avoids a build error on windows
570 // only.
571 std::string sFilename = filename.string();
572 const char* lFilename = sFilename.c_str();
573
574 // Initialize the exporter.
575 bool lExportStatus = lExporter->Initialize(lFilename, -1, manager->GetIOSettings());
576
577 MT_THROW_IF(
578 !lExportStatus,
579 "Unable to initialize fbx exporter {}",
580 lExporter->GetStatus().GetErrorString());
581
582 // Normalize namespace: ensure it ends with ':' if not empty
583 std::string namespacePrefix(fbxNamespace);
584 if (!namespacePrefix.empty() && namespacePrefix.back() != ':') {
585 namespacePrefix += ":";
586 }
587
588 // ---------------------------------------------
589 // create the scene
590 // ---------------------------------------------
591 ::fbxsdk::FbxScene* scene = ::fbxsdk::FbxScene::Create(manager, "momentum_scene");
592 ::fbxsdk::FbxNode* root = scene->GetRootNode();
593
594 // set the coordinate system
595 ::fbxsdk::FbxAxisSystem axis = toFbx(coordSystemInfo);
596 axis.ConvertScene(scene);
597
598 // ---------------------------------------------
599 // create the skeleton nodes
600 // ---------------------------------------------
601 std::vector<::fbxsdk::FbxNode*> skeletonNodes;
602 std::unordered_map<size_t, fbxsdk::FbxNode*> jointToNodeMap;
603
604 ::fbxsdk::FbxNode* skeletonRootNode = nullptr;
605
606 for (size_t i = 0; i < character.skeleton.joints.size(); i++) {
607 const auto& joint = character.skeleton.joints[i];
608
609 // create the node
610 ::fbxsdk::FbxNode* skeletonNode = ::fbxsdk::FbxNode::Create(scene, joint.name.c_str());
611 // create node attribute
612 ::fbxsdk::FbxSkeleton* skeletonAttribute =
613 ::fbxsdk::FbxSkeleton::Create(scene, joint.name.c_str());
614
615 if (joint.parent == kInvalidIndex) {
616 skeletonRootNode = skeletonNode;
617 skeletonAttribute->SetSkeletonType(::fbxsdk::FbxSkeleton::eRoot);
618 } else {
619 skeletonAttribute->SetSkeletonType(::fbxsdk::FbxSkeleton::eLimbNode);
620 }
621 skeletonNode->SetNodeAttribute(skeletonAttribute);
622 jointToNodeMap[i] = skeletonNode;
623
624 // set translation offset
625 skeletonNode->LclTranslation.Set(FbxDouble3(
626 joint.translationOffset[0], joint.translationOffset[1], joint.translationOffset[2]));
627
628 // set pre-rotation
629 const auto angles = rotationMatrixToEulerZYX(joint.preRotation.toRotationMatrix());
630 skeletonNode->SetPivotState(FbxNode::eSourcePivot, FbxNode::ePivotActive);
631 skeletonNode->SetRotationActive(true);
632 skeletonNode->SetPreRotation(
633 ::fbxsdk::FbxNode::eSourcePivot,
634 FbxDouble3(toDeg(angles[2]), toDeg(angles[1]), toDeg(angles[0])));
635
636 // add to list
637 skeletonNodes.emplace_back(skeletonNode);
638 }
639
640 // Second pass: handle the parenting, in case the parents are not in order
641 for (size_t i = 0; i < character.skeleton.joints.size(); i++) {
642 const auto& joint = character.skeleton.joints[i];
643
644 // set parent if it has one
645 auto* skeletonNode = jointToNodeMap[i];
646 if (joint.parent != kInvalidIndex) {
647 skeletonNodes[joint.parent]->AddChild(skeletonNode);
648 }
649 }
650
651 // ---------------------------------------------
652 // create the locator nodes
653 // ---------------------------------------------
654 createLocatorNodes(character, scene, skeletonNodes);
655
656 // ---------------------------------------------
657 // create the collision geometry nodes
658 // ---------------------------------------------
659 createCollisionGeometryNodes(character, scene, skeletonNodes);
660
661 // ---------------------------------------------
662 // create the mesh nodes
663 // ---------------------------------------------
664 if (saveMesh && character.mesh != nullptr) {
665 // Add the mesh
666 const int numVertices = character.mesh.get()->vertices.size();
667 const int numFaces = character.mesh.get()->faces.size();
668 ::fbxsdk::FbxNode* meshNode = ::fbxsdk::FbxNode::Create(scene, "body_mesh");
669 ::fbxsdk::FbxMesh* lMesh = ::fbxsdk::FbxMesh::Create(scene, "mesh");
670 lMesh->SetControlPointCount(numVertices);
671 lMesh->InitNormals(numVertices);
672 for (int i = 0; i < numVertices; i++) {
673 FbxVector4 point(
674 character.mesh.get()->vertices[i].x(),
675 character.mesh.get()->vertices[i].y(),
676 character.mesh.get()->vertices[i].z());
677 FbxVector4 normal(
678 character.mesh.get()->normals[i].x(),
679 character.mesh.get()->normals[i].y(),
680 character.mesh.get()->normals[i].z());
681 lMesh->SetControlPointAt(point, normal, i);
682 }
683 // Add polygons to lMesh
684 for (int iFace = 0; iFace < numFaces; iFace++) {
685 lMesh->BeginPolygon();
686 for (int i = 0; i < 3; i++) { // We have tris for models. This could be extended for
687 // supporting Quads or npoly if needed.
688 lMesh->AddPolygon(character.mesh.get()->faces[iFace][i]);
689 }
690 lMesh->EndPolygon();
691 }
692 lMesh->BuildMeshEdgeArray();
693 meshNode->SetNodeAttribute(lMesh);
694
695 // ---------------------------------------------
696 // add texture coordinates
697 // ---------------------------------------------
698 if (!character.mesh->texcoords.empty()) {
699 const fbxsdk::FbxLayerElement::EType uvType = fbxsdk::FbxLayerElement::eTextureDiffuse;
700
701 // Initialize UV set and indices first. Both functions must be called before adding UVs
702 // and UV indices.
703 lMesh->InitTextureUV(0, uvType);
704 lMesh->InitTextureUVIndices(
705 ::fbxsdk::FbxLayerElement::EMappingMode::eByPolygonVertex, uvType);
706
707 // Add UVs
708 for (const auto& texcoords : character.mesh->texcoords) {
709 // flip y back to fbx convention - refer to reading code
710 lMesh->AddTextureUV(::fbxsdk::FbxVector2(texcoords[0], 1.0f - texcoords[1]), uvType);
711 }
712
713 // Set UV indices for each face. We only have triangles.
714 int faceCount = 0;
715 for (const auto& texcoords : character.mesh->texcoord_faces) {
716 lMesh->SetTextureUVIndex(faceCount, 0, texcoords[0], uvType);
717 lMesh->SetTextureUVIndex(faceCount, 1, texcoords[1], uvType);
718 lMesh->SetTextureUVIndex(faceCount++, 2, texcoords[2], uvType);
719 }
720 }
721
722 // ---------------------------------------------
723 // create the skinning
724 // ---------------------------------------------
725 // Add the mesh skinning
726 // Momentum skinning is saved in two matrices: index and weight (size numvertices x
727 // not-ordered-joints). The index contains the joint index and the weight is the normalized
728 // weight the vertex for that joint.
729
730 MT_THROW_IF(
731 permissive == Permissive::No && !character.skinWeights,
732 " Failed to save the character '{}' to {}. The character has no skinning weights and permissive mode is not enabled. Only mesh-only characters are allowed in permissive mode.",
733 character.name,
734 filename.string());
735
736 if (character.skinWeights != nullptr) {
737 saveSkinWeightsToFbx(character, scene, lMesh, jointToNodeMap);
738 }
739
740 // if we have blendshapes, add them to the mesh
741 if (character.blendShape && character.blendShape->shapeSize() > 0) {
742 saveBlendShapesToFbx(character, scene, lMesh);
743 }
744 // Add the mesh under the root
745 root->AddChild(meshNode);
746 }
747
748 // ---------------------------------------------
749 // add the skeleton to the root
750 // ---------------------------------------------
751 if (!skeletonNodes.empty()) {
752 root->AddChild(skeletonRootNode);
753 }
754
755 // ---------------------------------------------
756 // create animation curves if we have motion
757 // ---------------------------------------------
758 if (jointValues.cols() != 0) {
759 if (jointValues.rows() == character.parameterTransform.numJointParameters()) {
760 createAnimationCurves(
761 character, scene, skeletonNodes, jointValues, framerate, skipActiveJointParamCheck);
762 } else {
763 MT_LOGE(
764 "Rows of joint values {} do not match joint parameter dimension {} so not saving any motion.",
765 jointValues.rows(),
766 character.parameterTransform.numJointParameters());
767 }
768 }
769
770 // ---------------------------------------------
771 // create marker nodes and animation
772 // ---------------------------------------------
773 if (!markerSequence.empty()) {
774 createMarkerNodes(scene, markerSequence, framerate);
775 }
776
777 // ---------------------------------------------
778 // apply namespace prefix to all nodes
779 // ---------------------------------------------
780 if (!namespacePrefix.empty()) {
781 prependNamespaceToAllNodes(scene->GetRootNode(), namespacePrefix);
782 }
783
784 // ---------------------------------------------
785 // close the fbx exporter
786 // ---------------------------------------------
787
788 // finally export the scene
789 lExporter->Export(scene);
790 lExporter->Destroy();
791
792 // destroy the scene and the manager
793 if (scene != nullptr) {
794 scene->Destroy();
795 }
796 manager->Destroy();
797}
