70template <typename T>
71void CharacterStateT<T>::set(
72 const CharacterParameters& inParameters,
73 const CharacterT<T>& referenceCharacter,
74 bool updateMesh,
75 bool updateCollision,
76 bool applyLimits) {
77 parameters = inParameters;
78 if (parameters.offsets.size() == 0) {
79 parameters.offsets = VectorXf::Zero(referenceCharacter.parameterTransform.numJointParameters());
80 }
81 MT_CHECK(
82 parameters.pose.size() == referenceCharacter.parameterTransform.numAllModelParameters(),
83 "{} is not {}",
84 parameters.pose.size(),
85 referenceCharacter.parameterTransform.numAllModelParameters());
86 MT_CHECK(
87 parameters.offsets.size() == referenceCharacter.parameterTransform.numJointParameters(),
88 "{} is not {}",
89 parameters.offsets.size(),
90 referenceCharacter.parameterTransform.numJointParameters());
91
92 // create skeleton state first
93 JointParameters jointParams = referenceCharacter.parameterTransform.apply(parameters);
94 if (applyLimits) {
95 jointParams = applyPassiveJointParameterLimits(referenceCharacter.parameterLimits, jointParams);
96 }
97 skeletonState.set(jointParams, referenceCharacter.skeleton);
98
99 // create locator state
100 locatorState.update(skeletonState, referenceCharacter.locators);
101
102 // create skinning if we have a mesh
103 if (updateMesh && referenceCharacter.mesh && referenceCharacter.skinWeights) {
104 // check if we need to create a mesh
105 if (!meshState || meshState->vertices.size() != referenceCharacter.mesh->vertices.size()) {
106 meshState = std::make_unique<Mesh>(*referenceCharacter.mesh);
107 }
108
109 // check if we have pose blendshapes and use them before skinning if there
110 if (referenceCharacter.poseShapes) {
111 // calculate pose shapes
112 const auto vs = referenceCharacter.poseShapes->compute(skeletonState);
113
114 // apply skinning
115 meshState->vertices = applySSD(
116 referenceCharacter.inverseBindPose, *referenceCharacter.skinWeights, vs, skeletonState);
117
118 // update normals
119 meshState->updateNormals();
120 } else if (referenceCharacter.blendShape) {
121 // For CharacterT<float>, we can cast directly to Character
122 // For CharacterT<double>, we need to cast to Character (CharacterT<float>) first
123 if constexpr (std::is_same_v<T, float>) {
124 skinWithBlendShapes(
125 static_cast<const Character&>(referenceCharacter),
126 skeletonState,
127 parameters.pose,
128 *meshState);
129 } else {
130 // For double, create a temporary Character (CharacterT<float>) object
131 // This is a simplified conversion that may not preserve all properties
132 Character tempCharacter(
133 referenceCharacter.skeleton,
134 referenceCharacter.parameterTransform,
135 referenceCharacter.parameterLimits,
136 referenceCharacter.locators,
137 referenceCharacter.mesh.get(),
138 referenceCharacter.skinWeights.get(),
139 referenceCharacter.collision.get(),
140 referenceCharacter.poseShapes.get(),
141 referenceCharacter.blendShape,
142 nullptr, // No face expression blend shape
143 referenceCharacter.name,
144 referenceCharacter.inverseBindPose);
145
146 skinWithBlendShapes(tempCharacter, skeletonState, parameters.pose, *meshState);
147 }
148 } else {
149 // apply simple skinning
150 applySSD(
151 referenceCharacter.inverseBindPose,
152 *referenceCharacter.skinWeights,
153 *referenceCharacter.mesh,
154 skeletonState,
155 *meshState);
156 }
157 }
158
159 // update collision geometry if present
160 if (updateCollision && referenceCharacter.collision) {
161 if (!collisionState || collisionState->origin.size() != referenceCharacter.collision->size()) {
162 collisionState = std::make_unique<CollisionGeometryState>();
163 }
164 collisionState->update(skeletonState, *referenceCharacter.collision);
165 }
166}
