35[[nodiscard]] CharacterFormat parseCharacterFormat(const filesystem::path& filepath) noexcept {
36 std::string ext = filepath.extension().string();
37 if (ext.empty()) {
38 return CharacterFormat::Unknown;
39 }
40
41 std::transform(ext.begin(), ext.end(), ext.begin(), ::tolower);
42 if ((ext == ".glb") || (ext == ".gltf")) {
43 return CharacterFormat::Gltf;
44 } else if (ext == ".fbx") {
45 return CharacterFormat::Fbx;
46 } else if ((ext == ".usd") || (ext == ".usda") || (ext == ".usdc")) {
47 return CharacterFormat::Usd;
48 } else {
49 return CharacterFormat::Unknown;
50 }
51}

53[[nodiscard]] std::optional<Character> loadCharacterByFormat(
54 const CharacterFormat format,

91Character loadFullCharacter(
92 const std::string& characterPath,
93 const std::string& parametersPath,
94 const std::string& locatorsPath) {
95 // Parse format
96 const auto format = parseCharacterFormat(characterPath);
97 MT_THROW_IF(
98 format == CharacterFormat::Unknown, "Unknown character format for path: {}", characterPath);
99
100 // Load character
101 auto character = loadCharacterByFormat(format, characterPath);
102 MT_THROW_IF(!character, "Failed to load buffered character");
103
104 // load parameter transform
105 if (!parametersPath.empty()) {
106 auto def = loadMomentumModel(parametersPath);
107 loadParameters(def, *character);
108 }
109
110 // Load locators
111 if (!locatorsPath.empty()) {
112 (*character).locators =
113 loadLocators(locatorsPath, (*character).skeleton, (*character).parameterTransform);
114 }
115
116 return character.value();
117}

143void saveCharacter(
144 const filesystem::path& filename,
145 const Character& character,
146 const float fps,
147 const MatrixXf& motion,
148 std::span<const std::vector<Marker>> markerSequence,
149 const FileSaveOptions& options) {
150 // Parse format from file extension
151 const auto format = parseCharacterFormat(filename);
152 MT_THROW_IF(
153 format == CharacterFormat::Unknown,
154 "Unknown character format for path: {}. Supported formats: .fbx, .glb, .gltf",
155 filename.string());
156
157 if (format == CharacterFormat::Gltf) {
158 saveGltfCharacter(
159 filename,
160 character,
161 fps,
162 {character.parameterTransform.name, motion},
163 {},
164 markerSequence,
165 options);
166 } else if (format == CharacterFormat::Fbx) {
167 // Save as FBX
168 saveFbx(
169 filename, character, motion, VectorXf(), static_cast<double>(fps), markerSequence, options);
170 } else {
171 MT_THROW(
172 "{} is not a supported format. Supported formats: .fbx, .glb, .gltf", filename.string());
173 }
