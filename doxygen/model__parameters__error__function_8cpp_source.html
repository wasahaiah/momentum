42template <typename T>
43double ModelParametersErrorFunctionT<T>::getError(
44 const ModelParametersT<T>& parameters,
45 const SkeletonStateT<T>& /* state */,
46 const MeshStateT<T>& /* meshState */) {
47 // ignore if we don't have any reasonable data
48 if (targetParameters_.size() != parameters.size() || targetWeights_.size() != parameters.size()) {
49 return 0.0;
50 }
51
52 // calculate difference between parameters and desired parameters
53 double error = 0;
54 for (Eigen::Index i = 0; i < parameters.size(); ++i) {
55 if (this->enabledParameters_.test(i)) {
56 const auto pdiff = targetWeights_(i) * (parameters(i) - targetParameters_(i));
57 error += pdiff * pdiff;
58 }
59 }
60
61 // return error
62 return error * this->weight_ * kMotionWeight;
63}

65template <typename T>
66double ModelParametersErrorFunctionT<T>::getGradient(
67 const ModelParametersT<T>& parameters,
68 const SkeletonStateT<T>& /* state */,
69 const MeshStateT<T>& /* meshState */,
70 Ref<Eigen::VectorX<T>> gradient) {
71 // ignore if we don't have any reasonable data
72 if (targetParameters_.size() != parameters.size() || targetWeights_.size() != parameters.size()) {
73 return 0.0;
74 }
75
76 double error = 0;
77 for (Eigen::Index i = 0; i < parameters.size(); ++i) {
78 if (this->enabledParameters_.test(i)) {
79 const auto pdiff = targetWeights_(i) * (parameters(i) - targetParameters_(i));
80 gradient(i) += 2.0f * targetWeights_(i) * pdiff * this->weight_ * kMotionWeight;
81 error += pdiff * pdiff;
82 }
83 }
84
85 // return error

94template <typename T>
95double ModelParametersErrorFunctionT<T>::getJacobian(
96 const ModelParametersT<T>& parameters,
97 const SkeletonStateT<T>& /* state */,
98 const MeshStateT<T>& /* meshState */,
99 Ref<Eigen::MatrixX<T>> jacobian,
100 Ref<Eigen::VectorX<T>> residual,
101 int& usedRows) {
102 MT_PROFILE_FUNCTION();
103
104 usedRows = 0;
105
106 // ignore if we don't have any reasonable data
107 if (targetParameters_.size() != parameters.size() || targetWeights_.size() != parameters.size()) {
108 return 0.0;
109 }
110
111 const float sWeight = std::sqrt(this->weight_ * kMotionWeight);
112
113 Eigen::Index out = 0;
114
115 // calculate difference between parameters and desired parameters
116 double error = 0;
117 for (Eigen::Index i = 0; i < parameters.size(); ++i) {
118 if (this->enabledParameters_.test(i) && targetWeights_(i) > 0) {
119 const auto pdiff = targetWeights_(i) * (parameters(i) - targetParameters_(i));
120 error += pdiff * pdiff;
121 residual(out) = pdiff * sWeight;
122 jacobian(out, i) = sWeight * targetWeights_(i);
123 ++out;
124 }
125 }
126
127 usedRows = gsl::narrow_cast<int>(out);
128
129 // return error
130 return error * this->weight_ * kMotionWeight;
131}
