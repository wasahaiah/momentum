70template <typename T>
71Eigen::Vector3<T> computeTargetBaryPosition(
72 const MeshT<T>& mesh,
73 const PointTriangleVertexConstraintT<T>& c) {
74 return c.tgtTriangleBaryCoords[0] * mesh.vertices[c.tgtTriangleIndices[0]] +
75 c.tgtTriangleBaryCoords[1] * mesh.vertices[c.tgtTriangleIndices[1]] +
76 c.tgtTriangleBaryCoords[2] * mesh.vertices[c.tgtTriangleIndices[2]];
77}

79template <typename T>
80Eigen::Vector3<T> computeTargetTriangleNormal(
81 const MeshT<T>& mesh,
82 const PointTriangleVertexConstraintT<T>& c) {
83 return (mesh.vertices[c.tgtTriangleIndices[1]] - mesh.vertices[c.tgtTriangleIndices[0]])
84 .cross(mesh.vertices[c.tgtTriangleIndices[2]] - mesh.vertices[c.tgtTriangleIndices[0]])
85 .normalized();
86}

88template <typename T>
89Eigen::Vector3<T> computeTargetPosition(
90 const MeshT<T>& mesh,
91 const PointTriangleVertexConstraintT<T>& c) {
92 return computeTargetBaryPosition(mesh, c) + c.depth * computeTargetTriangleNormal(mesh, c);
93}

95template <typename T>
96double PointTriangleVertexErrorFunctionT<T>::getError(
97 const ModelParametersT<T>& modelParameters,
98 const SkeletonStateT<T>& state,
99 const MeshStateT<T>& meshState) {
100 MT_PROFILE_FUNCTION();
101
102 MT_CHECK_NOTNULL(meshState.posedMesh_);
103
104 // loop over all constraints and calculate the error
105 double error = 0.0;
106
107 if (constraintType_ == VertexConstraintType::Position) {
108 for (size_t i = 0; i < constraints_.size(); ++i) {
109 const PointTriangleVertexConstraintT<T>& constr = constraints_[i];
110
111 const Eigen::Vector3<T> srcPoint = meshState.posedMesh_->vertices[constr.srcVertexIndex];
112 const Eigen::Vector3<T> tgtPoint = computeTargetPosition(*meshState.posedMesh_, constr);
113 const Eigen::Vector3<T> diff = srcPoint - tgtPoint;
114 error += constr.weight * diff.squaredNorm() * kPositionWeight;
115 }
116 } else {
117 const auto [sourceNormalWeight, targetNormalWeight] = computeNormalWeights();
118 for (size_t i = 0; i < constraints_.size(); ++i) {
119 const PointTriangleVertexConstraintT<T>& constr = constraints_[i];
120
121 const Eigen::Vector3<T> srcPoint = meshState.posedMesh_->vertices[constr.srcVertexIndex];
122
123 const Eigen::Vector3<T> sourceNormal = meshState.posedMesh_->normals[constr.srcVertexIndex];
124 Eigen::Vector3<T> targetNormal = computeTargetTriangleNormal(*meshState.posedMesh_, constr);
125
126 const Eigen::Vector3<T> tgtPoint =
127 computeTargetBaryPosition(*meshState.posedMesh_, constr) + constr.depth * targetNormal;
128
129 if (sourceNormal.dot(targetNormal) < 0) {
130 targetNormal *= -1;
131 }
132 const Eigen::Vector3<T> normal =
133 sourceNormalWeight * sourceNormal + targetNormalWeight * targetNormal;
134
135 // calculate point->plane distance and error
136 const Eigen::Vector3<T> diff = srcPoint - tgtPoint;
137 const T dist = normal.dot(diff);
138 error += constr.weight * dist * dist * kPlaneWeight;
139 }
140 }
141
142 // return error
143 return error * this->weight_;
144}

146template <typename T>
147void gradient_jointParams_to_modelParams(
148 const T grad_jointParams,
149 const Eigen::Index iJointParam,
150 const ParameterTransform& parameterTransform,
151 Eigen::Ref<Eigen::VectorX<T>> gradient) {
152 // explicitly multiply with the parameter transform to generate parameter space gradients
153 for (auto index = parameterTransform.transform.outerIndexPtr()[iJointParam];
154 index < parameterTransform.transform.outerIndexPtr()[iJointParam + 1];
155 ++index) {
156 gradient[parameterTransform.transform.innerIndexPtr()[index]] +=
157 grad_jointParams * parameterTransform.transform.valuePtr()[index];
158 }
159}

191template <typename T>
192Eigen::Vector3<T> calculateDWorldPos(
193 const momentum::Character& character,
194 const SkeletonStateT<T>& state,
195 const int vertexIndex,
196 const Eigen::Vector3<T>& d_restPos) {
197 const auto& skinWeights = *character.skinWeights;
198
199 Eigen::Vector3<T> d_worldPos = Eigen::Vector3<T>::Zero();
200 for (uint32_t i = 0; i < kMaxSkinJoints; ++i) {
201 const auto w = skinWeights.weight(vertexIndex, i);
202 const auto parentBone = skinWeights.index(vertexIndex, i);
203 if (w > 0) {
204 d_worldPos += w *
205 (state.jointState[parentBone].transform.toLinear() *
206 (character.inverseBindPose[parentBone].linear().template cast<T>() * d_restPos));
207 }
208 }
209
210 return d_worldPos;
211}

213template <typename T>
214double PointTriangleVertexErrorFunctionT<T>::calculatePositionGradient(
215 const ModelParametersT<T>& modelParameters,
216 const SkeletonStateT<T>& state,
217 const MeshStateT<T>& meshState,
218 const PointTriangleVertexConstraintT<T>& constr,
219 Eigen::Ref<Eigen::VectorX<T>> gradient) const {
220 const Eigen::Vector3<T> srcPos = meshState.posedMesh_->vertices[constr.srcVertexIndex];
221 const Eigen::Vector3<T> tgtPos = computeTargetPosition(*meshState.posedMesh_, constr);
222
223 const Eigen::Vector3<T> diff = srcPos - tgtPos;
224
225 // calculate the difference between target and position and error
226 const T wgt = constr.weight * 2.0f * kPositionWeight * this->weight_;
227
228 const auto d_targetPos_d_tgtTriVertexPos =
229 compute_d_targetPos_d_vertexPos(constr, *meshState.posedMesh_);
230
231 // To simplify things we'll have one loop that goes through all 3 target triangle vertices _and_
232 // the source vertex.
233 // iter 0-3: target triangle vertices
234 // iter 4: source vertex
235 for (int iTriVert = 0; iTriVert < 4; ++iTriVert) {
236 const auto vertexIndex =
237 iTriVert < 3 ? constr.tgtTriangleIndices[iTriVert] : constr.srcVertexIndex;
238 // Flip sign for the target vertex since the actual error is (srcPoint - tgtPoint)
239 const Eigen::Matrix3<T> d_targetPos_d_vertexIndex = iTriVert < 3
240 ? Eigen::Matrix3<T>(-d_targetPos_d_tgtTriVertexPos[iTriVert])
241 : Eigen::Matrix3<T>::Identity();
242
243 SkinningWeightIteratorT<T> skinningIter(
244 this->character_, *meshState.restMesh_, state, vertexIndex);
245
246 // IN handle derivatives wrt jointParameters
247 while (!skinningIter.finished()) {
248 size_t jointIndex = 0;
249 T boneWeight;
250 Eigen::Vector3<T> pos;
251 std::tie(jointIndex, boneWeight, pos) = skinningIter.next();
252
253 // check for valid index
254 MT_CHECK(jointIndex < this->skeleton_.joints.size());
255
256 const auto& jointState = state.jointState[jointIndex];
257 const size_t paramIndex = jointIndex * kParametersPerJoint;
258 const Eigen::Vector3<T> posd = pos - jointState.translation();
259
260 // calculate derivatives based on active joints
261 for (size_t d = 0; d < 3; d++) {
262 if (this->activeJointParams_[paramIndex + d]) {
263 // Gradient wrt translation:
264 gradient_jointParams_to_modelParams(
265 boneWeight *
266 diff.dot(d_targetPos_d_vertexIndex * jointState.getTranslationDerivative(d)) *
267 wgt,
268 paramIndex + d,
269 this->parameterTransform_,
270 gradient);
271 }
272 if (this->activeJointParams_[paramIndex + 3 + d]) {
273 // Gradient wrt rotation:
274 gradient_jointParams_to_modelParams(
275 boneWeight *
276 diff.dot(d_targetPos_d_vertexIndex * jointState.getRotationDerivative(d, posd)) *
277 wgt,
278 paramIndex + 3 + d,
279 this->parameterTransform_,
280 gradient);
281 }
282 }
283 if (this->activeJointParams_[paramIndex + 6]) {
284 // Gradient wrt scale:
285 gradient_jointParams_to_modelParams(
286 boneWeight * diff.dot(d_targetPos_d_vertexIndex * jointState.getScaleDerivative(posd)) *
287 wgt,
288 paramIndex + 6,
289 this->parameterTransform_,
290 gradient);
291 }
292 } // OUT handle derivatives wrt jointParameters
293
294 // IN handle derivatives wrt blend shape parameters
295 if (this->character_.blendShape) {
296 for (Eigen::Index iBlendShape = 0;
297 iBlendShape < this->parameterTransform_.blendShapeParameters.size();
298 ++iBlendShape) {
299 const auto paramIdx = this->parameterTransform_.blendShapeParameters[iBlendShape];
300 if (paramIdx < 0) {
301 continue;
302 }
303
304 const Eigen::Vector3<T> d_restPos =
305 this->character_.blendShape->getShapeVectors()
306 .template block<3, 1>(3 * vertexIndex, iBlendShape, 3, 1)
307 .template cast<T>();
308 Eigen::Vector3<T> d_worldPos =
309 calculateDWorldPos(character_, state, vertexIndex, d_restPos);
310
311 gradient[paramIdx] += wgt * diff.dot(d_targetPos_d_vertexIndex * d_worldPos);
312 }
313 }
314 // OUT handle derivatives wrt blend shape parameters
315
316 // IN handle derivatives wrt face expression blend shape parameters
317 if (this->character_.faceExpressionBlendShape) {
318 for (Eigen::Index iBlendShape = 0;
319 iBlendShape < this->parameterTransform_.faceExpressionParameters.size();
320 ++iBlendShape) {
321 const auto paramIdx = this->parameterTransform_.faceExpressionParameters[iBlendShape];
322 if (paramIdx < 0) {
323 continue;
324 }
325
326 const Eigen::Vector3<T> d_restPos =
327 this->character_.faceExpressionBlendShape->getShapeVectors()
328 .template block<3, 1>(3 * vertexIndex, iBlendShape, 3, 1)
329 .template cast<T>();
330 Eigen::Vector3<T> d_worldPos =
331 calculateDWorldPos(character_, state, vertexIndex, d_restPos);
332
333 gradient[paramIdx] += wgt * diff.dot(d_targetPos_d_vertexIndex * d_worldPos);
334 }
335 }
336 // OUT handle derivatives wrt face expression blend shape parameters
337 }
338
339 return constr.weight * diff.squaredNorm() * kPositionWeight;
340}

342template <typename T>
343std::array<Eigen::Matrix3<T>, 3> compute_d_targetNormal_d_vertexPos(
344 const PointTriangleVertexConstraintT<T>& cons,
345 const MeshT<T>& mesh) {
346 std::array<Eigen::Matrix3<T>, 3> result;
347 for (auto& r : result) {
348 r.setZero();
349 }
350
351 const std::array<Eigen::Vector3<T>, 3> tgtTrianglePositions = {
352 mesh.vertices[cons.tgtTriangleIndices[0]],
353 mesh.vertices[cons.tgtTriangleIndices[1]],
354 mesh.vertices[cons.tgtTriangleIndices[2]]};
355
356 Eigen::Vector3<T> n = (tgtTrianglePositions[1] - tgtTrianglePositions[0])
357 .cross(tgtTrianglePositions[2] - tgtTrianglePositions[0]);
358 T n_norm = n.norm();
359 if (n_norm < 1e-6) {
360 // ignore normal gradients
361 return result;
362 }
363 n /= n_norm;
364
365 const T area_times_2 = n_norm;
366
367 for (int k = 0; k < 3; ++k) {
368 // https://www.cs.cmu.edu/~kmcrane/Projects/Other/TriangleMeshDerivativesCheatSheet.pdf
369 const Eigen::Vector3<T> e =
370 tgtTrianglePositions[(k + 2) % 3] - tgtTrianglePositions[(k + 1) % 3];
371 result[k] += (e.cross(n) * n.transpose()) / area_times_2;
372 }
373
374 return result;
375}

377template <typename T>
378std::array<Eigen::Matrix3<T>, 3> compute_d_targetPos_d_vertexPos(
379 const PointTriangleVertexConstraintT<T>& cons,
380 const MeshT<T>& mesh) {
381 std::array<Eigen::Matrix3<T>, 3> result;
382 result[0] = Eigen::Matrix3<T>::Identity() * cons.tgtTriangleBaryCoords[0];
383 result[1] = Eigen::Matrix3<T>::Identity() * cons.tgtTriangleBaryCoords[1];
384 result[2] = Eigen::Matrix3<T>::Identity() * cons.tgtTriangleBaryCoords[2];
385
386 const auto dNormal = compute_d_targetNormal_d_vertexPos(cons, mesh);
387 for (int k = 0; k < 3; ++k) {
388 result[k] += cons.depth * dNormal[k];
389 }
390
391 return result;
392}

394template <typename T>
395double PointTriangleVertexErrorFunctionT<T>::calculatePositionJacobian(
396 const ModelParametersT<T>& modelParameters,
397 const SkeletonStateT<T>& state,
398 const MeshStateT<T>& meshState,
399 const PointTriangleVertexConstraintT<T>& constr,
400 Ref<Eigen::MatrixX<T>> jac,
401 Ref<Eigen::VectorX<T>> res) const {
402 // calculate the difference between target and position and error
403 const Eigen::Vector3<T> srcPos = meshState.posedMesh_->vertices[constr.srcVertexIndex];
404 const Eigen::Vector3<T> tgtPos = computeTargetPosition(*meshState.posedMesh_, constr);
405
406 // The derivative of the target position wrt the triangle vertices:
407 // p_tgt = bary_0 * p_0 + bary_1 * p_1 + bary_2 * p_2 + depth * n
408
409 const Eigen::Vector3<T> diff = meshState.posedMesh_->vertices[constr.srcVertexIndex] - tgtPos;
410
411 const T wgt = std::sqrt(constr.weight * kPositionWeight * this->weight_);
412
413 const auto d_targetPos_d_tgtTriVertexPos =
414 compute_d_targetPos_d_vertexPos(constr, *meshState.posedMesh_);
415
416// Verify derivative:
417#if 0
418 {
419 const int kTriVert = 0;
420 auto meshTmp = *this->posedMesh_;
421 const float eps = 1e-5;
422 const Eigen::Vector3<T> srcPosOrig = meshTmp.vertices[constr.tgtTriangleIndices[kTriVert]];
423 Eigen::Matrix3<T> diff_est = Eigen::Matrix3<T>::Zero();
424 for (int k = 0; k < 3; ++k) {
425 meshTmp.vertices[constr.tgtTriangleIndices[kTriVert]] =
426 srcPosOrig + eps * Eigen::Vector3<T>::Unit(k);
427 const Eigen::Vector3<T> tgtPosPlus = computeTargetPosition(meshTmp, constr);
428 meshTmp.vertices[constr.tgtTriangleIndices[kTriVert]] = srcPosOrig;
429
430 const Eigen::Vector3<T> tgtPosPlusDiff = (tgtPosPlus - tgtPos) / eps;
431 diff_est.col(k) = tgtPosPlusDiff;
432 }
433
434 const T derivativeMismatch = (d_targetPos_d_srcPos[kTriVert] - diff_est).norm();
435 if (derivativeMismatch > 1e-6) {
436 MT_LOGI("Target position derivative mismatch detected: {}", derivativeMismatch);
437 MT_LOGI(" d_targetPos_d_srcPos (computed): \n{}", d_targetPos_d_srcPos[kTriVert]);
438 MT_LOGI(" d_targetPos_d_srcPos (estimated): \n{}", diff_est);
439 }
440 }
441#endif
442
443 // To simplify things we'll have one loop that goes through all 3 target triangle vertices _and_
444 // the source vertex.
445 // iter 0-3: target triangle vertices
446 // iter 4: source vertex
447 for (int iTriVert = 0; iTriVert < 4; ++iTriVert) {
448 const auto vertexIndex =
449 iTriVert < 3 ? constr.tgtTriangleIndices[iTriVert] : constr.srcVertexIndex;
450 // Flip sign for the target vertex since the actual error is (srcPoint - tgtPoint)
451 const Eigen::Matrix3<T> d_targetPos_d_vertexPos = iTriVert < 3
452 ? Eigen::Matrix3<T>(-d_targetPos_d_tgtTriVertexPos[iTriVert])
453 : Eigen::Matrix3<T>::Identity();
454
455 SkinningWeightIteratorT<T> skinningIter(
456 this->character_, *meshState.restMesh_, state, vertexIndex);
457
458 // IN handle derivatives wrt jointParameters
459 while (!skinningIter.finished()) {
460 size_t jointIndex = 0;
461 T boneWeight;
462 Eigen::Vector3<T> pos;
463 std::tie(jointIndex, boneWeight, pos) = skinningIter.next();
464
465 // check for valid index
466 MT_CHECK(jointIndex < this->skeleton_.joints.size());
467
468 const auto& jointState = state.jointState[jointIndex];
469 const size_t paramIndex = jointIndex * kParametersPerJoint;
470 const Eigen::Vector3<T> posd = pos - jointState.translation();
471
472 // calculate derivatives based on active joints
473 for (size_t d = 0; d < 3; d++) {
474 if (this->activeJointParams_[paramIndex + d]) {
475 jacobian_jointParams_to_modelParams<T>(
476 boneWeight * wgt * d_targetPos_d_vertexPos * jointState.getTranslationDerivative(d),
477 paramIndex + d,
478 this->parameterTransform_,
479 jac);
480 }
481 if (this->activeJointParams_[paramIndex + 3 + d]) {
482 jacobian_jointParams_to_modelParams<T>(
483 boneWeight * wgt * d_targetPos_d_vertexPos *
484 jointState.getRotationDerivative(d, posd),
485 paramIndex + d + 3,
486 this->parameterTransform_,
487 jac);
488 }
489 }
490 if (this->activeJointParams_[paramIndex + 6]) {
491 jacobian_jointParams_to_modelParams<T>(
492 boneWeight * wgt * d_targetPos_d_vertexPos * jointState.getScaleDerivative(posd),
493 paramIndex + 6,
494 this->parameterTransform_,
495 jac);
496 }
497 }
498 // OUT handle derivatives wrt jointParameters
499
500 // IN handle derivatives wrt blend shape parameters
501 if (this->character_.blendShape) {
502 for (Eigen::Index iBlendShape = 0;
503 iBlendShape < this->parameterTransform_.blendShapeParameters.size();
504 ++iBlendShape) {
505 const auto paramIdx = this->parameterTransform_.blendShapeParameters[iBlendShape];
506 if (paramIdx < 0) {
507 continue;
508 }
509
510 const Eigen::Vector3<T> d_restPos =
511 this->character_.blendShape->getShapeVectors()
512 .template block<3, 1>(3 * vertexIndex, iBlendShape, 3, 1)
513 .template cast<T>();
514 Eigen::Vector3<T> d_worldPos =
515 calculateDWorldPos(this->character_, state, vertexIndex, d_restPos);
516
517 jac.col(paramIdx) += wgt * d_targetPos_d_vertexPos * d_worldPos;
518 }
519 }
520 // OUT handle derivatives wrt blend shape parameters
521
522 // IN handle derivatives wrt face expression blend shape parameters
523 if (this->character_.faceExpressionBlendShape) {
524 for (Eigen::Index iBlendShape = 0;
525 iBlendShape < this->parameterTransform_.faceExpressionParameters.size();
526 ++iBlendShape) {
527 const auto paramIdx = this->parameterTransform_.faceExpressionParameters[iBlendShape];
528 if (paramIdx < 0) {
529 continue;
530 }
531
532 const Eigen::Vector3<T> d_restPos =
533 this->character_.faceExpressionBlendShape->getShapeVectors()
534 .template block<3, 1>(3 * vertexIndex, iBlendShape, 3, 1)
535 .template cast<T>();
536 Eigen::Vector3<T> d_worldPos =
537 calculateDWorldPos(this->character_, state, vertexIndex, d_restPos);
538
539 jac.col(paramIdx) += wgt * d_targetPos_d_vertexPos * d_worldPos;
540 }
541 }
542 }
543 // OUT handle derivatives wrt face expression blend shape parameters
544 res = diff * wgt;
545
546 return wgt * wgt * diff.squaredNorm();
547}

549template <typename T>
550double PointTriangleVertexErrorFunctionT<T>::calculateNormalGradient(
551 const ModelParametersT<T>& modelParameters,
552 const SkeletonStateT<T>& state,
553 const MeshStateT<T>& meshState,
554 const PointTriangleVertexConstraintT<T>& constr,
555 const T sourceNormalWeight,
556 const T targetNormalWeight,
557 Eigen::Ref<Eigen::VectorX<T>> gradient) const {
558 const Eigen::Vector3<T> sourceNormal = meshState.posedMesh_->normals[constr.srcVertexIndex];
559 Eigen::Vector3<T> targetNormal = computeTargetTriangleNormal(*meshState.posedMesh_, constr);
560 const Eigen::Vector3<T> tgtPosition = computeTargetPosition(*meshState.posedMesh_, constr);
561
562 T targetNormalSign = 1;
563 if (sourceNormal.dot(targetNormal) < 0) {
564 targetNormal *= -1;
565 targetNormalSign = -1;
566 }
567 const Eigen::Vector3<T> normal =
568 sourceNormalWeight * sourceNormal + targetNormalWeight * targetNormal;
569
570 // calculate the difference between target and position and error
571 const Eigen::Vector3<T> diff =
572 meshState.posedMesh_->vertices[constr.srcVertexIndex] - tgtPosition;
573 const T dist = diff.dot(normal);
574
575 const T wgt = constr.weight * 2.0f * kPlaneWeight * this->weight_;
576
577 const auto d_targetPos_d_tgtTriVertexPos =
578 compute_d_targetPos_d_vertexPos(constr, *meshState.posedMesh_);
579 const auto d_targetNormal_d_tgtTriVertexPos =
580 compute_d_targetNormal_d_vertexPos(constr, *meshState.posedMesh_);
581
582 for (int iTriVert = 0; iTriVert < 4; ++iTriVert) {
583 const auto vertexIndex =
584 iTriVert < 3 ? constr.tgtTriangleIndices[iTriVert] : constr.srcVertexIndex;
585 // Flip sign for the target vertex since the actual error is (srcPoint - tgtPoint)
586 const Eigen::Matrix3<T> d_targetPos_d_vertexPos = iTriVert < 3
587 ? Eigen::Matrix3<T>(-d_targetPos_d_tgtTriVertexPos[iTriVert])
588 : Eigen::Matrix3<T>::Identity();
589
590 SkinningWeightIteratorT<T> skinningIter(
591 this->character_, *meshState.restMesh_, state, vertexIndex);
592
593 // IN handle derivatives wrt jointParameters
594 while (!skinningIter.finished()) {
595 const auto [jointIndex, boneWeight, pos] = skinningIter.next();
596
597 // check for valid index
598 MT_CHECK(jointIndex < this->skeleton_.joints.size());
599
600 const auto& jointState = state.jointState[jointIndex];
601 const size_t paramIndex = jointIndex * kParametersPerJoint;
602 const Eigen::Vector3<T> posd = pos - jointState.translation();
603
604 // calculate derivatives based on active joints
605 for (size_t d = 0; d < 3; d++) {
606 if (this->activeJointParams_[paramIndex + d]) {
607 // Gradient wrt translation:
608 gradient_jointParams_to_modelParams<T>(
609 boneWeight * dist *
610 normal.dot(d_targetPos_d_vertexPos * jointState.getTranslationDerivative(d)) *
611 wgt,
612 paramIndex + d,
613 this->parameterTransform_,
614 gradient);
615 }
616 if (this->activeJointParams_[paramIndex + 3 + d]) {
617 // Error is (p_src - p_tgt) . (w_src * n_src + w_tgt * n_tgt)
618 // Full derivative will be:
619 // (d/dTheta (p_src - p_tgt)) . (w_src * n_src + w_tgt * n_tgt) +
620 // (p_src - p_tgt) . (w_src * dn_src/dTheta) +
621 // (p_src - p_tgt) . (w_tgt * dn_tgt/dTheta))
622
623 // Jacobian wrt rotation:
624 T diff_total =
625 normal.dot(d_targetPos_d_vertexPos * jointState.getRotationDerivative(d, posd));
626
627 if (iTriVert < 3) {
628 const Eigen::Vector3<T> diff_tgt_normal = targetNormalSign *
629 d_targetNormal_d_tgtTriVertexPos[iTriVert] *
630 jointState.getRotationDerivative(d, posd);
631 diff_total += targetNormalWeight * diff.dot(diff_tgt_normal);
632 } else {
633 // Jacobian wrt rotation:
634 const Eigen::Vector3<T> diff_src_normal =
635 jointState.getRotationDerivative(d, sourceNormal);
636 diff_total += sourceNormalWeight * diff.dot(diff_src_normal);
637 }
638
639 gradient_jointParams_to_modelParams<T>(
640 boneWeight * wgt * diff_total * dist,
641 paramIndex + d + 3,
642 this->parameterTransform_,
643 gradient);
644 }
645 }
646 if (this->activeJointParams_[paramIndex + 6]) {
647 // Jacobian wrt scale:
648 gradient_jointParams_to_modelParams<T>(
649 boneWeight * wgt *
650 normal.dot(d_targetPos_d_vertexPos * jointState.getScaleDerivative(posd)) * dist,
651 paramIndex + 6,
652 this->parameterTransform_,
653 gradient);
654 }
655 }
656 // OUT handle derivatives wrt jointParameters
657
658 // IN handle derivatives wrt blend shape parameters:
659 for (Eigen::Index iBlendShape = 0;
660 iBlendShape < this->parameterTransform_.blendShapeParameters.size();
661 ++iBlendShape) {
662 const auto paramIdx = this->parameterTransform_.blendShapeParameters[iBlendShape];
663 if (paramIdx < 0) {
664 continue;
665 }
666
667 const Eigen::Vector3<T> d_restPos =
668 this->character_.blendShape->getShapeVectors()
669 .template block<3, 1>(3 * vertexIndex, iBlendShape, 3, 1)
670 .template cast<T>();
671 Eigen::Vector3<T> d_worldPos = calculateDWorldPos(character_, state, vertexIndex, d_restPos);
672
673 gradient(paramIdx) += wgt * dist * normal.dot(d_targetPos_d_vertexPos * d_worldPos);
674 if (iTriVert < 3) {
675 const Eigen::Vector3<T> diff_tgt_normal =
676 d_targetNormal_d_tgtTriVertexPos[iTriVert] * d_worldPos;
677 gradient(paramIdx) += targetNormalSign * wgt * dist * diff.dot(diff_tgt_normal);
678 }
679 }
680 // OUT handle derivatives wrt blend shape parameters:
681 }
682
683 return constr.weight * dist * dist * kPlaneWeight;
684}

686template <typename T>
687double PointTriangleVertexErrorFunctionT<T>::calculateNormalJacobian(
688 const ModelParametersT<T>& modelParameters,
689 const SkeletonStateT<T>& state,
690 const MeshStateT<T>& meshState,
691 const PointTriangleVertexConstraintT<T>& constr,
692 const T sourceNormalWeight,
693 const T targetNormalWeight,
694 Ref<Eigen::MatrixX<T>> jac,
695 T& res) const {
696 const Eigen::Vector3<T> sourceNormal = meshState.posedMesh_->normals[constr.srcVertexIndex];
697 Eigen::Vector3<T> targetNormal = computeTargetTriangleNormal(*meshState.posedMesh_, constr);
698 const Eigen::Vector3<T> tgtPosition = computeTargetPosition(*meshState.posedMesh_, constr);
699
700 T targetNormalSign = 1;
701 if (sourceNormal.dot(targetNormal) < 0) {
702 targetNormal *= -1;
703 targetNormalSign = -1;
704 }
705 const Eigen::Vector3<T> normal =
706 sourceNormalWeight * sourceNormal + targetNormalWeight * targetNormal;
707
708 // calculate the difference between target and position and error
709 const Eigen::Vector3<T> diff =
710 meshState.posedMesh_->vertices[constr.srcVertexIndex] - tgtPosition;
711 const T dist = diff.dot(normal);
712
713 const T wgt = std::sqrt(constr.weight * kPositionWeight * this->weight_);
714
715 const auto d_targetPos_d_tgtTriVertexPos =
716 compute_d_targetPos_d_vertexPos(constr, *meshState.posedMesh_);
717 const auto d_targetNormal_d_tgtTriVertexPos =
718 compute_d_targetNormal_d_vertexPos(constr, *meshState.posedMesh_);
719
720 for (int iTriVert = 0; iTriVert < 4; ++iTriVert) {
721 const auto vertexIndex =
722 iTriVert < 3 ? constr.tgtTriangleIndices[iTriVert] : constr.srcVertexIndex;
723 // Flip sign for the target vertex since the actual error is (srcPoint - tgtPoint)
724 const Eigen::Matrix3<T> d_targetPos_d_vertexPos = iTriVert < 3
725 ? Eigen::Matrix3<T>(-d_targetPos_d_tgtTriVertexPos[iTriVert])
726 : Eigen::Matrix3<T>::Identity();
727
728 SkinningWeightIteratorT<T> skinningIter(
729 this->character_, *meshState.restMesh_, state, vertexIndex);
730
731 // IN handle derivatives wrt jointParameters
732 while (!skinningIter.finished()) {
733 const auto [jointIndex, boneWeight, pos] = skinningIter.next();
734
735 // check for valid index
736 MT_CHECK(jointIndex < this->skeleton_.joints.size());
737
738 const auto& jointState = state.jointState[jointIndex];
739 const size_t paramIndex = jointIndex * kParametersPerJoint;
740 const Eigen::Vector3<T> posd = pos - jointState.translation();
741
742 // calculate derivatives based on active joints
743 for (size_t d = 0; d < 3; d++) {
744 if (this->activeJointParams_[paramIndex + d]) {
745 // Jacobian wrt translation:
746 jacobian_jointParams_to_modelParams<T>(
747 boneWeight * wgt *
748 normal.dot(d_targetPos_d_vertexPos * jointState.getTranslationDerivative(d)),
749 paramIndex + d,
750 this->parameterTransform_,
751 jac);
752 }
753 if (this->activeJointParams_[paramIndex + 3 + d]) {
754 // Error is (p_src - p_tgt) . (w_src * n_src + w_tgt * n_tgt)
755 // Full derivative will be:
756 // (d/dTheta (p_src - p_tgt)) . (w_src * n_src + w_tgt * n_tgt) +
757 // (p_src - p_tgt) . (w_src * dn_src/dTheta) +
758 // (p_src - p_tgt) . (w_tgt * dn_tgt/dTheta))
759
760 // Jacobian wrt rotation:
761 T diff_total =
762 normal.dot(d_targetPos_d_vertexPos * jointState.getRotationDerivative(d, posd));
763
764 if (iTriVert < 3) {
765 const Eigen::Vector3<T> diff_tgt_normal = targetNormalSign *
766 d_targetNormal_d_tgtTriVertexPos[iTriVert] *
767 jointState.getRotationDerivative(d, posd);
768 diff_total += targetNormalWeight * diff.dot(diff_tgt_normal);
769 } else {
770 // Jacobian wrt rotation:
771 const Eigen::Vector3<T> diff_src_normal =
772 jointState.getRotationDerivative(d, sourceNormal);
773 diff_total += sourceNormalWeight * diff.dot(diff_src_normal);
774 }
775
776 jacobian_jointParams_to_modelParams<T>(
777 boneWeight * wgt * diff_total, paramIndex + d + 3, this->parameterTransform_, jac);
778 }
779 }
780 if (this->activeJointParams_[paramIndex + 6]) {
781 // Jacobian wrt scale:
782 jacobian_jointParams_to_modelParams<T>(
783 boneWeight * wgt *
784 normal.dot(d_targetPos_d_vertexPos * jointState.getScaleDerivative(posd)),
785 paramIndex + 6,
786 this->parameterTransform_,
787 jac);
788 }
789 }
790 // OUT handle derivatives wrt jointParameters
791
792 // IN handle derivatives wrt blend shape parameters:
793 for (Eigen::Index iBlendShape = 0;
794 iBlendShape < this->parameterTransform_.blendShapeParameters.size();
795 ++iBlendShape) {
796 const auto paramIdx = this->parameterTransform_.blendShapeParameters[iBlendShape];
797 if (paramIdx < 0) {
798 continue;
799 }
800
801 const Eigen::Vector3<T> d_restPos =
802 this->character_.blendShape->getShapeVectors()
803 .template block<3, 1>(3 * vertexIndex, iBlendShape, 3, 1)
804 .template cast<T>();
805 Eigen::Vector3<T> d_worldPos = calculateDWorldPos(character_, state, vertexIndex, d_restPos);
806
807 jac(0, paramIdx) += wgt * normal.dot(d_targetPos_d_vertexPos * d_worldPos);
808 if (iTriVert < 3) {
809 const Eigen::Vector3<T> diff_tgt_normal =
810 d_targetNormal_d_tgtTriVertexPos[iTriVert] * d_worldPos;
811 jac(0, paramIdx) += targetNormalSign * wgt * diff.dot(diff_tgt_normal);
812 }
813 }
814 // OUT handle derivatives wrt blend shape parameters:
815 }
816
817 res = dist * wgt;
818
819 return wgt * wgt * dist * dist;
820}

835template <typename T>
836double PointTriangleVertexErrorFunctionT<T>::getGradient(
837 const ModelParametersT<T>& modelParameters,
838 const SkeletonStateT<T>& state,
839 const MeshStateT<T>& meshState,
840 Eigen::Ref<Eigen::VectorX<T>> gradient) {
841 MT_CHECK_NOTNULL(meshState.posedMesh_);
842
843 double error = 0;
844
845 if (constraintType_ == VertexConstraintType::Position) {
846 for (size_t iCons = 0; iCons < constraints_.size(); ++iCons) {
847 error += calculatePositionGradient(
848 modelParameters, state, meshState, constraints_[iCons], gradient);
849 }
850 } else {
851 T sourceNormalWeight;
852 T targetNormalWeight;
853 std::tie(sourceNormalWeight, targetNormalWeight) = computeNormalWeights();

856 error += calculateNormalGradient(
857 modelParameters,
858 state,
859 meshState,
860 constraints_[iCons],
861 sourceNormalWeight,
862 targetNormalWeight,
863 gradient);
864 }
865 }
866
867 return this->weight_ * error;

870template <typename T>
871double PointTriangleVertexErrorFunctionT<T>::getJacobian(
872 const ModelParametersT<T>& modelParameters,
873 const SkeletonStateT<T>& state,
874 const MeshStateT<T>& meshState,
875 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
876 Eigen::Ref<Eigen::VectorX<T>> residual,
877 int& usedRows) {
878 MT_CHECK(
879 jacobian.cols() == static_cast<Eigen::Index>(this->parameterTransform_.transform.cols()));
880 MT_CHECK(jacobian.rows() >= (Eigen::Index)(1 * constraints_.size()));
881 MT_CHECK(residual.rows() >= (Eigen::Index)(1 * constraints_.size()));
882
883 MT_CHECK_NOTNULL(meshState.posedMesh_);
884
885 double error = 0;
886
887 if (constraintType_ == VertexConstraintType::Position) {
888 MT_PROFILE_EVENT("VertexErrorFunction - position jacobians");
889
890 for (size_t iCons = 0; iCons < constraints_.size(); ++iCons) {
891 error += calculatePositionJacobian(
892 modelParameters,
893 state,
894 meshState,
895 constraints_[iCons],
896 jacobian.block(3 * iCons, 0, 3, modelParameters.size()),
897 residual.middleRows(3 * iCons, 3));
898 }
899 usedRows = 3 * constraints_.size();
900 } else {
901 MT_PROFILE_EVENT("VertexErrorFunction - normal jacobians");
902 T sourceNormalWeight;
903 T targetNormalWeight;
904 std::tie(sourceNormalWeight, targetNormalWeight) = computeNormalWeights();
905
906 for (size_t iCons = 0; iCons < constraints_.size(); ++iCons) {
907 error += calculateNormalJacobian(
908 modelParameters,
909 state,
910 meshState,
911 constraints_[iCons],
912 sourceNormalWeight,
913 targetNormalWeight,
914 jacobian.block(iCons, 0, 1, modelParameters.size()),
915 residual(iCons));
916 }
917
918 usedRows = constraints_.size();
919 }
920
921 return error;
922}
