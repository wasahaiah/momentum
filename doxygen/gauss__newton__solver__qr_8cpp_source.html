55template <typename T>
56void GaussNewtonSolverQRT<T>::doIteration() {
57 MT_PROFILE_FUNCTION();
58
59 const auto sf = static_cast<SkeletonSolverFunctionT<T>*>(this->solverFunction_);
60
61 const auto skeleton = sf->getSkeleton();
62 const auto parameterTransform = sf->getParameterTransform();
63
64 {
65 MT_PROFILE_EVENT("JtJR - update state");
66 skeletonState_->set(parameterTransform->apply(this->parameters_), *skeleton);
67 }
68
69 if (sf->needsMeshState()) {
70 MT_PROFILE_EVENT("JtJR - update mesh state");
71 meshState_->update(this->parameters_, *skeletonState_, sf->getCharacter());
72 }
73
74 const auto nFullParams = gsl::narrow<Eigen::Index>(this->parameters_.size());
75
76 std::vector<Eigen::Index> enabledParameters;
77 enabledParameters.reserve(nFullParams);
78 for (int iFullParam = 0; iFullParam < nFullParams; ++iFullParam) {
79 if (this->activeParameters_.test(iFullParam)) {
80 enabledParameters.push_back(iFullParam);
81 }
82 }
83
84 const auto nSubsetParams = gsl::narrow_cast<int>(enabledParameters.size());
85
86 // momentum solves the problem (J^T*J + lambda*I) x = J^T*r;
87 // the QR solver wants the square root of that lambda.
88 qrSolver_.reset(nSubsetParams, std::sqrt(regularization_));
89
90 double error_orig = 0.0;
91 for (auto errorFunction : sf->getErrorFunctions()) {
92 if (errorFunction->getWeight() <= 0) {
93 continue;
94 }
95
96 const auto rows = gsl::narrow<Eigen::Index>(errorFunction->getJacobianSize());
97
98 jacobian_.resizeAndSetZero(rows, nFullParams);
99 residual_.resizeAndSetZero(rows);
100
101 int usedRows = 0;
102 error_orig += errorFunction->getJacobian(
103 this->parameters_,
104 *skeletonState_,
105 *meshState_,
106 jacobian_.mat(),
107 residual_.mat(),
108 usedRows);
109 if (usedRows == 0) {
110 continue;
111 }
112
113 qrSolver_.addMutating(
114 ColumnIndexedMatrix<Eigen::MatrixX<T>>(
115 jacobian_.mat().topRows(usedRows), enabledParameters),
116 residual_.mat().topRows(usedRows));
117 }
118
119 this->error_ = error_orig;
120
121 const Eigen::VectorX<T> subsetDelta = qrSolver_.result();
122
123 Eigen::VectorX<T> searchDir = Eigen::VectorX<T>::Zero(nFullParams);
124 for (int iSubsetParam = 0; iSubsetParam < nSubsetParams; ++iSubsetParam) {
125 const auto iFullParam = enabledParameters[iSubsetParam];
126 searchDir(iFullParam) = subsetDelta(iSubsetParam);
127 }
128
129 if (doLineSearch_) {
130 const double innerProd = -qrSolver_.At_times_b().dot(subsetDelta);
131
132 // Line search:
133 const float c_1 = 1e-4f;
134 const float tau = 0.5f;
135 float alpha = 1.0f;
136
137 const Eigen::VectorX<T> parameters_orig = this->parameters_;
138 for (size_t kStep = 0; kStep < 10 && std::fpclassify(alpha) == FP_NORMAL; ++kStep) {
139 // update the this->parameters_
140 this->parameters_ = parameters_orig;
141 this->solverFunction_->updateParameters(this->parameters_, alpha * searchDir);
142
143 const double error_new = this->solverFunction_->getError(this->parameters_);
144
145 if ((error_orig - error_new) >= c_1 * alpha * -innerProd) {
146 break;
147 }
148
149 // Reduce step size:
150 alpha = alpha * tau;
151 }
152 } else {
153 this->solverFunction_->updateParameters(this->parameters_, searchDir);
154 }
155}
