68 struct PairHash {
69 std::size_t operator()(const std::pair<size_t, size_t>& p) const {
70 // Use the smaller value as the high-order bits and the larger value as the low-order bits
71 return (static_cast<uint64_t>(std::min(p.first, p.second)) << (sizeof(size_t) * 8 / 2)) |
72 std::max(p.first, p.second);
73 }
74 };

79 (a.first == b.second && a.second == b.first);
