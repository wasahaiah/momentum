30SimdNormalConstraints::SimdNormalConstraints(const Skeleton* skel) {
31 // resize all arrays to the number of joints
32 MT_CHECK(skel->joints.size() <= kMaxJoints);
33 numJoints = static_cast<int>(skel->joints.size());
34 const auto dataSize = kMaxConstraints * numJoints;
35 MT_CHECK(dataSize % kSimdAlignment == 0);
36
37 // create memory for all floats
38 constexpr size_t nBlocks = 10;
39 data = makeAlignedUnique<float, kSimdAlignment>(dataSize * nBlocks);
40 float* dataPtr = data.get();
41 std::fill_n(dataPtr, dataSize * nBlocks, 0.0f);
42
43 offsetX = dataPtr + dataSize * 0;
44 offsetY = dataPtr + dataSize * 1;
45 offsetZ = dataPtr + dataSize * 2;
46 normalX = dataPtr + dataSize * 3;
47 normalY = dataPtr + dataSize * 4;
48 normalZ = dataPtr + dataSize * 5;
49 targetX = dataPtr + dataSize * 6;
50 targetY = dataPtr + dataSize * 7;
51 targetZ = dataPtr + dataSize * 8;
52 weights = dataPtr + dataSize * 9;
53}

57bool SimdNormalConstraints::addConstraint(
58 const size_t jointIndex,
59 const Vector3f& offset,
60 const Vector3f& normal,
61 const Vector3f& target,
62 const float targetWeight) {
63 MT_CHECK(jointIndex < constraintCount.size());
64
65 // add the constraint to the corresponding arrays of the jointIndex if there's enough space
66 uint32_t index = 0;
67 while (true) {
68 index = constraintCount[jointIndex];
69 if (index == kMaxConstraints) {
70 return false;
71 }
72
73 if (constraintCount[jointIndex].compare_exchange_weak(index, index + (uint32_t)1)) {
74 break;
75 }
76 }
77
78 const auto finalIndex = jointIndex * kMaxConstraints + index;
79 offsetX[finalIndex] = offset.x();
80 offsetY[finalIndex] = offset.y();
81 offsetZ[finalIndex] = offset.z();
82 normalX[finalIndex] = normal.x();
83 normalY[finalIndex] = normal.y();
84 normalZ[finalIndex] = normal.z();
85 targetX[finalIndex] = target.x();
86 targetY[finalIndex] = target.y();
87 targetZ[finalIndex] = target.z();
88 weights[finalIndex] = targetWeight;
89 return true;
90}

118double SimdNormalErrorFunction::getError(
119 const ModelParameters& /* params */,
120 const SkeletonState& state,
121 const MeshState& /* meshState */) {
122 if (constraints_ == nullptr) {
123 return 0.0f;
124 }
125
126 MT_CHECK((size_t)constraints_->numJoints <= state.jointState.size());
127
128 // loop over all joints, as these are our base units
129 auto error = drjit::zeros<DoubleP>(); // use double to prevent rounding errors
130 for (int jointId = 0; jointId < constraints_->numJoints; jointId++) {
131 const size_t constraintCount = constraints_->constraintCount[jointId];
132 MT_CHECK(jointId < static_cast<int>(state.jointState.size()));
133 const auto& jointState = state.jointState[jointId];
134 const Eigen::Matrix3f jointRotMat = jointState.rotation().toRotationMatrix();
135
136 for (uint32_t index = 0; index < constraintCount; index += kSimdPacketSize) {
137 const auto constraintOffsetIndex = jointId * SimdNormalConstraints::kMaxConstraints + index;
138 const Vector3fP target{
139 drjit::load<FloatP>(&constraints_->targetX[constraintOffsetIndex]),
140 drjit::load<FloatP>(&constraints_->targetY[constraintOffsetIndex]),
141 drjit::load<FloatP>(&constraints_->targetZ[constraintOffsetIndex])};
142
143 const Vector3fP offset{
144 drjit::load<FloatP>(&constraints_->offsetX[constraintOffsetIndex]),
145 drjit::load<FloatP>(&constraints_->offsetY[constraintOffsetIndex]),
146 drjit::load<FloatP>(&constraints_->offsetZ[constraintOffsetIndex])};
147 const Vector3fP pos_world = jointState.transform * offset;
148
149 const Vector3fP normal{
150 drjit::load<FloatP>(&constraints_->normalX[constraintOffsetIndex]),
151 drjit::load<FloatP>(&constraints_->normalY[constraintOffsetIndex]),
152 drjit::load<FloatP>(&constraints_->normalZ[constraintOffsetIndex])};
153 const Vector3fP normal_world = momentum::operator*(jointRotMat, normal);
154
155 const FloatP dist = dot(pos_world - target, normal_world);
156
157 const auto weight = drjit::load<FloatP>(&constraints_->weights[constraintOffsetIndex]);
158 error += weight * drjit::square(dist);
159 }
160 }
161
162 // sum the SIMD error register for final result
163 return static_cast<float>(drjit::sum(error) * kPlaneWeight * weight_);
164}

166double SimdNormalErrorFunction::getGradient(
167 const ModelParameters& /*params*/,
168 const SkeletonState& state,
169 const MeshState& /* meshState */,
170 Ref<VectorXf> gradient) {
171 if (constraints_ == nullptr) {
172 return 0.0f;
173 }
174
175 auto error = drjit::zeros<DoubleP>(); // use double to prevent rounding errors
176 for (int jointId = 0; jointId < constraints_->numJoints; jointId++) {
177 const size_t constraintCount = constraints_->constraintCount[jointId];
178 if (jointId >= static_cast<int>(state.jointState.size())) {
179 break;
180 }
181 const Eigen::Matrix3f jointRotMat = state.jointState[jointId].rotation().toRotationMatrix();
182 const auto& jointState_cons = state.jointState[jointId];
183 for (uint32_t index = 0; index < constraintCount; index += kSimdPacketSize) {
184 const auto constraintOffsetIndex = jointId * SimdNormalConstraints::kMaxConstraints + index;
185
186 const Vector3fP offset{
187 drjit::load<FloatP>(&constraints_->offsetX[constraintOffsetIndex]),
188 drjit::load<FloatP>(&constraints_->offsetY[constraintOffsetIndex]),
189 drjit::load<FloatP>(&constraints_->offsetZ[constraintOffsetIndex])};
190 const Vector3fP pos_world = jointState_cons.transform * offset;
191
192 const Vector3fP normal{
193 drjit::load<FloatP>(&constraints_->normalX[constraintOffsetIndex]),
194 drjit::load<FloatP>(&constraints_->normalY[constraintOffsetIndex]),
195 drjit::load<FloatP>(&constraints_->normalZ[constraintOffsetIndex])};
196 const Vector3fP normal_world = momentum::operator*(jointRotMat, normal);
197
198 const Vector3fP target{
199 drjit::load<FloatP>(&constraints_->targetX[constraintOffsetIndex]),
200 drjit::load<FloatP>(&constraints_->targetY[constraintOffsetIndex]),
201 drjit::load<FloatP>(&constraints_->targetZ[constraintOffsetIndex])};
202 const FloatP dist = dot(pos_world - target, normal_world);
203
204 const auto weight = drjit::load<FloatP>(&constraints_->weights[constraintOffsetIndex]);
205 const FloatP wgt = weight * (2.0f * kPlaneWeight);
206
207 error += weight * drjit::square(dist);
208
209 // loop over all joints the constraint is attached to and calculate gradient
210 size_t jointIndex = jointId;
211 while (jointIndex != kInvalidIndex) {
212 // check for valid index
213 const auto& jointState = state.jointState[jointIndex];
214
215 const size_t paramIndex = jointIndex * kParametersPerJoint;
216 const Vector3fP posd = momentum::operator-(pos_world, jointState.translation());
217
218 // calculate derivatives based on active joints
219 for (size_t d = 0; d < 3; d++) {
220 if (this->activeJointParams_[paramIndex + d]) {
221 // calculate joint gradient
222 const FloatP val =
223 dist * momentum::dot(normal_world, jointState.getTranslationDerivative(d)) * wgt;
224 // explicitly multiply with the parameter transform to generate parameter space
225 // gradients
226 for (auto ptIndex = parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
227 ptIndex < parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
228 ++ptIndex) {
229 gradient[parameterTransform_.transform.innerIndexPtr()[ptIndex]] +=
230 drjit::sum(val) * parameterTransform_.transform.valuePtr()[ptIndex];
231 }
232 }
233 if (this->activeJointParams_[paramIndex + 3 + d]) {
234 // calculate joint gradient
235 const auto crossProd =
236 cross(normal_world, momentum::operator-(target, jointState.translation()));
237 const FloatP val =
238 -dist * momentum::dot(jointState.rotationAxis.col(d), crossProd) * wgt;
239 // explicitly multiply with the parameter transform to generate parameter space
240 // gradients
241 for (auto ptIndex = parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3];
242 ptIndex < parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
243 ++ptIndex) {
244 gradient[parameterTransform_.transform.innerIndexPtr()[ptIndex]] +=
245 drjit::sum(val) * parameterTransform_.transform.valuePtr()[ptIndex];
246 }
247 }
248 }
249 if (this->activeJointParams_[paramIndex + 6]) {
250 // calculate joint gradient
251 const FloatP val = dist * ln2() * dot(normal_world, posd) * wgt;
252 // explicitly multiply with the parameter transform to generate parameter space gradients
253 for (auto ptIndex = parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
254 ptIndex < parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
255 ++ptIndex) {
256 gradient[parameterTransform_.transform.innerIndexPtr()[ptIndex]] +=
257 drjit::sum(val) * parameterTransform_.transform.valuePtr()[ptIndex];
258 }
259 }
260
261 // go to the next joint
262 jointIndex = this->skeleton_.joints[jointIndex].parent;
263 }
264 }
265 }
266
267 return static_cast<float>(drjit::sum(error) * kPlaneWeight * weight_);
268}

270__vectorcall DRJIT_INLINE void jacobian_jointParams_to_modelParams(
271 const FloatP& jacobian_jointParams,
272 const Eigen::Index iJointParam,
273 const ParameterTransform& parameterTransform_,
274 Eigen::Ref<Eigen::MatrixX<float>> jacobian) {
275 // explicitly multiply with the parameter transform to generate parameter space gradients
276 for (auto index = parameterTransform_.transform.outerIndexPtr()[iJointParam];
277 index < parameterTransform_.transform.outerIndexPtr()[iJointParam + 1];
278 ++index) {
279 const auto modelParamIdx = parameterTransform_.transform.innerIndexPtr()[index];
280 float* jacPtr = jacobian.col(modelParamIdx).data();
281 drjit::store(
282 jacPtr,
283 drjit::load<FloatP>(jacPtr) +
284 parameterTransform_.transform.valuePtr()[index] * jacobian_jointParams);
285 }

288double SimdNormalErrorFunction::getJacobian(
289 const ModelParameters& /*params*/,
290 const SkeletonState& state,
291 const MeshState& /* meshState */,
292 Ref<MatrixXf> jacobian,
293 Ref<VectorXf> residual,
294 int& usedRows) {
295 MT_PROFILE_FUNCTION();
296 MT_CHECK(jacobian.cols() == static_cast<Eigen::Index>(parameterTransform_.transform.cols()));
297
298 if (constraints_ == nullptr) {
299 return 0.0f;
300 }
301
302 MT_CHECK((size_t)constraints_->numJoints <= jacobianOffset_.size());
303
304 // storage for joint errors
305 std::vector<double> jointErrors(constraints_->numJoints);
306
307 // loop over all joints, as these are our base units
308 auto dispensoOptions = dispenso::ParForOptions();
309 dispensoOptions.maxThreads = maxThreads_;
310 dispenso::parallel_for(
311 0,
312 constraints_->numJoints,
313 [&](const size_t jointId) {
314 const size_t constraintCount = constraints_->constraintCount[jointId];
315 const auto& jointState_cons = state.jointState[jointId];
316 const Eigen::Matrix3f jointRotMat = jointState_cons.rotation().toRotationMatrix();
317 auto jointError = drjit::zeros<DoubleP>(); // use double to prevent rounding errors
318
319 for (uint32_t index = 0; index < constraintCount; index += kSimdPacketSize) {
320 const auto constraintOffsetIndex =
321 jointId * SimdNormalConstraints::kMaxConstraints + index;
322 const auto jacobianOffsetCur = jacobianOffset_[jointId] + index;
323
324 const Vector3fP offset{
325 drjit::load<FloatP>(&constraints_->offsetX[constraintOffsetIndex]),
326 drjit::load<FloatP>(&constraints_->offsetY[constraintOffsetIndex]),
327 drjit::load<FloatP>(&constraints_->offsetZ[constraintOffsetIndex])};
328 const Vector3fP pos_world = jointState_cons.transform * offset;
329
330 const Vector3fP normal{
331 drjit::load<FloatP>(&constraints_->normalX[constraintOffsetIndex]),
332 drjit::load<FloatP>(&constraints_->normalY[constraintOffsetIndex]),
333 drjit::load<FloatP>(&constraints_->normalZ[constraintOffsetIndex])};
334 const Vector3fP normal_world = momentum::operator*(jointRotMat, normal);
335
336 const Vector3fP target{
337 drjit::load<FloatP>(&constraints_->targetX[constraintOffsetIndex]),
338 drjit::load<FloatP>(&constraints_->targetY[constraintOffsetIndex]),
339 drjit::load<FloatP>(&constraints_->targetZ[constraintOffsetIndex])};
340 const FloatP dist = dot(pos_world - target, normal_world);
341
342 const auto weight = drjit::load<FloatP>(&constraints_->weights[constraintOffsetIndex]);
343 const FloatP wgt = drjit::sqrt(weight * kPlaneWeight * this->weight_);
344
345 jointError += weight * drjit::square(dist);
346
347 drjit::store(residual.segment(jacobianOffsetCur, kSimdPacketSize).data(), dist * wgt);
348
349 // loop over all joints the constraint is attached to and calculate gradient
350 size_t jointIndex = jointId;
351 while (jointIndex != kInvalidIndex) {
352 // check for valid index
353 const auto& jointState = state.jointState[jointIndex];
354 const size_t paramIndex = jointIndex * kParametersPerJoint;
355 // calculate derivatives based on active joints
356 for (size_t d = 0; d < 3; d++) {
357 if (this->activeJointParams_[paramIndex + d]) {
358 const FloatP jc =
359 momentum::dot(normal_world, jointState.getTranslationDerivative(d)) * wgt;
360 jacobian_jointParams_to_modelParams(
361 jc,
362 paramIndex + d,
363 parameterTransform_,
364 jacobian.middleRows(jacobianOffsetCur, kSimdPacketSize));
365 }
366 }
367
368 const Vector3fP tgtd = momentum::operator-(target, jointState.translation());
369 for (size_t d = 0; d < 3; ++d) {
370 if (this->activeJointParams_[paramIndex + 3 + d]) {
371 const Vector3fP crossProd = cross(normal_world, tgtd);
372 const Vector3f axis = jointState.rotationAxis.col(d);
373 const FloatP jc = -momentum::dot(axis, crossProd) * wgt;
374 jacobian_jointParams_to_modelParams(
375 jc,
376 paramIndex + d + 3,
377 parameterTransform_,
378 jacobian.middleRows(jacobianOffsetCur, kSimdPacketSize));
379 }
380 }
381
382 if (this->activeJointParams_[paramIndex + 6]) {
383 const Vector3fP posd = momentum::operator-(pos_world, jointState.translation());
384 const FloatP jc = dot(normal_world, posd) * (ln2() * wgt);
385 jacobian_jointParams_to_modelParams(
386 jc,
387 paramIndex + 6,
388 parameterTransform_,
389 jacobian.middleRows(jacobianOffsetCur, kSimdPacketSize));
390 }
391
392 // go to the next joint
393 jointIndex = this->skeleton_.joints[jointIndex].parent;
394 }
395 }
396
397 jointErrors[jointId] += drjit::sum(jointError);
398 },
399 dispensoOptions);
400
401 double error =
402 kPlaneWeight * weight_ * std::accumulate(jointErrors.begin(), jointErrors.end(), 0.0);
403
404 usedRows = gsl::narrow_cast<int>(jacobian.rows());
405
406 // sum the SIMD error register for final result
407 return static_cast<float>(error);
408}

410size_t SimdNormalErrorFunction::getJacobianSize() const {
411 if (constraints_ == nullptr) {
412 return 0;
413 }
414
415 jacobianOffset_.resize(constraints_->numJoints);
416 size_t num = 0;
417 for (int i = 0; i < constraints_->numJoints; i++) {
418 jacobianOffset_[i] = num;
419 const size_t count = constraints_->constraintCount[i];
420 const auto numPackets = (count + kSimdPacketSize - 1) / kSimdPacketSize;
421 num += numPackets * kSimdPacketSize;
422 }
423 return num;
424}

428inline double __vectorcall sum8(const __m256 x) {
429 // extract the higher and lower 4 floats
430 const __m128 high = _mm256_extractf128_ps(x, 1);
431 const __m128 low = _mm256_castps256_ps128(x);
432 // convert them to 256 bit doubles and add them up
433 const __m256d val = _mm256_add_pd(_mm256_cvtps_pd(high), _mm256_cvtps_pd(low));
434 // sum the 4 doubles up and return the result
435 const __m128d valupper = _mm256_extractf128_pd(val, 1);
436 const __m128d vallower = _mm256_castpd256_pd128(val);
437 _mm256_zeroupper();
438 const __m128d valval = _mm_add_pd(valupper, vallower);
439 const __m128d res = _mm_add_pd(_mm_permute_pd(valval, 1), valval);
440 return _mm_cvtsd_f64(res);
441}

443double SimdNormalErrorFunctionAVX::getJacobian(
444 const ModelParameters& /* params */,
445 const SkeletonState& state,
446 const MeshState& /* meshState */,
447 Ref<MatrixXf> jacobian,
448 Ref<VectorXf> residual,
449 int& usedRows) {
450 MT_PROFILE_FUNCTION();
451 MT_CHECK(jacobian.cols() == static_cast<Eigen::Index>(parameterTransform_.transform.cols()));
452
453 if (constraints_ == nullptr) {
454 return 0.0f;
455 }
456
457 MT_CHECK((size_t)constraints_->numJoints <= jacobianOffset_.size());
458
459 // storage for joint errors
460 std::vector<double> ets_error;
461
462 // need to make sure we're actually at a 32 byte data offset at the first offset for AVX access
463 const size_t addressOffset = computeOffset<kAvxAlignment>(jacobian);
464 checkAlignment<kAvxAlignment>(jacobian, addressOffset);
465
466 // loop over all joints, as these are our base units
467 auto dispensoOptions = dispenso::ParForOptions();
468 dispensoOptions.maxThreads = maxThreads_;
469 dispenso::parallel_for(
470 ets_error,
471 []() { return 0.0; },
472 0,
473 constraints_->numJoints,
474 [&](double& error_local, const size_t jointId) {
475 // get initial offset
476 const auto offset = jacobianOffset_[jointId] + addressOffset;
477
478 // pre-load some joint specific values
479 const auto transformation = state.jointState[jointId].transform.toMatrix();
480
481 __m256 posx;
482 __m256 posy;
483 __m256 posz;
484 __m256 nmlx;
485 __m256 nmly;
486 __m256 nmlz;
487 __m256 dist;
488
489 const auto jointOffset = jointId * SimdNormalConstraints::kMaxConstraints;
490
491 // loop over all constraints in increments of kAvxPacketSize
492 const uint32_t count = constraints_->constraintCount[jointId];
493 for (uint32_t index = 0; index < count; index += kAvxPacketSize) {
494 // transform offset by joint transformation : pos = transform * offset
495 // also transform normal by transformation : nml = (transform.linear() *
496 // normal).normalized()
497 const __m256 valx = _mm256_load_ps(&constraints_->offsetX[jointOffset + index]);
498 const __m256 nx = _mm256_load_ps(&constraints_->normalX[jointOffset + index]);
499 posx = _mm256_mul_ps(valx, _mm256_broadcast_ss(&transformation.data()[0]));
500 posy = _mm256_mul_ps(valx, _mm256_broadcast_ss(&transformation.data()[1]));
501 posz = _mm256_mul_ps(valx, _mm256_broadcast_ss(&transformation.data()[2]));
502 nmlx = _mm256_mul_ps(nx, _mm256_broadcast_ss(&transformation.data()[0]));
503 nmly = _mm256_mul_ps(nx, _mm256_broadcast_ss(&transformation.data()[1]));
504 nmlz = _mm256_mul_ps(nx, _mm256_broadcast_ss(&transformation.data()[2]));
505 const __m256 valy = _mm256_load_ps(&constraints_->offsetY[jointOffset + index]);
506 const __m256 ny = _mm256_load_ps(&constraints_->normalY[jointOffset + index]);
507 posx = _mm256_fmadd_ps(valy, _mm256_broadcast_ss(&transformation.data()[4]), posx);
508 posy = _mm256_fmadd_ps(valy, _mm256_broadcast_ss(&transformation.data()[5]), posy);
509 posz = _mm256_fmadd_ps(valy, _mm256_broadcast_ss(&transformation.data()[6]), posz);
510 nmlx = _mm256_fmadd_ps(ny, _mm256_broadcast_ss(&transformation.data()[4]), nmlx);
511 nmly = _mm256_fmadd_ps(ny, _mm256_broadcast_ss(&transformation.data()[5]), nmly);
512 nmlz = _mm256_fmadd_ps(ny, _mm256_broadcast_ss(&transformation.data()[6]), nmlz);
513 const __m256 valz = _mm256_load_ps(&constraints_->offsetZ[jointOffset + index]);
514 const __m256 nz = _mm256_load_ps(&constraints_->normalZ[jointOffset + index]);
515 posx = _mm256_fmadd_ps(valz, _mm256_broadcast_ss(&transformation.data()[8]), posx);
516 posy = _mm256_fmadd_ps(valz, _mm256_broadcast_ss(&transformation.data()[9]), posy);
517 posz = _mm256_fmadd_ps(valz, _mm256_broadcast_ss(&transformation.data()[10]), posz);
518 nmlx = _mm256_fmadd_ps(nz, _mm256_broadcast_ss(&transformation.data()[8]), nmlx);
519 nmly = _mm256_fmadd_ps(nz, _mm256_broadcast_ss(&transformation.data()[9]), nmly);
520 nmlz = _mm256_fmadd_ps(nz, _mm256_broadcast_ss(&transformation.data()[10]), nmlz);
521 posx = _mm256_add_ps(_mm256_broadcast_ss(&transformation.data()[12]), posx);
522 posy = _mm256_add_ps(_mm256_broadcast_ss(&transformation.data()[13]), posy);
523 posz = _mm256_add_ps(_mm256_broadcast_ss(&transformation.data()[14]), posz);
524
525 __m256 norm = _mm256_mul_ps(nmlx, nmlx);
526 norm = _mm256_fmadd_ps(nmly, nmly, norm);
527 norm = _mm256_fmadd_ps(nmlz, nmlz, norm);
528 norm = _mm256_rsqrt_ps(_mm256_max_ps(norm, _mm256_set1_ps(0.0001f)));
529 nmlx = _mm256_mul_ps(nmlx, norm);
530 nmly = _mm256_mul_ps(nmly, norm);
531 nmlz = _mm256_mul_ps(nmlz, norm);
532
533 // calculate distance of point to plane : dist = normal.dot(pos) -
534 // normal.dot(data->target)
535 const __m256 tgtx = _mm256_load_ps(&constraints_->targetX[jointOffset + index]);
536 const __m256 tgty = _mm256_load_ps(&constraints_->targetY[jointOffset + index]);
537 const __m256 tgtz = _mm256_load_ps(&constraints_->targetZ[jointOffset + index]);
538 dist = _mm256_mul_ps(posx, nmlx);
539 dist = _mm256_fmadd_ps(posy, nmly, dist);
540 dist = _mm256_fmadd_ps(posz, nmlz, dist);
541 dist = _mm256_fnmadd_ps(tgtx, nmlx, dist);
542 dist = _mm256_fnmadd_ps(tgty, nmly, dist);
543 dist = _mm256_fnmadd_ps(tgtz, nmlz, dist);
544
545 // calculate square-root of weight : wgt = sqrt(kPlaneWeight * weight
546 // * constraintWeight)
547 const float w = kPlaneWeight * weight_;
548 const __m256 wt = _mm256_mul_ps(
549 _mm256_broadcast_ss(&w), _mm256_load_ps(&constraints_->weights[jointOffset + index]));
550 const __m256 wgt = _mm256_sqrt_ps(wt);
551
552 // calculate residual : res = wgt * dist
553 const __m256 res = _mm256_mul_ps(wgt, dist);
554
555 // calculate error as squared residual : err = res * res
556 const __m256 err = _mm256_mul_ps(res, res);
557
558 // sum up the values and add to error
559 error_local += sum8(err);
560
561 // loop over all joints the constraint is attached to and calculate gradient
562 size_t jointIndex = jointId;
563 while (jointIndex != kInvalidIndex) {
564 // check for valid index
565 const auto& jointState = state.jointState[jointIndex];
566 const size_t paramIndex = jointIndex * kParametersPerJoint;
567
568 // calculate difference between constraint position and joint center : posd =
569 // pos - jointState.translation
570 const __m256 posdx = _mm256_sub_ps(posx, _mm256_broadcast_ss(&jointState.x()));
571 const __m256 posdy = _mm256_sub_ps(posy, _mm256_broadcast_ss(&jointState.y()));
572 const __m256 posdz = _mm256_sub_ps(posz, _mm256_broadcast_ss(&jointState.z()));
573
574 // calculate difference between target position and joint center : tgtd =
575 // tgt - jointState.translation
576 const __m256 tgtdx = _mm256_sub_ps(tgtx, _mm256_broadcast_ss(&jointState.x()));
577 const __m256 tgtdy = _mm256_sub_ps(tgty, _mm256_broadcast_ss(&jointState.y()));
578 const __m256 tgtdz = _mm256_sub_ps(tgtz, _mm256_broadcast_ss(&jointState.z()));
579
580 // calculate derivatives based on active joints
581 for (size_t d = 0; d < 3; d++) {
582 if (activeJointParams_[paramIndex + d]) {
583 // calculate jacobian with respect to joint : jac =
584 // normal.dot(axis) * wgt;
585 const Vector3f& axis = jointState.translationAxis.col(d);
586 const __m256 axisx = _mm256_broadcast_ss(&axis.x());
587 const __m256 axisy = _mm256_broadcast_ss(&axis.y());
588 const __m256 axisz = _mm256_broadcast_ss(&axis.z());
589 const __m256 jac = _mm256_mul_ps(
590 _mm256_fmadd_ps(
591 nmlz, axisz, _mm256_fmadd_ps(nmly, axisy, _mm256_mul_ps(nmlx, axisx))),
592 wgt);
593
594 // explicitly multiply with the parameter transform to generate parameter space
595 // jacobians
596 for (auto pIndex = parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
597 pIndex < parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
598 ++pIndex) {
599 const float pw = parameterTransform_.transform.valuePtr()[pIndex];
600 const __m256 pjac = _mm256_mul_ps(_mm256_broadcast_ss(&pw), jac);
601 auto* const address = &jacobian(
602 offset + index, parameterTransform_.transform.innerIndexPtr()[pIndex]);
603 MT_CHECK((uintptr_t)address % 32 == 0);
604 const __m256 prev = _mm256_load_ps(address);
605 _mm256_store_ps(address, _mm256_add_ps(prev, pjac));
606 }
607 }
608 if (activeJointParams_[paramIndex + 3 + d]) {
609 // calculate jacobian with respect to joint : jac =
610 // axis.dot(tgtd.cross(normal)) * wgt;
611 const __m256 crossx = _mm256_fmsub_ps(tgtdy, nmlz, _mm256_mul_ps(tgtdz, nmly));
612 const __m256 crossy = _mm256_fmsub_ps(tgtdz, nmlx, _mm256_mul_ps(tgtdx, nmlz));
613 const __m256 crossz = _mm256_fmsub_ps(tgtdx, nmly, _mm256_mul_ps(tgtdy, nmlx));
614 const Vector3f& axis = jointState.rotationAxis.col(d);
615 const __m256 axisx = _mm256_broadcast_ss(&axis.x());
616 const __m256 axisy = _mm256_broadcast_ss(&axis.y());
617 const __m256 axisz = _mm256_broadcast_ss(&axis.z());
618 const __m256 jac = _mm256_mul_ps(
619 _mm256_fmadd_ps(
620 axisz,
621 crossz,
622 _mm256_fmadd_ps(axisy, crossy, _mm256_mul_ps(axisx, crossx))),
623 wgt);
624
625 // explicitly multiply with the parameter transform to generate parameter space
626 // jacobians
627 for (auto pIndex =
628 parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3];
629 pIndex < parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
630 ++pIndex) {
631 const float pw = parameterTransform_.transform.valuePtr()[pIndex];
632 const __m256 pjac = _mm256_mul_ps(_mm256_broadcast_ss(&pw), jac);
633 auto* const address = &jacobian(
634 offset + index, parameterTransform_.transform.innerIndexPtr()[pIndex]);
635 MT_CHECK((uintptr_t)address % 32 == 0);
636 const __m256 prev = _mm256_load_ps(address);
637 _mm256_store_ps(address, _mm256_add_ps(prev, pjac));
638 }
639 }
640 }
641 if (activeJointParams_[paramIndex + 6]) {
642 // calculate jacobian with respect to joint :
643 // jac = normal.dot(posd) * wgt * LN2;
644 constexpr double kLn2d = ln2<double>();
645 const __m256 jac = _mm256_mul_ps(
646 _mm256_mul_ps(
647 _mm256_fmadd_ps(
648 nmlz, posdz, _mm256_fmadd_ps(nmly, posdy, _mm256_mul_ps(nmlx, posdx))),
649 wgt),
650 _mm256_set_ps(kLn2d, kLn2d, kLn2d, kLn2d, kLn2d, kLn2d, kLn2d, kLn2d));
651
652 // explicitly multiply with the parameter transform to generate parameter space
653 // jacobians
654 for (auto pIndex = parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
655 pIndex < parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
656 ++pIndex) {
657 const float pw = parameterTransform_.transform.valuePtr()[pIndex];
658 const __m256 pjac = _mm256_mul_ps(_mm256_broadcast_ss(&pw), jac);
659 auto* const address = &jacobian(
660 offset + index, parameterTransform_.transform.innerIndexPtr()[pIndex]);
661 MT_CHECK((uintptr_t)address % 32 == 0);
662 const __m256 prev = _mm256_load_ps(address);
663 _mm256_store_ps(address, _mm256_add_ps(prev, pjac));
664 }
665 }
666
667 // go to the next joint
668 jointIndex = skeleton_.joints[jointIndex].parent;
669 }
670
671 // store the residual
672 _mm256_storeu_ps(&residual(offset + index), res);
673 }
674 },
675 dispensoOptions);
676
677 double error = std::accumulate(ets_error.begin(), ets_error.end(), 0.0);
678
679 usedRows = gsl::narrow_cast<int>(jacobian.rows());
680
681 // sum the AVX error register for final result
682 return static_cast<float>(error);
683}

685size_t SimdNormalErrorFunctionAVX::getJacobianSize() const {
686 if (constraints_ == nullptr) {
687 return 0;
688 }
689
690 jacobianOffset_.resize(constraints_->numJoints);
691 size_t num = 0;
692 for (int i = 0; i < constraints_->numJoints; i++) {
693 jacobianOffset_[i] = num;
694 const size_t count = constraints_->constraintCount[i];
695 const auto numPackets = (count + kAvxPacketSize - 1) / kAvxPacketSize;
696 num += numPackets * kAvxPacketSize;
697 }
698 return num;
699}
