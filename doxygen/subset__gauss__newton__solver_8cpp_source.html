50 const auto nFullParams = this->parameters_.size();
51 this->enabledParameters_.clear();
52 this->enabledParameters_.reserve(nFullParams);
53 for (int iFullParam = 0; iFullParam < nFullParams; ++iFullParam) {
54 if (this->activeParameters_.test(iFullParam)) {

71template <typename T>
72void SubsetGaussNewtonSolverT<T>::doIteration() {
73 MT_PROFILE_FUNCTION();
74
75 // get the jacobian and residual
76 size_t jacobianRows = 0;
77 this->error_ = this->solverFunction_->getJacobian(
78 this->parameters_, this->jacobian_, this->residual_, jacobianRows);
79 const double error_orig = this->error_;
80
81 const int nSubsetParams = gsl::narrow_cast<int>(this->enabledParameters_.size());
82
83 // Fill in J for this subset
84
85 // Move the solved columns of the Jacobian into place.
86 // Doing this in-place in the jacobian minimizes the total work.
87 // For the common case where all the "dropped" parameters at the end,
88 // this will effectively be a no-op, since enabledParameters_[iSubsetParam] = iFullParam
89 // for all parameters prior to the dropped ones.
90 for (int iSubsetParam = 0; iSubsetParam < nSubsetParams; ++iSubsetParam) {
91 // Only move columns that need it:
92 if (this->enabledParameters_[iSubsetParam] > iSubsetParam) {
93 this->jacobian_.col(iSubsetParam) =
94 this->jacobian_.col(this->enabledParameters_[iSubsetParam]);
95 }
96 }
97
98 const auto JBlock = this->jacobian_.topLeftCorner(jacobianRows, nSubsetParams);
99
100 // Compute grad = (J^T * r) and hess ~= (J^T * J) for the actual solved subset:
101 // The gradient is just Jt * r but only keeping `jacobianRows` which can be smaller than the
102 // number of rows of the `jacobian` matrix.
103 this->subsetGradient_.noalias() = this->residual_.head(jacobianRows).transpose() * JBlock;
104
105 // The Hessian takes a regularizer on top of Jt * J
106 this->subsetHessian_.template triangularView<Eigen::Lower>() = JBlock.transpose() * JBlock;
107 this->subsetHessian_.diagonal().array() += this->regularization_;
108
109 // LLT only reads the lower triangular part
110 subsetDelta_.noalias() = llt_.compute(this->subsetHessian_).solve(subsetGradient_);
111
112 // We don't need to set this->delta_ to zero because it is already set to zero when
113 // enabledParameters_ is updated.
114 for (int iSubsetParam = 0; iSubsetParam < nSubsetParams; ++iSubsetParam) {
115 const int iFullParam = this->enabledParameters_[iSubsetParam];
116 delta_(iFullParam) = subsetDelta_(iSubsetParam);
117 }
118 // update the parameters
119 if (doLineSearch_) {
120 const double innerProd = -this->subsetGradient_.dot(subsetDelta_);
121
122 // Line search:
123 const float c_1 = 1e-4F;
124 const float tau = 0.5F;
125 float alpha = 1.0F;
126
127 const Eigen::VectorX<T> parameters_orig = this->parameters_;
128 for (size_t kStep = 0; kStep < 10 && std::fpclassify(alpha) == FP_NORMAL; ++kStep) {
129 // update the this->parameters_
130 this->parameters_ = parameters_orig;
131 this->solverFunction_->updateParameters(this->parameters_, alpha * delta_);
132
133 const double error_new = this->solverFunction_->getError(this->parameters_);
134
135 if ((error_orig - error_new) >= c_1 * alpha * -innerProd) {
136 break;
137 }
138
139 // Reduce step size:
140 alpha = alpha * tau;
141 }
142 } else {
143 this->solverFunction_->updateParameters(this->parameters_, delta_);
144 }
145}
