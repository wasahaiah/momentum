21template <typename T>
22BlendWeightsT<T> extractBlendWeights(
23 const ParameterTransform& paramTransform,
24 const ModelParametersT<T>& modelParams) {
25 BlendWeightsT<T> result = Eigen::VectorX<T>::Zero(paramTransform.blendShapeParameters.size());
26 for (Eigen::Index iBasis = 0; iBasis < result.size(); ++iBasis) {
27 const auto paramIdx = paramTransform.blendShapeParameters[iBasis];
28 if (paramIdx >= 0 && paramIdx < modelParams.size()) {
29 result(iBasis) = modelParams(paramIdx);
30 }
31 }
32 return result;
33}

59template <typename T>
60void skinWithBlendShapes(
61 const Character& character,
62 const SkeletonStateT<T>& state,
63 const BlendWeightsT<T>& blendWeights,
64 MeshT<T>& outputMesh) {
65 MT_CHECK((bool)character.skinWeights);
66 MT_CHECK((bool)character.mesh);
67
68 const auto& inverseBindPose = character.inverseBindPose;
69 const auto& skin = *character.skinWeights;
70
71 // some sanity checks
72 MT_CHECK(
73 state.jointState.size() == inverseBindPose.size(),
74 "{} is not {}",
75 state.jointState.size(),
76 inverseBindPose.size());
77 MT_CHECK(
78 skin.index.rows() == gsl::narrow<int>(character.mesh->vertices.size()),
79 "{} is not {}",
80 skin.index.rows(),
81 character.mesh->vertices.size());
82 MT_CHECK(
83 skin.weight.rows() == gsl::narrow<int>(character.mesh->vertices.size()),
84 "{} is not {}",
85 skin.weight.rows(),
86 character.mesh->vertices.size());
87
88 // first create a list of transformations from bindpose to final output pose
89 std::vector<Eigen::Matrix4<T>> transformations(state.jointState.size());
90 for (size_t i = 0; i < state.jointState.size(); i++) {
91 transformations[i] = (state.jointState[i].transform * inverseBindPose[i].cast<T>()).matrix();
92 }
93
94 MT_CHECK(
95 !character.blendShape ||
96 blendWeights.size() <= character.blendShape->getShapeVectors().cols(),
97 "blendWeights.size(): {}; ShapeVectors.cols(): {}",
98 blendWeights.size(),
99 character.blendShape->getShapeVectors().cols());
100
101 outputMesh.vertices.resize(character.mesh->vertices.size(), Eigen::Vector3<T>::Zero());
102
103 // go over all vertices and perform transformation
104 // SSE version for linux/windows
105 dispenso::parallel_for(
106 dispenso::makeChunkedRange(0, skin.index.rows(), dispenso::ParForChunking::kAuto),
107 [&blendWeights, &character, &transformations, &outputMesh](
108 const size_t rangeBegin, const size_t rangeEnd) {
109 const auto& skinWeights = *character.skinWeights;
110 for (size_t iVert = rangeBegin; iVert != rangeEnd; iVert++) {
111 // Compute rest position from blend shape:
112 Eigen::Vector3<T> p_rest;
113 if (character.blendShape) {
114 const auto& blendShape = *character.blendShape;
115 p_rest = (blendShape.getBaseShape()[iVert].cast<T>() +
116 blendShape.getShapeVectors()
117 .block(3 * iVert, 0, 3, blendWeights.size())
118 .template cast<T>() *
119 blendWeights.v)
120 .eval();
121 } else {
122 p_rest = character.mesh->vertices[iVert].cast<T>().eval();
123 }
124
125 Eigen::Vector3<T> p_skinned = Eigen::Vector3<T>::Zero();
126 for (uint32_t k = 0; k < kMaxSkinJoints; ++k) {
127 const auto bIndex = skinWeights.index(iVert, k);
128 const T bWeight = skinWeights.weight(iVert, k);
129 if (bWeight == 0) {
130 break;
131 }
132 p_skinned +=
133 (bWeight * (transformations[bIndex] * p_rest.homogeneous())).template head<3>();
134 }
135 outputMesh.vertices[iVert] = p_skinned;
136 }
137 });
138
139 // Because of the blend shapes, we can't "skin" the normals so we just have to compute them from
140 // the mesh.
141 outputMesh.updateNormals();
142}
