33template <typename T>
34Vector3<T> quaternionLogMapRelativeDerivativeQ1(
35 const Quaternion<T>& q1,
36 const Quaternion<T>& q2,
37 const Vector3<T>& dir,
38 const Eigen::Matrix<T, 3, 4>& dLogDq) {
39 const Vector3<T> vHalf = dir / T(2);
40
41 // Compute d/dt [exp(t*v) * q1]|_{t=0} = [0, v/2] * q1
42 const Vector3<T> q1_vec(q1.x(), q1.y(), q1.z());
43 const T dq1_w = -vHalf.dot(q1_vec);
44 const Vector3<T> dq1_vec = vHalf * q1.w() + vHalf.cross(q1_vec);
45
46 // Then multiply by q2^{-1} on the left: q2^{-1} * [dq1_w, dq1_vec]
47 const Vector3<T> q2_vec(q2.x(), q2.y(), q2.z());
48 const T dqRel_w = q2.w() * dq1_w - (-q2_vec).dot(dq1_vec);
49 const Vector3<T> dqRel_vec = q2.w() * dq1_vec + dq1_w * (-q2_vec) + (-q2_vec).cross(dq1_vec);
50
51 // Apply chain rule: d(log)/d(t) = d(log)/d(q_rel) * d(q_rel)/d(t)
52 Eigen::Vector4<T> dqRelDt;
53 dqRelDt << dqRel_vec, dqRel_w;
54 return dLogDq * dqRelDt;
55}

58template <typename T>
59Vector3<T> quaternionLogMapRelativeDerivativeQ2(
60 const Quaternion<T>& q1,
61 const Quaternion<T>& q2,
62 const Vector3<T>& dir,
63 const Eigen::Matrix<T, 3, 4>& dLogDq) {
64 const Vector3<T> vHalf = dir / T(2);
65
66 // Compute q2^{-1} * [0, v/2]
67 const Vector3<T> q2_vec(q2.x(), q2.y(), q2.z());
68 const Vector3<T> q2inv_vec = -q2_vec;
69 const T temp_w = -q2inv_vec.dot(vHalf);
70 const Vector3<T> temp_vec = q2.w() * vHalf + q2inv_vec.cross(vHalf);
71
72 // Then multiply by q1 on the right: [temp_w, temp_vec] * q1
73 const Vector3<T> q1_vec(q1.x(), q1.y(), q1.z());
74 const T dqRel_w = temp_w * q1.w() - temp_vec.dot(q1_vec);
75 const Vector3<T> dqRel_vec = temp_w * q1_vec + q1.w() * temp_vec + temp_vec.cross(q1_vec);
76
77 // Apply chain rule with negative sign from derivative of inverse
78 Eigen::Vector4<T> dqRelDt;
79 dqRelDt << -dqRel_vec, -dqRel_w;
80 return dLogDq * dqRelDt;
81}

96template <typename T>
97T computeLogMapRotationGradientPrev(
98 const Quaternion<T>& prevRot,
99 const Quaternion<T>& nextRot,
100 const Quaternion<T>& targetRot,
101 const Vector3<T>& axis,
102 const Eigen::Matrix<T, 3, 4>& dLogDq) {
103 const Quaternion<T> transformedPrevRot = targetRot * prevRot;
104 const Quaternion<T> qRel = nextRot.conjugate() * transformedPrevRot;
105 const Vector3<T> logmapVec = quaternionLogMap(qRel);
106 const Vector3<T> transformedAxis = targetRot * axis;
107 const Vector3<T> jacobian =
108 quaternionLogMapRelativeDerivativeQ1(transformedPrevRot, nextRot, transformedAxis, dLogDq);
109 return T(2) * logmapVec.dot(jacobian);
110}

134template <typename T>
135T computeMatrixDiffRotationGradientPrev(
136 const Quaternion<T>& prevRot,
137 const Quaternion<T>& targetRot,
138 const Vector3<T>& axis,
139 const Matrix3<T>& rotDiff) {
140 const auto rotD = targetRot * crossProductMatrix(axis) * prevRot;
141 return -T(2) * rotD.cwiseProduct(rotDiff).sum();
142}

162template <typename T>
163Eigen::Matrix<T, 3, 1> computeLogMapRotationJacobianPrev(
164 const Quaternion<T>& prevRot,
165 const Quaternion<T>& nextRot,
166 const Quaternion<T>& targetRot,
167 const Vector3<T>& axis,
168 T weight,
169 const Eigen::Matrix<T, 3, 4>& dLogDq) {
170 const Quaternion<T> transformedPrevRot = targetRot * prevRot;

269template <typename T>
270double StateSequenceErrorFunctionT<T>::getError(
271 std::span<const ModelParametersT<T>> /* modelParameters */,
272 std::span<const SkeletonStateT<T>> skelStates,
273 std::span<const MeshStateT<T>> /* meshStates */) const {
274 MT_PROFILE_FUNCTION();
275
276 // loop over all joints and check for smoothness
277 double error = 0.0;
278
279 MT_CHECK(skelStates.size() == 2);
280
281 const auto& prevState = skelStates[0];
282 const auto& nextState = skelStates[1];
283
284 // go over the joint states and calculate the rotation difference contribution on joints that are
285 // at the end
286 for (size_t i = 0; i < this->skeleton_.joints.size(); ++i) {
287 // calculate orientation error based on the selected error type
288 T rotationError = T(0);
289
290 const TransformT<T> targetTransform =
291 (i < targetState_.size()) ? targetState_[i] : TransformT<T>();
292 const TransformT<T> transformedPrev = targetTransform * prevState.jointState[i].transform;
293 const Eigen::Quaternion<T>& prevRot = transformedPrev.rotation;
294 const Eigen::Quaternion<T>& nextRot = nextState.jointState[i].rotation();
295
296 switch (rotationErrorType_) {
297 case RotationErrorType::QuaternionLogMap: {
298 // Compute the relative rotation: q_rel = nextRot^{-1} * prevRot
299 const Eigen::Quaternion<T> qRel = nextRot.conjugate() * prevRot;
300
301 // Compute the logmap of the relative rotation
302 const Eigen::Vector3<T> logmapVec = quaternionLogMap(qRel);
303
304 // The error is the squared norm of the logmap
305 rotationError = logmapVec.squaredNorm();
306 break;
307 }
308 case RotationErrorType::RotationMatrixDifference: {
309 // Note: |R0 - RT| is a valid norm on SO3, it doesn't have the same slope as the squared
310 // angle difference but its derivative doesn't have a singularity at the minimum, so is more
311 // stable
312 const Eigen::Matrix3<T> rotDiff = nextRot.toRotationMatrix() - prevRot.toRotationMatrix();
313 rotationError = rotDiff.squaredNorm();
314 break;
315 }
316 }
317
318 error += rotationError * kOrientationWeight * rotWgt_ * targetRotationWeights_[i];
319
320 // calculate position error
321 const Eigen::Vector3<T> diff =
322 nextState.jointState[i].translation() - transformedPrev.translation;
323 error += diff.squaredNorm() * kPositionWeight * posWgt_ * targetPositionWeights_[i];
324 }
325
326 // return error
327 return error * this->weight_;
328}

330template <typename T>
331double StateSequenceErrorFunctionT<T>::getGradient(
332 std::span<const ModelParametersT<T>> /* modelParameters */,
333 std::span<const SkeletonStateT<T>> skelStates,
334 std::span<const MeshStateT<T>> /* meshStates */,
335 Eigen::Ref<Eigen::VectorX<T>> gradient) const {
336 MT_PROFILE_FUNCTION();
337
338 MT_CHECK(skelStates.size() == 2);
339 const auto& prevState = skelStates[0];
340 const auto& nextState = skelStates[1];
341
342 const Eigen::Index nParam = this->parameterTransform_.numAllModelParameters();
343 MT_CHECK(gradient.size() == 2 * nParam);
344
345 // loop over all joints and check for smoothness
346 double error = 0.0;
347
348 // also go over the joint states and calculate the rotation difference contribution on joints that
349 // are at the end
350 for (size_t i = 0; i < this->skeleton_.joints.size(); ++i) {
351 if (targetRotationWeights_[i] == 0 && targetPositionWeights_[i] == 0) {
352 continue;
353 }
354
355 // calculate orientation gradient
356 const TransformT<T> targetTransform =
357 (i < targetState_.size()) ? targetState_[i] : TransformT<T>();
358 const TransformT<T> transformedPrev = targetTransform * prevState.jointState[i].transform;
359 const Eigen::Quaternion<T>& prevRot = transformedPrev.rotation;
360 const Eigen::Quaternion<T>& nextRot = nextState.jointState[i].rotation();
361
362 T rotationError = T(0);
363 Eigen::Matrix3<T> rotDiff; // Only needed for RotationMatrixDifference
364 Eigen::Matrix<T, 3, 4> dLogDq; // Only needed for QuaternionLogMap
365
366 switch (rotationErrorType_) {
367 case RotationErrorType::QuaternionLogMap: {
368 // Compute the relative rotation: q_rel = nextRot^{-1} * prevRot
369 const Eigen::Quaternion<T> qRel = nextRot.conjugate() * prevRot;
370
371 // Precompute the derivative of logmap w.r.t. quaternion components
372 dLogDq = quaternionLogMapDerivative(qRel);
373
374 // Compute the logmap of the relative rotation
375 const Eigen::Vector3<T> logmapVec = quaternionLogMap(qRel);
376
377 // The error is the squared norm of the logmap
378 rotationError = logmapVec.squaredNorm();
379 break;
380 }
381 case RotationErrorType::RotationMatrixDifference: {
382 rotDiff = nextRot.toRotationMatrix() - prevRot.toRotationMatrix();
383 rotationError = rotDiff.squaredNorm();
384 break;
385 }
386 }
387
388 const T rwgt = kOrientationWeight * rotWgt_ * this->weight_ * targetRotationWeights_[i];
389 error += rotationError * rwgt;
390
391 // calculate position gradient
392 const Eigen::Vector3<T> diff =
393 nextState.jointState[i].translation() - transformedPrev.translation;
394 const T pwgt = kPositionWeight * posWgt_ * this->weight_ * targetPositionWeights_[i];
395 error += diff.squaredNorm() * pwgt;
396 // Gradient for prevState
397 {
398 size_t ancestorJointIndex = i;
399 while (ancestorJointIndex != kInvalidIndex) {
400 MT_CHECK(ancestorJointIndex < this->skeleton_.joints.size());
401
402 const size_t paramIndex = ancestorJointIndex * kParametersPerJoint;
403 const Eigen::Vector3<T> posd = prevState.jointState[i].translation() -
404 prevState.jointState[ancestorJointIndex].translation();
405
406 for (size_t d = 0; d < 3; d++) {
407 // position gradient
408 if (this->activeJointParams_[paramIndex + d]) {
409 const T val = -T(2) *
410 diff.dot(targetTransform.rotate(
411 prevState.jointState[ancestorJointIndex].getTranslationDerivative(d))) *
412 pwgt;
413 gradient_jointParams_to_modelParams<T>(
414 val, paramIndex + d, this->parameterTransform_, gradient.segment(0, nParam));
415 }
416 if (this->activeJointParams_[paramIndex + 3 + d]) {
417 const Eigen::Vector3<T> axis =
418 prevState.jointState[ancestorJointIndex].rotationAxis.col(d);
419
420 // (1) Changes in position error due to rotation of ancestor joint
421 const T positionErrorGradient = -T(2) *
422 diff.dot(targetTransform.rotate(
423 prevState.jointState[ancestorJointIndex].getRotationDerivative(d, posd)));
424
425 // (2) Changes in rotation error due to rotation of ancestor joint
426 T rotationErrorGradient = 0;
427 switch (rotationErrorType_) {
428 case RotationErrorType::QuaternionLogMap:
429 rotationErrorGradient = computeLogMapRotationGradientPrev(
430 prevState.jointState[i].rotation(),
431 nextRot,
432 targetTransform.rotation,
433 axis,
434 dLogDq);
435 break;
436 case RotationErrorType::RotationMatrixDifference:
437 rotationErrorGradient = computeMatrixDiffRotationGradientPrev(
438 prevState.jointState[i].rotation(), targetTransform.rotation, axis, rotDiff);
439 break;
440 }
441
442 const T val = pwgt * positionErrorGradient + rwgt * rotationErrorGradient;
443 gradient_jointParams_to_modelParams<T>(
444 val, paramIndex + 3 + d, this->parameterTransform_, gradient.segment(0, nParam));
445 }
446 }
447 if (this->activeJointParams_[paramIndex + 6]) {
448 const T val = -T(2) *
449 diff.dot(targetTransform.rotate(
450 prevState.jointState[ancestorJointIndex].getScaleDerivative(posd))) *
451 pwgt;
452 gradient_jointParams_to_modelParams<T>(
453 val, paramIndex + 6, this->parameterTransform_, gradient.segment(0, nParam));
454 }
455
456 ancestorJointIndex = this->skeleton_.joints[ancestorJointIndex].parent;
457 }
458 }
459
460 // Gradient for nextState
461 {
462 size_t ancestorJointIndex = i;
463 while (ancestorJointIndex != kInvalidIndex) {
464 MT_CHECK(ancestorJointIndex < this->skeleton_.joints.size());
465
466 const size_t paramIndex = ancestorJointIndex * kParametersPerJoint;
467 const Eigen::Vector3<T> posd = nextState.jointState[i].translation() -
468 nextState.jointState[ancestorJointIndex].translation();
469
470 for (size_t d = 0; d < 3; d++) {
471 // position gradient
472 if (this->activeJointParams_[paramIndex + d]) {
473 const T val = T(2) *
474 diff.dot(nextState.jointState[ancestorJointIndex].getTranslationDerivative(d)) *
475 pwgt;
476 gradient_jointParams_to_modelParams<T>(
477 val, paramIndex + d, this->parameterTransform_, gradient.segment(nParam, nParam));
478 }
479 if (this->activeJointParams_[paramIndex + 3 + d]) {
480 const Eigen::Vector3<T> axis =
481 nextState.jointState[ancestorJointIndex].rotationAxis.col(d);
482
483 // (1) Changes in position error due to rotation of ancestor joint
484 const T positionErrorGradient = T(2) *
485 diff.dot(nextState.jointState[ancestorJointIndex].getRotationDerivative(d, posd)) *
486 pwgt;
487
488 // (2) Changes in rotation error due to rotation of ancestor joint
489 T rotationErrorGradient;
490 switch (rotationErrorType_) {
491 case RotationErrorType::QuaternionLogMap:
492 rotationErrorGradient =
493 computeLogMapRotationGradientNext(prevRot, nextRot, axis, dLogDq);
494 break;
495 case RotationErrorType::RotationMatrixDifference:
496 rotationErrorGradient = computeMatrixDiffRotationGradientNext(
497 nextState.jointState[i].rotation(), axis, rotDiff);
498 break;
499 }
500
501 const T val = positionErrorGradient + rwgt * rotationErrorGradient;
502 gradient_jointParams_to_modelParams<T>(
503 val,
504 paramIndex + 3 + d,
505 this->parameterTransform_,
506 gradient.segment(nParam, nParam));
507 }
508 }
509 if (this->activeJointParams_[paramIndex + 6]) {
510 const T val = T(2) *
511 diff.dot(nextState.jointState[ancestorJointIndex].getScaleDerivative(posd)) * pwgt;
512 gradient_jointParams_to_modelParams<T>(
513 val, paramIndex + 6, this->parameterTransform_, gradient.segment(nParam, nParam));
514 }
515
516 ancestorJointIndex = this->skeleton_.joints[ancestorJointIndex].parent;
517 }
518 }
519 }
520
521 // return error
522 return error;
523}

525template <typename T>
526size_t StateSequenceErrorFunctionT<T>::getJacobianSize() const {
527 const auto nActiveJoints =
528 (targetPositionWeights_.array() != 0 || targetRotationWeights_.array() != 0).count();
529 if (rotationErrorType_ == RotationErrorType::QuaternionLogMap) {
530 // 3 (translation) + 3 (logmap) = 6 per joint
531 return nActiveJoints * 6;
532 } else {
533 // 3 (translation) + 9 (matrix) = 12 per joint
534 return nActiveJoints * 12;
535 }
536}

538template <typename T>
539double StateSequenceErrorFunctionT<T>::getJacobian(
540 std::span<const ModelParametersT<T>> /* modelParameters */,
541 std::span<const SkeletonStateT<T>> skelStates,
542 std::span<const MeshStateT<T>> /* meshStates */,
543 Eigen::Ref<Eigen::MatrixX<T>> jacobian_full,
544 Eigen::Ref<Eigen::VectorX<T>> residual_full,
545 int& usedRows) const {
546 MT_PROFILE_FUNCTION();
547
548 MT_CHECK(skelStates.size() == 2);
549 const auto& prevState = skelStates[0];
550 const auto& nextState = skelStates[1];
551
552 const Eigen::Index nParam = this->parameterTransform_.numAllModelParameters();
553 MT_CHECK(jacobian_full.cols() == 2 * nParam);
554
555 double error = 0.0;
556
557 Eigen::Index offset = 0;
558
559 // calculate state difference jacobians
560 for (size_t i = 0; i < this->skeleton_.joints.size(); ++i) {
561 if (targetRotationWeights_[i] == 0 && targetPositionWeights_[i] == 0) {
562 continue;
563 }
564
565 const TransformT<T> targetTransform =
566 (i < targetState_.size()) ? targetState_[i] : TransformT<T>();
567 const TransformT<T> transformedPrev = targetTransform * prevState.jointState[i].transform;
568
569 // Determine block size based on rotation error type
570 // QuaternionLogMap: 3 (translation) + 3 (logmap) = 6
571 // RotationMatrixDifference: 3 (translation) + 9 (matrix) = 12
572 const size_t rotationResidualSize =
573 (rotationErrorType_ == RotationErrorType::QuaternionLogMap) ? 3 : 9;
574 const size_t blockSize = 3 + rotationResidualSize;
575
576 // calculate translation gradient
577 const auto transDiff =
578 (nextState.jointState[i].translation() - transformedPrev.translation).eval();
579
580 // calculate orientation gradient and error
581 const Eigen::Quaternion<T>& prevRot = transformedPrev.rotation;
582 const Eigen::Quaternion<T>& nextRot = nextState.jointState[i].rotation();
583
584 // calculate the difference between target and position and error
585 const T pwgt = kPositionWeight * posWgt_ * this->weight_ * targetPositionWeights_[i];
586 const T wgt = std::sqrt(pwgt);
587 error += transDiff.squaredNorm() * pwgt;
588 residual_full.template segment<3>(offset).noalias() = transDiff * wgt;
589
590 const T rwgt = kOrientationWeight * rotWgt_ * this->weight_ * targetRotationWeights_[i];
591 const T awgt = std::sqrt(rwgt);
592
593 Eigen::Matrix<T, 3, 4> dLogDq; // Precomputed derivative (only needed for logmap)
594 switch (rotationErrorType_) {
595 case RotationErrorType::QuaternionLogMap: {
596 // Compute the relative rotation: q_rel = nextRot^{-1} * prevRot
597 const Eigen::Quaternion<T> qRel = nextRot.conjugate() * prevRot;
598
599 // Precompute the derivative of logmap w.r.t. quaternion components
600 // This will be reused for all 3 rotation axes
601 dLogDq = quaternionLogMapDerivative(qRel);
602
603 // Compute the logmap of the relative rotation
604 const Eigen::Vector3<T> logmapVec = quaternionLogMap(qRel);
605
606 // The error is the squared norm of the logmap
607 error += logmapVec.squaredNorm() * rwgt;
608
609 // The residual for Jacobian is the logmap vector itself (not squared)
610 residual_full.template segment<3>(offset + 3).noalias() = logmapVec * awgt;
611 break;
612 }
613 case RotationErrorType::RotationMatrixDifference: {
614 // 9D RotationMatrixDifference
615 const Eigen::Matrix3<T> rotDiff = nextRot.toRotationMatrix() - prevRot.toRotationMatrix();
616 error += rotDiff.squaredNorm() * rwgt;
617 residual_full.template segment<9>(offset + 3).noalias() =
618 Map<const Eigen::Matrix<T, 9, 1>>(rotDiff.data(), rotDiff.size()) * awgt;
619 break;
620 }
621 }
622
623 // Jacobian for prevState
624 {
625 size_t ancestorJointIndex = i;
626 while (ancestorJointIndex != kInvalidIndex) {
627 MT_CHECK(ancestorJointIndex < this->skeleton_.joints.size());
628
629 const auto& jointState = prevState.jointState[ancestorJointIndex];
630 const size_t paramIndex = ancestorJointIndex * kParametersPerJoint;
631 const Eigen::Vector3<T> posd =
632 prevState.jointState[i].translation() - jointState.translation();
633
634 for (size_t d = 0; d < 3; d++) {
635 if (this->activeJointParams_[paramIndex + d]) {
636 const Eigen::Vector3<T> jc =
637 -(targetTransform.rotate(jointState.getTranslationDerivative(d))) * wgt;
638 jacobian_jointParams_to_modelParams<T>(
639 jc,
640 paramIndex + d,
641 this->parameterTransform_,
642 jacobian_full.block(offset, 0, blockSize, nParam).template topRows<3>());
643 }
644
645 if (this->activeJointParams_[paramIndex + 3 + d]) {
646 const Eigen::Vector3<T> axis =
647 prevState.jointState[ancestorJointIndex].rotationAxis.col(d);
648
649 // (1) Translation part: Changes in position error due to rotation of ancestor joint
650 const Eigen::Vector3<T> jc =
651 -(targetTransform.rotate(jointState.getRotationDerivative(d, posd))) * wgt;
652 jacobian_jointParams_to_modelParams<T>(
653 jc,
654 paramIndex + d + 3,
655 this->parameterTransform_,
656 jacobian_full.block(offset, 0, blockSize, nParam).template topRows<3>());
657
658 // (2) Rotation part: Changes in rotation error due to rotation of ancestor joint
659 switch (rotationErrorType_) {
660 case RotationErrorType::QuaternionLogMap: {
661 const Eigen::Matrix<T, 3, 1> rotJac = computeLogMapRotationJacobianPrev(
662 prevState.jointState[i].rotation(),
663 nextRot,
664 targetTransform.rotation,
665 axis,
666 awgt,
667 dLogDq);
668 jacobian_jointParams_to_modelParams<T>(
669 rotJac,
670 paramIndex + d + 3,
671 this->parameterTransform_,
672 jacobian_full.block(offset, 0, blockSize, nParam)
673 .middleRows(3, rotationResidualSize));
674 break;
675 }
676 case RotationErrorType::RotationMatrixDifference: {
677 const Eigen::Matrix<T, 9, 1> rotJac = computeMatrixDiffRotationJacobianPrev(
678 prevState.jointState[i].rotation(), targetTransform.rotation, axis, awgt);
679 jacobian_jointParams_to_modelParams<T>(
680 rotJac,
681 paramIndex + d + 3,
682 this->parameterTransform_,
683 jacobian_full.block(offset, 0, blockSize, nParam)
684 .middleRows(3, rotationResidualSize));
685 break;
686 }
687 }
688 }
689 }
690
691 if (this->activeJointParams_[paramIndex + 6]) {
692 const Eigen::Vector3<T> jc =
693 -(targetTransform.rotate(jointState.getScaleDerivative(posd))) * wgt;
694 jacobian_jointParams_to_modelParams<T>(
695 jc,
696 paramIndex + 6,
697 this->parameterTransform_,
698 jacobian_full.block(offset, 0, blockSize, nParam).template topRows<3>());
699 }
700
701 ancestorJointIndex = this->skeleton_.joints[ancestorJointIndex].parent;
702 }
703 }
704
705 // Jacobian for nextState
706 {
707 size_t ancestorJointIndex = i;
708 while (ancestorJointIndex != kInvalidIndex) {
709 MT_CHECK(ancestorJointIndex < this->skeleton_.joints.size());
710
711 const auto& jointState = nextState.jointState[ancestorJointIndex];
712 const size_t paramIndex = ancestorJointIndex * kParametersPerJoint;
713 const Eigen::Vector3<T> posd =
714 nextState.jointState[i].translation() - jointState.translation();
715
716 for (size_t d = 0; d < 3; d++) {
717 if (this->activeJointParams_[paramIndex + d]) {
718 const Eigen::Vector3<T> jc = jointState.getTranslationDerivative(d) * wgt;
719 jacobian_jointParams_to_modelParams<T>(
720 jc,
721 paramIndex + d,
722 this->parameterTransform_,
723 jacobian_full.block(offset, nParam, blockSize, nParam).template topRows<3>());
724 }
725
726 if (this->activeJointParams_[paramIndex + 3 + d]) {
727 const Eigen::Vector3<T> axis =
728 nextState.jointState[ancestorJointIndex].rotationAxis.col(d);
729
730 // (1) Translation part: Changes in position error due to rotation of ancestor joint
731 const Eigen::Vector3<T> jc = jointState.getRotationDerivative(d, posd) * wgt;
732 jacobian_jointParams_to_modelParams<T>(
733 jc,
734 paramIndex + d + 3,
735 this->parameterTransform_,
736 jacobian_full.block(offset, nParam, blockSize, nParam).template topRows<3>());
737
738 // (2) Rotation part: Changes in rotation error due to rotation of ancestor joint
739 switch (rotationErrorType_) {
740 case RotationErrorType::QuaternionLogMap: {
741 const Eigen::Matrix<T, 3, 1> rotJac =
742 computeLogMapRotationJacobianNext(prevRot, nextRot, axis, awgt, dLogDq);
743 jacobian_jointParams_to_modelParams<T>(
744 rotJac,
745 paramIndex + d + 3,
746 this->parameterTransform_,
747 jacobian_full.block(offset, nParam, blockSize, nParam)
748 .middleRows(3, rotationResidualSize));
749 break;
750 }
751 case RotationErrorType::RotationMatrixDifference: {
752 const Eigen::Matrix<T, 9, 1> rotJac = computeMatrixDiffRotationJacobianNext(
753 nextState.jointState[i].rotation(), axis, awgt);
754 jacobian_jointParams_to_modelParams<T>(
755 rotJac,
756 paramIndex + d + 3,
757 this->parameterTransform_,
758 jacobian_full.block(offset, nParam, blockSize, nParam)
759 .middleRows(3, rotationResidualSize));
760 break;
761 }
762 }
763 }
764 }
765
766 if (this->activeJointParams_[paramIndex + 6]) {
767 const Eigen::Vector3<T> jc = jointState.getScaleDerivative(posd) * wgt;
768 jacobian_jointParams_to_modelParams<T>(
769 jc,
770 paramIndex + 6,
771 this->parameterTransform_,
772 jacobian_full.block(offset, nParam, blockSize, nParam).template topRows<3>());
773 }
774
775 ancestorJointIndex = this->skeleton_.joints[ancestorJointIndex].parent;
776 }
777 }
778
779 offset += blockSize;
780 }
781
782 usedRows = gsl::narrow_cast<int>(offset);
783
784 // return error
785 return error;
786}
