149std::tuple<momentum::Character, momentum::ModelParameters> loadCalibratedModel(

151 auto [c, m, id, fps] = loadCharacterWithMotion(modelFile);
152
153 // make sure we actually only convert joint id back to scale id parameters
154 const ModelParameters identity =

160std::tuple<momentum::Character, momentum::ModelParameters> loadCharacterWithIdentity(
161 const ModelOptions& modelFiles) {
162 Character character;
163 ModelParameters identity;
164
165 // TODO: Very hacky!
166 if (modelFiles.model.find(".glb") != std::string::npos &&
167 modelFiles.parameters
168 .empty()) { // Assume input has precomputed proportions and marker offsets etc.
169 auto [c, m, id, fps] = loadCharacterWithMotion(modelFiles.model);
170 character = c;
171 if (id.size() > 0) {
172 identity = jointIdentityToModelIdentity(c, c.parameterTransform.getScalingParameters(), id);
173 } else {
174 identity = ModelParameters::Zero(character.parameterTransform.numAllModelParameters());
175 }
176
177 if (!modelFiles.locators.empty()) {
178 character.locators =
179 loadLocators(modelFiles.locators, character.skeleton, character.parameterTransform);
180 }
181 } else { // Assume input is the template
182 character = loadFullCharacter(modelFiles.model, modelFiles.parameters, modelFiles.locators);
183 identity = ModelParameters::Zero(character.parameterTransform.numAllModelParameters());
184 }
185
186 return {character, identity};
187}

189void saveMotion(
190 const std::string& outFile,
191 const momentum::Character& character,
192 const momentum::ModelParameters& identity,
193 Eigen::MatrixXf& finalMotion,
194 std::span<const std::vector<momentum::Marker>> markerData,
195 const double fps,
196 const bool saveMarkerMesh) {
197 const filesystem::path output(outFile);
198 const auto ext = output.extension();
199
200 ModelParameters id =
201 extractParameters(identity, character.parameterTransform.getScalingParameters());
202
203 // gltf io assumes the identity info is removed from the motion matrix
204 removeIdentity(character.parameterTransform.getScalingParameters(), id, finalMotion);
205 const VectorXf idVec = character.parameterTransform.apply(id).v;
206
207 if (ext == ".fbx") {
208 saveFbx(output, character, finalMotion, idVec, fps);
209 } else if (ext == ".glb" || ext == ".gltf") {
210 GltfBuilder fileBuilder;
211 fileBuilder.addMotion(
212 character,
213 fps,
214 std::make_tuple(character.parameterTransform.name, finalMotion),
215 std::make_tuple(character.skeleton.getJointNames(), idVec));
216 fileBuilder.addMarkerSequence(
217 fps,
218 markerData,
219 saveMarkerMesh ? GltfBuilder::MarkerMesh::UnitCube : GltfBuilder::MarkerMesh::None);
220 fileBuilder.save(outFile);
221 }
222}
