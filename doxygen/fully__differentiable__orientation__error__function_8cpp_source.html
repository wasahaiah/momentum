52template <typename T>
53double FullyDifferentiableOrientationErrorFunctionT<T>::getError(
54 const ModelParametersT<T>& /*params*/,
55 const SkeletonStateT<T>& state,
56 const MeshStateT<T>& /* meshState */) {
57 // loop over all constraints and calculate the error
58 double error = 0.0;
59
60 for (size_t i = 0; i < constraints_.size(); ++i) {
61 const OrientationConstraintT<T>& ct = constraints_[i];
62
63 // calculate orientation error
64 // Note: |R0 - RT|Â² is a valid norm on SO3, it doesn't have the same slope as the squared
65 // angle difference
66 // but it's derivative doesn't have a singularity at the minimum, so is more stable
67 const Eigen::Quaternion<T> rot = state.jointState[ct.parent].rotation() * ct.offset;
68 const Eigen::Matrix3<T> rotDiff = rot.toRotationMatrix() - ct.target.toRotationMatrix();
69 error += ct.weight * rotDiff.squaredNorm() * kOrientationWeight;
70 }
71
72 // return error
73 return error * this->weight_;
74}

76template <typename T>
77double FullyDifferentiableOrientationErrorFunctionT<T>::getGradient(
78 const ModelParametersT<T>& /*params*/,
79 const SkeletonStateT<T>& state,
80 const MeshStateT<T>& /* meshState */,
81 Ref<Eigen::VectorX<T>> gradient) {
82 const auto& parameterTransform = this->parameterTransform_;
83
84 // storage for joint gradients
85 jointGrad_.setZero(parameterTransform.numAllModelParameters());
86
87 // loop over all constraints and calculate the error
88 double error = 0.0;
89
90 for (size_t i = 0; i < constraints_.size(); ++i) {
91 const OrientationConstraintT<T>& ct = constraints_[i];
92 if (ct.weight == 0) {
93 continue;
94 }
95
96 error += calculateOrientationGradient(state, constraints_[i], jointGrad_);
97 }
98
99 // convert joint gradient to parameter gradient and return
100 gradient.noalias() += jointGrad_ * this->weight_;
101
102 // return error
103 return error * this->weight_;
104}

106template <typename T>
107double FullyDifferentiableOrientationErrorFunctionT<T>::getJacobian(
108 const ModelParametersT<T>& params,
109 const SkeletonStateT<T>& state,
110 const MeshStateT<T>& /* meshState */,
111 Ref<Eigen::MatrixX<T>> jacobian,
112 Ref<Eigen::VectorX<T>> residual,
113 int& usedRows) {
114 // loop over all constraints and calculate the error
115 double error = 0.0;
116 std::atomic_int offset(0);
117
118 for (size_t i = 0; i < constraints_.size(); ++i) {
119 const OrientationConstraintT<T>& ct = constraints_[i];
120 if (ct.weight == 0) {
121 continue;
122 }
123
124 const auto currentOffset = offset.fetch_add(9, std::memory_order_relaxed);
125 error += calculateOrientationJacobian(
126 state,
127 constraints_[i],
128 jacobian.block(currentOffset, 0, 9, params.size()),
129 residual.middleRows(currentOffset, 9));
130 }
131
132 usedRows = gsl::narrow_cast<int>(jacobian.rows());
133
134 return error;
135}

137template <typename T>
138T FullyDifferentiableOrientationErrorFunctionT<T>::calculateOrientationJacobian(
139 const SkeletonStateT<T>& state,
140 const OrientationConstraintT<T>& constr,
141 Ref<Eigen::MatrixX<T>> jac,
142 Ref<Eigen::VectorX<T>> res) const {
143 const auto& parameterTransform = this->parameterTransform_;
144
145 // calculate orientation error
146 const Eigen::Quaternion<T> rot = state.jointState[constr.parent].rotation() * constr.offset;
147 const Eigen::Quaternion<T>& target = constr.target;
148 const Eigen::Matrix3<T> rotDiff = rot.toRotationMatrix() - target.toRotationMatrix();
149 const T wgt = kOrientationWeight * constr.weight * this->weight_;
150 const T swgt = std::sqrt(wgt);
151
152 // loop over all joints the constraint is attached to and calculate gradient
153 size_t jointIndex = constr.parent;
154 while (jointIndex != kInvalidIndex) {
155 // check for valid index
156 MT_CHECK(jointIndex < this->skeleton_.joints.size());
157
158 const size_t paramIndex = jointIndex * kParametersPerJoint;
159
160 // calculate derivatives based on active joints
161 for (size_t d = 0; d < 3; d++) {
162 if (this->activeJointParams_[paramIndex + 3 + d]) {
163 const Eigen::Vector3<T> axis = state.jointState[jointIndex].rotationAxis.col(d);
164 const Eigen::Matrix3<T> rotD = crossProductMatrix(axis) * rot * swgt;
165 const auto ja = Map<const Eigen::VectorX<T>>(rotD.data(), rotD.size());
166
167 // explicitly multiply with the parameter transform to generate parameter space gradients
168 for (auto index = parameterTransform.transform.outerIndexPtr()[paramIndex + 3 + d];
169 index < parameterTransform.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
170 ++index) {
171 jac.col(parameterTransform.transform.innerIndexPtr()[index]) +=
172 ja * parameterTransform.transform.valuePtr()[index];
173 }
174 }
175 }
176
177 // go to the next joint
178 jointIndex = this->skeleton_.joints[jointIndex].parent;
179 }
180
181 res = Map<const Eigen::VectorX<T>>(rotDiff.data(), rotDiff.size()) * swgt;
182
183 return rotDiff.squaredNorm() * wgt;
184}

218template <typename T>
219void FullyDifferentiableOrientationErrorFunctionT<T>::setInputImp(
220 const std::string& name,
221 Eigen::Ref<const Eigen::VectorX<T>> value) {
222 // Normalization is a tricky topic here. When we set the values, we must normalize normals and
223 // quaternions etc because the rest of the code won't work correctly without this (for example
224 // computing the angle between unnormalized vectors requires normalizing them first). However,
225 // if we do normalize the values that are passed in, then the derivatives of those values are
226 // no longer correct. For example, imagine that we pass in v = (1000, 0, 0) as the normal to a
227 // plane. Clearly we must turn this into (1, 0, 0) so the IK solve does something reasonable, but
228 // when we do we find that our dv/dTheta is off by a factor of 1000. It's hard to address that
229 // inside this code, the right way to do it is to require that the inputs be normalized inside
230 // Pytorch before getting passed in, which will ensure all the derivatives work out correctly.
231 if (name == kWeights) {
232 for (size_t i = 0; i < constraints_.size(); ++i) {
233 constraints_[i].weight = static_cast<float>(value[i]);
234 }
235 } else if (name == kOffsets) {
236 for (size_t i = 0; i < constraints_.size(); ++i) {
237 constraints_[i].offset = Eigen::Quaternion<T>(value.template segment<4>(4 * i)).normalized();
238 }
239 } else if (name == kTargets) {
240 for (size_t i = 0; i < constraints_.size(); ++i) {
241 constraints_[i].target = Eigen::Quaternion<T>(value.template segment<4>(4 * i)).normalized();
242 }
243 } else {
244 MT_THROW(
245 "Unknown input to FullyDifferentiableOrientationErrorFunctionT<T>::getInput: {}", name);
246 }

249template <typename T>
250T FullyDifferentiableOrientationErrorFunctionT<T>::calculateOrientationGradient(
251 const SkeletonStateT<T>& state,
252 const OrientationConstraintT<T>& constr,
253 Eigen::VectorX<T>& jGrad) const {
254 const auto& parameterTransform = this->parameterTransform_;
255
256 // calculate orientation error
257 const Eigen::Quaternion<T> rot = state.jointState[constr.parent].rotation() * constr.offset;
258 const Eigen::Matrix3<T> rotDiff = rot.toRotationMatrix() - constr.target.toRotationMatrix();
259 const T wgt = kOrientationWeight * constr.weight;
260
261 // loop over all joints the constraint is attached to and calculate gradient
262 size_t jointIndex = constr.parent;
263 while (jointIndex != kInvalidIndex) {
264 // check for valid index
265 MT_CHECK(jointIndex < this->skeleton_.joints.size());
266
267 const size_t paramIndex = jointIndex * kParametersPerJoint;
268
269 // calculate derivatives based on active joints
270 for (size_t d = 0; d < 3; d++) {
271 if (this->activeJointParams_[paramIndex + 3 + d]) {
272 // calculate joint gradient
273 const Eigen::Vector3<T> axis = state.jointState[jointIndex].rotationAxis.col(d);
274 const auto rotD = crossProductMatrix(axis) * rot;
275 const T val = 2.0f * wgt * rotD.cwiseProduct(rotDiff).sum();
276 // explicitly multiply with the parameter transform to generate parameter space gradients
277 for (auto index = parameterTransform.transform.outerIndexPtr()[paramIndex + 3 + d];
278 index < parameterTransform.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
279 ++index) {
280 jGrad[parameterTransform.transform.innerIndexPtr()[index]] +=
281 val * parameterTransform.transform.valuePtr()[index];
282 }
283 }
284 }
285
286 // go to the next joint
287 jointIndex = this->skeleton_.joints[jointIndex].parent;
288 }
289
290 return rotDiff.squaredNorm() * wgt;
291}

293template <typename T>
294template <typename JetType>
295[[nodiscard]] JetType
296FullyDifferentiableOrientationErrorFunctionT<T>::calculateOrientationGradient_dot(
297 const SkeletonStateT<T>& state,
298 const size_t constrParent,
299 const JetType& constrWeight,
300 const Eigen::Quaternion<JetType>& constrOrientationOffset,
301 const Eigen::Quaternion<JetType>& constrOrientationTarget,
302 Eigen::Ref<const Eigen::VectorX<T>> vec) const {
303 JetType result;
304
305 // calculate orientation error
306 const Eigen::Quaternion<JetType> rot =
307 (state.jointState[constrParent].rotation().template cast<JetType>() *
308 constrOrientationOffset);
309 const Eigen::Matrix3<JetType> rotDiff =
310 rot.toRotationMatrix() - constrOrientationTarget.toRotationMatrix();
311 const JetType wgt = this->weight_ * this->kOrientationWeight * constrWeight;
312
313 // loop over all joints the constraint is attached to and calculate gradient
314 size_t jointIndex = constrParent;
315 while (jointIndex != kInvalidIndex) {
316 // check for valid index
317 assert(jointIndex < this->skeleton_.joints.size());
318
319 const size_t paramIndex = jointIndex * kParametersPerJoint;
320
321 // calculate derivatives based on active joints
322 for (size_t d = 0; d < 3; d++) {
323 if (this->activeJointParams_[paramIndex + 3 + d]) {
324 // calculate joint gradient
325 const Eigen::Vector3<T> axis = state.jointState[jointIndex].rotationAxis.col(d);
326 const Eigen::Matrix3<JetType> rotD = crossProductMatrix(axis) * rot;
327 const JetType val = T(2) * wgt * rotD.cwiseProduct(rotDiff).sum();
328 result += times_parameterTransform_times_v(
329 val, paramIndex + 3 + d, this->parameterTransform_, vec);
330 }
331 }
332
333 // go to the next joint
334 jointIndex = this->skeleton_.joints[jointIndex].parent;
335 }
336
337 return result;
338}

340template <typename T>
341Eigen::VectorX<T> FullyDifferentiableOrientationErrorFunctionT<T>::d_gradient_d_input_dot(
342 const std::string& inputName,
343 const ModelParametersT<T>& /*modelParams*/,
344 const SkeletonStateT<T>& state,
345 Eigen::Ref<const Eigen::VectorX<T>> inputVec) {
346 // Use automatic differentiation on the gradient function:
347 if (inputName == kWeights) {
348 Eigen::VectorX<T> result = Eigen::VectorX<T>::Zero(constraints_.size());
349 using JetType = ceres::Jet<T, 1>;
350 for (size_t iCons = 0; iCons < constraints_.size(); ++iCons) {
351 const auto& cons = constraints_[iCons];
352 result(iCons) = calculateOrientationGradient_dot<JetType>(
353 state,
354 cons.parent,
355 JetType(cons.weight, 0),
356 cons.offset.template cast<JetType>(),
357 cons.target.template cast<JetType>(),
358 inputVec)
359 .v[0];
360 }
361 return result;
362 } else if (inputName == kOffsets) {
363 Eigen::VectorX<T> result = Eigen::VectorX<T>::Zero(4 * constraints_.size());
364 using JetType = ceres::Jet<T, 4>;
365 for (size_t iCons = 0; iCons < constraints_.size(); ++iCons) {
366 const auto& cons = constraints_[iCons];
367 const auto dotProd = calculateOrientationGradient_dot<JetType>(
368 state,
369 cons.parent,
370 JetType(cons.weight),
371 Eigen::Quaternion<JetType>(buildJetVec<T, 4>(cons.offset.coeffs())).normalized(),
372 cons.target.template cast<JetType>(),
373 inputVec);
374 result.template segment<4>(4 * iCons) = dotProd.v;
375 }
376 return result;
377 } else if (inputName == kTargets) {
378 Eigen::VectorX<T> result = Eigen::VectorX<T>::Zero(4 * constraints_.size());
379 using JetType = ceres::Jet<T, 4>;
380 for (size_t iCons = 0; iCons < constraints_.size(); ++iCons) {
381 const auto& cons = constraints_[iCons];
382 const auto dotProd = calculateOrientationGradient_dot<JetType>(
383 state,
384 cons.parent,
385 JetType(cons.weight),
386 cons.offset.template cast<JetType>(),
387 Eigen::Quaternion<JetType>(buildJetVec<T, 4>(cons.target.coeffs())).normalized(),
388 inputVec);
389 result.template segment<4>(4 * iCons) = dotProd.v;
390 }
391 return result;
392 } else {
393 MT_THROW(
394 "Unknown input name in FullyDifferentiableOrientationErrorFunctionT<T>::d_gradient_d_input_dot: {}",
395 inputName);
396 }
