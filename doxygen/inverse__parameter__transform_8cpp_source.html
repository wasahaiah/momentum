29template <typename T>
30CharacterParametersT<T> InverseParameterTransformT<T>::apply(
31 const JointParametersT<T>& jointParameters) const {
32 MT_CHECK(
33 jointParameters.size() == transform.rows(),
34 "{} is not {}",
35 jointParameters.size(),
36 transform.rows());
37 MT_CHECK(offsets.size() == transform.rows(), "{} is not {}", offsets.size(), transform.rows());
38 CharacterParametersT<T> result;
39 result.pose = inverseTransform.solve(jointParameters.v - offsets);
40 result.offsets = jointParameters.v - transform * result.pose.v;
41 return result;
42}
