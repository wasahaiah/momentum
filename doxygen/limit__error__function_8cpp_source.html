38template <typename T>
39double LimitErrorFunctionT<T>::getError(
40 const ModelParametersT<T>& params,
41 const SkeletonStateT<T>& state,
42 const MeshStateT<T>& /* meshState */) {
43 MT_PROFILE_FUNCTION();
44
45 // check all is valid
46 MT_CHECK(
47 state.jointParameters.size() ==
48 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
49
50 // loop over all joints and check for limit violations
51 double error = 0.0;
52
53 // ---------------------------------------
54 // new joint error
55 // ---------------------------------------
56 for (const auto& limit : limits_) {
57 switch (limit.type) {
58 case MinMax: {
59 const auto& data = limit.data.minMax;
60 MT_CHECK(data.parameterIndex <= static_cast<size_t>(params.size()));
61 if (this->enabledParameters_.test(data.parameterIndex)) {
62 if (params(data.parameterIndex) < data.limits[0]) {
63 const T val = data.limits[0] - params(data.parameterIndex);
64 error += val * val * limit.weight;
65 }
66 if (params(data.parameterIndex) > data.limits[1]) {
67 const T val = data.limits[1] - params(data.parameterIndex);
68 error += val * val * limit.weight;
69 }
70 }
71 break;
72 }
73 case MinMaxJoint: {
74 const auto& data = limit.data.minMaxJoint;
75 const size_t parameterIndex = data.jointIndex * kParametersPerJoint + data.jointParameter;
76 MT_CHECK(parameterIndex < (size_t)state.jointParameters.size());
77 if (this->activeJointParams_[parameterIndex]) {
78 if (state.jointParameters(parameterIndex) < data.limits[0]) {
79 const T val = data.limits[0] - state.jointParameters[parameterIndex];
80 error += val * val * limit.weight;
81 }
82 if (state.jointParameters(parameterIndex) > data.limits[1]) {
83 const T val = data.limits[1] - state.jointParameters[parameterIndex];
84 error += val * val * limit.weight;
85 }
86 }
87 break;
88 }
89 case MinMaxJointPassive: {
90 break;
91 }
92 case Linear: {
93 const auto& data = limit.data.linear;
94 MT_CHECK(data.referenceIndex < static_cast<size_t>(params.size()));
95 MT_CHECK(data.targetIndex < static_cast<size_t>(params.size()));
96
97 if ((this->enabledParameters_.test(data.targetIndex) ||
98 this->enabledParameters_.test(data.referenceIndex)) &&
99 isInRange(data, params(data.targetIndex))) {
100 const T residual =
101 params(data.targetIndex) * data.scale - data.offset - params(data.referenceIndex);
102 error += residual * residual * limit.weight;
103 }
104 break;
105 }
106 case LinearJoint: {
107 const auto& data = limit.data.linearJoint;
108 const size_t referenceParameterIndex =
109 data.referenceJointIndex * kParametersPerJoint + data.referenceJointParameter;
110 const size_t targetParameterIndex =
111 data.targetJointIndex * kParametersPerJoint + data.targetJointParameter;
112 MT_CHECK(referenceParameterIndex < (size_t)state.jointParameters.size());
113 MT_CHECK(targetParameterIndex < (size_t)state.jointParameters.size());
114
115 if ((this->activeJointParams_[referenceParameterIndex] ||
116 this->activeJointParams_[targetParameterIndex]) &&
117 isInRange(data, state.jointParameters[targetParameterIndex])) {
118 const T residual = state.jointParameters(targetParameterIndex) * data.scale -
119 data.offset - state.jointParameters(referenceParameterIndex);
120 error += residual * residual * limit.weight;
121 }
122 break;
123 }
124
125 case HalfPlane: {
126 const auto& data = limit.data.halfPlane;
127 MT_CHECK(data.param1 < static_cast<size_t>(params.size()));
128 MT_CHECK(data.param2 < static_cast<size_t>(params.size()));
129
130 if ((this->enabledParameters_.test(data.param1) ||
131 this->enabledParameters_.test(data.param2))) {
132 const Eigen::Vector2<T> p(params(data.param1), params(data.param2));
133 const T residual = p.dot(data.normal.template cast<T>()) - data.offset;
134 if (residual < 0) {
135 error += residual * residual * limit.weight;
136 }
137 }
138 break;
139 }
140 case Ellipsoid: {
141 const auto& ct = limit.data.ellipsoid;
142
143 // get the constraint position in global space
144 const Eigen::Vector3<T> position =
145 state.jointState[ct.parent].transform * ct.offset.template cast<T>();
146
147 // get the constraint position in local ellipsoid space
148 const Eigen::Vector3<T> localPosition =
149 state.jointState[ct.ellipsoidParent].transform.inverse() * position;
150
151 // calculate constraint position in ellipsoid space
152 const Eigen::Vector3<T> ellipsoidPosition =
153 ct.ellipsoidInv.template cast<T>() * localPosition;
154
155 // project onto closest surface point
156 const Eigen::Vector3<T> normalizedPosition = ellipsoidPosition.normalized();
157
158 // go back to ellipsoid frame
159 const Eigen::Vector3<T> projectedPosition =
160 ct.ellipsoid.template cast<T>() * normalizedPosition;
161
162 // calculate the difference between projected position and actual position
163 const Eigen::Vector3<T> diff =
164 position - state.jointState[ct.ellipsoidParent].transform * projectedPosition;
165
166 error += diff.squaredNorm() * kPositionWeight * limit.weight;
167 break;
168 }
169 default:
170 // should never get here
171 MT_THROW("Unknown parameter type for joint limit");
172 break;
173 }
174 }
175
176 // return error
177 return error * kLimitWeight * this->weight_;
178}

180template <typename T>
181double LimitErrorFunctionT<T>::getGradient(
182 const ModelParametersT<T>& params,
183 const SkeletonStateT<T>& state,
184 const MeshStateT<T>& /* meshState */,
185 Ref<Eigen::VectorX<T>> gradient) {
186 MT_PROFILE_FUNCTION();
187
188 const auto& parameterTransform = this->parameterTransform_;
189
190 // loop over all joints and check for limit violations
191 double error = 0.0;
192
193 const T tWeight = kLimitWeight * this->weight_;
194
195 // ---------------------------------------
196 // new joint error
197 // ---------------------------------------
198 for (const auto& limit : limits_) {
199 switch (limit.type) {
200 case MinMax: {
201 const auto& data = limit.data.minMax;
202 MT_CHECK(data.parameterIndex < static_cast<size_t>(params.size()));
203 if (this->enabledParameters_.test(data.parameterIndex)) {
204 if (params(data.parameterIndex) < data.limits[0]) {
205 const T val = params(data.parameterIndex) - data.limits[0];
206 error += val * val * tWeight * limit.weight;
207 gradient[data.parameterIndex] += T(2) * val * tWeight * limit.weight;
208 }
209 if (params(data.parameterIndex) > data.limits[1]) {
210 const T val = params(data.parameterIndex) - data.limits[1];
211 error += val * val * tWeight * limit.weight;
212 gradient[data.parameterIndex] += T(2) * val * tWeight * limit.weight;
213 }
214 }
215 break;
216 }
217 case MinMaxJoint: {
218 const auto& data = limit.data.minMaxJoint;
219 const size_t parameterIndex = data.jointIndex * kParametersPerJoint + data.jointParameter;
220 MT_CHECK(parameterIndex < (size_t)state.jointParameters.size());
221 if (this->activeJointParams_[parameterIndex]) {
222 if (state.jointParameters(parameterIndex) < data.limits[0]) {
223 const T val = state.jointParameters[parameterIndex] - data.limits[0];
224 error += val * val * tWeight * limit.weight;
225 // explicitly multiply joint gradient with the parameter transform to generate parameter
226 // space gradients
227 const T jGrad = T(2) * val * tWeight * limit.weight;
228 for (auto index = parameterTransform.transform.outerIndexPtr()[parameterIndex];
229 index < parameterTransform.transform.outerIndexPtr()[parameterIndex + 1];
230 ++index) {
231 gradient[parameterTransform.transform.innerIndexPtr()[index]] +=
232 jGrad * parameterTransform.transform.valuePtr()[index];
233 }
234 }
235 if (state.jointParameters(parameterIndex) > data.limits[1]) {
236 const T val = state.jointParameters[parameterIndex] - data.limits[1];
237 error += val * val * tWeight * limit.weight;
238 // explicitly multiply joint gradient with the parameter transform to generate parameter
239 // space gradients
240 const T jGrad = T(2) * val * tWeight * limit.weight;
241 for (auto index = parameterTransform.transform.outerIndexPtr()[parameterIndex];
242 index < parameterTransform.transform.outerIndexPtr()[parameterIndex + 1];
243 ++index) {
244 gradient[parameterTransform.transform.innerIndexPtr()[index]] +=
245 jGrad * parameterTransform.transform.valuePtr()[index];
246 }
247 }
248 }
249 break;
250 }
251 case MinMaxJointPassive: {
252 break;
253 }
254 case Linear: {
255 const auto& data = limit.data.linear;
256 MT_CHECK(data.referenceIndex < static_cast<size_t>(params.size()));
257 MT_CHECK(data.targetIndex < static_cast<size_t>(params.size()));
258 if (isInRange(data, params(data.targetIndex))) {
259 const T residual =
260 params(data.targetIndex) * data.scale - data.offset - params(data.referenceIndex);
261 error += residual * residual * limit.weight * tWeight;
262
263 if (this->enabledParameters_.test(data.targetIndex)) {
264 gradient[data.targetIndex] += T(2) * residual * data.scale * limit.weight * tWeight;
265 }
266 if (this->enabledParameters_.test(data.referenceIndex)) {
267 gradient[data.referenceIndex] -= T(2) * residual * limit.weight * tWeight;
268 }
269 }
270 break;
271 }
272 case LinearJoint: {
273 const auto& data = limit.data.linearJoint;
274 const size_t referenceParameterIndex =
275 data.referenceJointIndex * kParametersPerJoint + data.referenceJointParameter;
276 const size_t targetParameterIndex =
277 data.targetJointIndex * kParametersPerJoint + data.targetJointParameter;
278 MT_CHECK(referenceParameterIndex < (size_t)state.jointParameters.size());
279 MT_CHECK(targetParameterIndex < (size_t)state.jointParameters.size());
280
281 if ((this->activeJointParams_[referenceParameterIndex] ||
282 this->activeJointParams_[targetParameterIndex]) &&
283 isInRange(data, state.jointParameters[targetParameterIndex])) {
284 const T residual = state.jointParameters(targetParameterIndex) * data.scale -
285 data.offset - state.jointParameters(referenceParameterIndex);
286 error += residual * residual * limit.weight * tWeight;
287
288 // explicitly multiply joint gradient with the parameter transform to generate parameter
289 // space gradients
290 const T targetGrad = T(2) * residual * data.scale * limit.weight * tWeight;
291 for (auto index = parameterTransform.transform.outerIndexPtr()[targetParameterIndex];
292 index < parameterTransform.transform.outerIndexPtr()[targetParameterIndex + 1];
293 ++index) {
294 gradient[parameterTransform.transform.innerIndexPtr()[index]] +=
295 targetGrad * parameterTransform.transform.valuePtr()[index];
296 }
297
298 const T referenceGrad = T(-2) * residual * limit.weight * tWeight;
299 for (auto index = parameterTransform.transform.outerIndexPtr()[referenceParameterIndex];
300 index < parameterTransform.transform.outerIndexPtr()[referenceParameterIndex + 1];
301 ++index) {
302 gradient[parameterTransform.transform.innerIndexPtr()[index]] +=
303 referenceGrad * parameterTransform.transform.valuePtr()[index];
304 }
305 }
306 break;
307 }
308 case HalfPlane: {
309 const auto& data = limit.data.halfPlane;
310 if ((this->enabledParameters_.test(data.param1) ||
311 this->enabledParameters_.test(data.param2))) {
312 const Eigen::Vector2<T> p(params(data.param1), params(data.param2));
313 const T residual = p.dot(data.normal.template cast<T>()) - data.offset;
314 if (residual < 0.0f) {
315 error += residual * residual * limit.weight * tWeight;
316
317 if (this->enabledParameters_.test(data.param1)) {
318 gradient[data.param1] += T(2) * residual * data.normal[0] * tWeight;
319 }
320 if (this->enabledParameters_.test(data.param2)) {
321 gradient[data.param2] += T(2) * residual * data.normal[1] * tWeight;
322 }
323 }
324 }
325 break;
326 }
327 case Ellipsoid: {
328 // NOTE: The gradient for these is currently simplified
329 // It assumes the ellipsoid is static and doesn't move with the parent joint
330 const auto& ct = limit.data.ellipsoid;
331
332 // get the constraint position in global space
333 const Eigen::Vector3<T> position =
334 state.jointState[ct.parent].transform * ct.offset.template cast<T>();
335
336 // get the constraint position in local ellipsoid space
337 const Eigen::Vector3<T> localPosition =
338 state.jointState[ct.ellipsoidParent].transform.inverse() * position;
339
340 // calculate constraint position in ellipsoid space
341 const Eigen::Vector3<T> ellipsoidPosition =
342 ct.ellipsoidInv.template cast<T>() * localPosition;
343
344 // project onto closest surface point
345 const Eigen::Vector3<T> normalizedPosition = ellipsoidPosition.normalized();
346
347 // go back to ellipsoid frame
348 const Eigen::Vector3<T> projectedPosition =
349 ct.ellipsoid.template cast<T>() * normalizedPosition;
350
351 // calculate the difference between projected position and actual position
352 const Eigen::Vector3<T> diff =
353 position - state.jointState[ct.ellipsoidParent].transform * projectedPosition;
354 const T wgt = T(2) * kPositionWeight * limit.weight * tWeight;
355
356 // loop over all joints the constraint is attached to and calculate gradient
357 size_t jointIndex = ct.parent;
358 while (jointIndex != ct.ellipsoidParent && jointIndex != kInvalidIndex) {
359 // check for valid index
360 MT_CHECK(jointIndex < this->skeleton_.joints.size());
361
362 const auto& jointState = state.jointState[jointIndex];
363 const size_t paramIndex = jointIndex * kParametersPerJoint;
364 const Eigen::Vector3<T> posd = position - jointState.translation();
365
366 // calculate derivatives based on active joints
367 for (size_t d = 0; d < 3; d++) {
368 if (this->activeJointParams_[paramIndex + d]) {
369 // calculate joint gradient
370 const T val = diff.dot(jointState.getTranslationDerivative(d)) * wgt;
371 // explicitly multiply with the parameter transform to generate parameter space
372 // gradients
373 for (auto index = parameterTransform.transform.outerIndexPtr()[paramIndex + d];
374 index < parameterTransform.transform.outerIndexPtr()[paramIndex + d + 1];
375 ++index) {
376 gradient[parameterTransform.transform.innerIndexPtr()[index]] +=
377 val * parameterTransform.transform.valuePtr()[index];
378 }
379 }
380 if (this->activeJointParams_[paramIndex + 3 + d]) {
381 // calculate joint gradient
382 const T val = diff.dot(jointState.getRotationDerivative(d, posd)) * wgt;
383 // explicitly multiply with the parameter transform to generate parameter space
384 // gradients
385 for (auto index = parameterTransform.transform.outerIndexPtr()[paramIndex + 3 + d];
386 index < parameterTransform.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
387 ++index) {
388 gradient[parameterTransform.transform.innerIndexPtr()[index]] +=
389 val * parameterTransform.transform.valuePtr()[index];
390 }
391 }
392 }
393 if (this->activeJointParams_[paramIndex + 6]) {
394 // calculate joint gradient
395 const T val = diff.dot(jointState.getScaleDerivative(posd)) * wgt;
396 // explicitly multiply with the parameter transform to generate parameter space
397 // gradients
398 for (auto index = parameterTransform.transform.outerIndexPtr()[paramIndex + 6];
399 index < parameterTransform.transform.outerIndexPtr()[paramIndex + 6 + 1];
400 ++index) {
401 gradient[parameterTransform.transform.innerIndexPtr()[index]] +=
402 val * parameterTransform.transform.valuePtr()[index];
403 }
404 }
405
406 // go to the next joint
407 jointIndex = this->skeleton_.joints[jointIndex].parent;
408 }
409
410 error += diff.squaredNorm() * kPositionWeight * tWeight * limit.weight;
411 break;
412 }
413 default:
414 // should never get here
415 MT_THROW("Unknown parameter type for joint limit");
416 break;
417 }
418 }
419
420 // return error
421 return error;
422}

424template <typename T>
425double LimitErrorFunctionT<T>::getJacobian(
426 const ModelParametersT<T>& params,
427 const SkeletonStateT<T>& state,
428 const MeshStateT<T>& /* meshState */,
429 Ref<Eigen::MatrixX<T>> jacobian,
430 Ref<Eigen::VectorX<T>> residual,
431 int& usedRows) {
432 MT_PROFILE_FUNCTION();
433
434 const auto& parameterTransform = this->parameterTransform_;
435
436 // loop over all joints and check for limit violations
437 double error = 0.0;
438
439 const T tWeight = kLimitWeight * this->weight_;
440
441 jacobian.setZero();
442 residual.setZero();
443
444 int count = 0;
445 // ---------------------------------------
446 // new joint error
447 // ---------------------------------------
448 for (const auto& limit : limits_) {
449 const T wgt = std::sqrt(kLimitWeight * this->weight_ * limit.weight);
450 switch (limit.type) {
451 case MinMax: {
452 const auto& data = limit.data.minMax;
453 MT_CHECK(data.parameterIndex < static_cast<size_t>(params.size()));
454 if (this->enabledParameters_.test(data.parameterIndex)) {
455 if (params(data.parameterIndex) < data.limits[0]) {
456 const T val = params(data.parameterIndex) - data.limits[0];
457 error += val * val * limit.weight * tWeight;
458 jacobian(count, data.parameterIndex) = wgt;
459 residual(count) = val * wgt;
460 } else if (params(data.parameterIndex) > data.limits[1]) {
461 const T val = params(data.parameterIndex) - data.limits[1];
462 error += val * val * limit.weight * tWeight;
463 jacobian(count, data.parameterIndex) = wgt;
464 residual(count) = val * wgt;
465 }
466 }
467 count++;
468 break;
469 }
470 case MinMaxJoint: {
471 // simple case, our jacobians are currently in joint space, just add them up
472 const auto& data = limit.data.minMaxJoint;
473 const size_t jointIndex = data.jointIndex * kParametersPerJoint + data.jointParameter;
474 MT_CHECK(jointIndex < (size_t)state.jointParameters.size());
475 if (this->activeJointParams_[jointIndex]) {
476 if (state.jointParameters(jointIndex) < data.limits[0]) {
477 const T val = state.jointParameters[jointIndex] - data.limits[0];
478 error += val * val * limit.weight * tWeight;
479 // explicitly multiply with the parameter transform to generate parameter space
480 // gradients
481 for (auto index = parameterTransform.transform.outerIndexPtr()[jointIndex];
482 index < parameterTransform.transform.outerIndexPtr()[jointIndex + 1];
483 ++index) {
484 jacobian(count, parameterTransform.transform.innerIndexPtr()[index]) +=
485 wgt * parameterTransform.transform.valuePtr()[index];
486 }
487 residual(count) = val * wgt;
488 } else if (state.jointParameters(jointIndex) > data.limits[1]) {
489 const T val = state.jointParameters[jointIndex] - data.limits[1];
490 error += val * val * limit.weight * tWeight;
491 // explicitly multiply with the parameter transform to generate parameter space
492 // gradients
493 for (auto index = parameterTransform.transform.outerIndexPtr()[jointIndex];
494 index < parameterTransform.transform.outerIndexPtr()[jointIndex + 1];
495 ++index) {
496 jacobian(count, parameterTransform.transform.innerIndexPtr()[index]) +=
497 wgt * parameterTransform.transform.valuePtr()[index];
498 }
499 residual(count) = val * wgt;
500 }
501 }
502 count++;
503 break;
504 }
505 case MinMaxJointPassive: {
506 break;
507 }
508 case Linear: {
509 const auto& data = limit.data.linear;
510 MT_CHECK(data.referenceIndex < static_cast<size_t>(params.size()));
511 MT_CHECK(data.targetIndex < static_cast<size_t>(params.size()));
512
513 if ((this->enabledParameters_.test(data.targetIndex) ||
514 this->enabledParameters_.test(data.referenceIndex)) &&
515 isInRange(data, params(data.targetIndex))) {
516 const T res =
517 params(data.targetIndex) * data.scale - data.offset - params(data.referenceIndex);
518 error += res * res * limit.weight * tWeight;
519 residual(count) = res * wgt;
520
521 if (this->enabledParameters_.test(data.targetIndex)) {
522 jacobian(count, data.targetIndex) = data.scale * wgt;
523 }
524
525 if (this->enabledParameters_.test(data.referenceIndex)) {
526 jacobian(count, data.referenceIndex) = -wgt;
527 }
528 }
529 count++;
530 break;
531 }
532 case LinearJoint: {
533 const auto& data = limit.data.linearJoint;
534 const size_t referenceParameterIndex =
535 data.referenceJointIndex * kParametersPerJoint + data.referenceJointParameter;
536 const size_t targetParameterIndex =
537 data.targetJointIndex * kParametersPerJoint + data.targetJointParameter;
538 MT_CHECK(referenceParameterIndex < (size_t)state.jointParameters.size());
539 MT_CHECK(targetParameterIndex < (size_t)state.jointParameters.size());
540 if ((this->activeJointParams_[referenceParameterIndex] ||
541 this->activeJointParams_[targetParameterIndex]) &&
542 isInRange(data, state.jointParameters(targetParameterIndex))) {
543 const T res = state.jointParameters(targetParameterIndex) * data.scale - data.offset -
544 state.jointParameters(referenceParameterIndex);
545 error += res * res * limit.weight * tWeight;
546
547 residual(count) = res * wgt;
548
549 // explicitly multiply with the parameter transform to generate parameter space
550 // gradients
551 if (this->activeJointParams_[targetParameterIndex]) {
552 for (auto index = parameterTransform.transform.outerIndexPtr()[targetParameterIndex];
553 index < parameterTransform.transform.outerIndexPtr()[targetParameterIndex + 1];
554 ++index) {
555 jacobian(count, parameterTransform.transform.innerIndexPtr()[index]) +=
556 data.scale * wgt * parameterTransform.transform.valuePtr()[index];
557 }
558 }
559
560 if (this->activeJointParams_[referenceParameterIndex]) {
561 for (auto index = parameterTransform.transform.outerIndexPtr()[referenceParameterIndex];
562 index < parameterTransform.transform.outerIndexPtr()[referenceParameterIndex + 1];
563 ++index) {
564 jacobian(count, parameterTransform.transform.innerIndexPtr()[index]) -=
565 wgt * parameterTransform.transform.valuePtr()[index];
566 }
567 }
568 }
569 count++;
570 break;
571 }
572 case HalfPlane: {
573 const auto& data = limit.data.halfPlane;
574 MT_CHECK(data.param1 < static_cast<size_t>(params.size()));
575 MT_CHECK(data.param2 < static_cast<size_t>(params.size()));
576
577 if ((this->enabledParameters_.test(data.param1) ||
578 this->enabledParameters_.test(data.param2))) {
579 const Eigen::Vector2<T> p(params(data.param1), params(data.param2));
580 const T res = p.dot(data.normal.template cast<T>()) - data.offset;
581 if (res < 0) {
582 error += res * res * limit.weight * tWeight;
583 residual(count) = res * wgt;
584
585 if (this->enabledParameters_.test(data.param1)) {
586 jacobian(count, data.param1) = data.normal[0] * wgt;
587 }
588
589 if (this->enabledParameters_.test(data.param2)) {
590 jacobian(count, data.param2) = data.normal[1] * wgt;
591 }
592 }
593 }
594 count++;
595 break;
596 }
597 case Ellipsoid: {
598 // NOTE: The jacobian for these is currently simplified
599 // It assumes the ellipsoid is static and doesn't move with the parent joint
600 const auto& ct = limit.data.ellipsoid;
601
602 // get the constraint position in global space
603 const Eigen::Vector3<T> position =
604 state.jointState[ct.parent].transform * ct.offset.template cast<T>();
605
606 // get the constraint position in local ellipsoid space
607 const Eigen::Vector3<T> localPosition =
608 state.jointState[ct.ellipsoidParent].transform.inverse() * position;
609
610 // calculate constraint position in ellipsoid space
611 const Eigen::Vector3<T> ellipsoidPosition =
612 ct.ellipsoidInv.template cast<T>() * localPosition;
613
614 // project onto closest surface point
615 const Eigen::Vector3<T> normalizedPosition = ellipsoidPosition.normalized();
616
617 // go back to ellipsoid frame
618 const Eigen::Vector3<T> projectedPosition =
619 ct.ellipsoid.template cast<T>() * normalizedPosition;
620
621 // calculate the difference between projected position and actual position
622 const Eigen::Vector3<T> diff =
623 position - state.jointState[ct.ellipsoidParent].transform * projectedPosition;
624
625 // calculate offset in jacobian
626 Eigen::Ref<Eigen::MatrixX<T>> jac = jacobian.block(count, 0, 3, params.size());
627 Eigen::Ref<Eigen::VectorX<T>> res = residual.middleRows(count, 3);
628 MT_CHECK(jac.cols() == static_cast<Eigen::Index>(parameterTransform.transform.cols()));
629 MT_CHECK(jac.rows() == 3);
630
631 // calculate the difference between target and position and error
632 const T jwgt = std::sqrt(kLimitWeight * kPositionWeight * this->weight_ * limit.weight);
633
634 // loop over all joints the constraint is attached to and calculate gradient
635 size_t jointIndex = ct.parent;
636 while (jointIndex != ct.ellipsoidParent && jointIndex != kInvalidIndex) {
637 // check for valid index
638 MT_CHECK(jointIndex < this->skeleton_.joints.size());
639
640 const auto& jointState = state.jointState[jointIndex];
641 const size_t paramIndex = jointIndex * kParametersPerJoint;
642 const Eigen::Vector3<T> posd = position - jointState.translation();
643
644 // calculate derivatives based on active joints
645 for (size_t d = 0; d < 3; d++) {
646 if (this->activeJointParams_[paramIndex + d]) {
647 const Eigen::Vector3<T> jc = jointState.getTranslationDerivative(d) * jwgt;
648 for (auto index = parameterTransform.transform.outerIndexPtr()[paramIndex + d];
649 index < parameterTransform.transform.outerIndexPtr()[paramIndex + d + 1];
650 ++index) {
651 jac.col(parameterTransform.transform.innerIndexPtr()[index]) +=
652 jc * parameterTransform.transform.valuePtr()[index];
653 }
654 }
655 if (this->activeJointParams_[paramIndex + 3 + d]) {
656 const Eigen::Vector3<T> jc = jointState.getRotationDerivative(d, posd) * jwgt;
657 for (auto index = parameterTransform.transform.outerIndexPtr()[paramIndex + d + 3];
658 index < parameterTransform.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
659 ++index) {
660 jac.col(parameterTransform.transform.innerIndexPtr()[index]) +=
661 jc * parameterTransform.transform.valuePtr()[index];
662 }
663 }
664 }
665 if (this->activeJointParams_[paramIndex + 6]) {
666 const Eigen::Vector3<T> jc = jointState.getScaleDerivative(posd) * jwgt;
667 for (auto index = parameterTransform.transform.outerIndexPtr()[paramIndex + 6];
668 index < parameterTransform.transform.outerIndexPtr()[paramIndex + 6 + 1];
669 ++index) {
670 jac.col(parameterTransform.transform.innerIndexPtr()[index]) +=
671 jc * parameterTransform.transform.valuePtr()[index];
672 }
673 }
674
675 // go to the next joint
676 jointIndex = this->skeleton_.joints[jointIndex].parent;
677 }
678
679 res = diff * jwgt;
680 error += jwgt * jwgt * diff.squaredNorm();
681 count += 3;
682
683 break;
684 }
685 default:
686 // should never get here
687 MT_THROW("Unknown parameter type for joint limit");
688 break;
689 }
690 }
691
692 usedRows = count;
693
694 // return error
695 return error;
696}
