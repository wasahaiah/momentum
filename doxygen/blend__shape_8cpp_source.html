24template <typename T>
25void BlendShape::computeShape(
26 const BlendWeightsT<T>& coefficients,
27 std::vector<Eigen::Vector3<T>>& output) const {
28 MT_CHECK(
29 gsl::narrow<Eigen::Index>(baseShape_.size()) * 3 == shapeVectors_.rows(),

41VectorXf BlendShape::estimateCoefficients(
42 std::span<const Vector3f> vertices,
43 const float regularization,
44 const VectorXf& weights) const {
45 MT_CHECK(
46 gsl::narrow<Eigen::Index>(baseShape_.size()) * 3 == shapeVectors_.rows(),
47 "{} is not {}",
48 baseShape_.size() * 3,
49 shapeVectors_.rows());
50 MT_CHECK(
51 vertices.size() == baseShape_.size(), "{} is not {}", vertices.size(), baseShape_.size());
52
53 const Map<const VectorXf> baseVec(&baseShape_[0][0], baseShape_.size() * 3);
54 const Map<const VectorXf> vertexVec(&vertices[0][0], vertices.size() * 3);
55
56 // if we're not using weights, use default pipeline
57 if (weights.size() != gsl::narrow<Eigen::Index>(vertices.size())) {
58 // check if factorization is valid
59 if (!factorizationValid_) {
60 factorization_.compute(shapeVectors_, Eigen::ComputeThinU | Eigen::ComputeThinV);
61 factorizationValid_ = true;
62 }
63
64 // estimate the shapeVectors for the given vertices
65 return factorization_.solve(vertexVec - baseVec);
66 } else {
67 // if we have weights, we definitely have to compute a new factorization that incorporates the
68 // weights
69 VectorXf wts(weights.size() * 3);
70 Eigen::Map<VectorXf, 0, Eigen::Stride<1, 3>>(&wts(0), weights.size()) = weights;
71 Eigen::Map<VectorXf, 0, Eigen::Stride<1, 3>>(&wts(1), weights.size()) = weights;
72 Eigen::Map<VectorXf, 0, Eigen::Stride<1, 3>>(&wts(2), weights.size()) = weights;
73
74 // calculate weighted coefficient matrix and rhs
75 const auto rows = shapeVectors_.rows();
76 const auto cols = shapeVectors_.cols();
77 MatrixXf A(rows + cols, cols);
78 for (int i = 0; i < wts.size(); i++) {
79 A.row(i) = shapeVectors_.row(i) * wts(i);
80 }
81 A.block(rows, 0, cols, cols) = MatrixXf::Identity(cols, cols) * regularization;
82
83 VectorXf b = VectorXf::Zero(rows + cols);
84 b.topRows(rows) = (vertexVec - baseVec).cwiseProduct(wts);
85
86 // solve normal equations to get results
87 return (A.transpose() * A).ldlt().solve(A.transpose() * b);
88 }
89}

109bool BlendShape::isApprox(const BlendShape& blendShape) const {
110 if (baseShape_.size() != blendShape.getBaseShape().size()) {
111 return false;
112 }
113 for (size_t i = 0; i < baseShape_.size(); i++) {
114 if (!baseShape_[i].isApprox(blendShape.getBaseShape()[i])) {
115 return false;
116 }
117 }
118
119 return shapeVectors_.isApprox(blendShape.getShapeVectors()) &&
120 (factorizationValid_ == blendShape.getFactorizationValid());
121}
