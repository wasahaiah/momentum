57template <typename T>
58TransformT<T> TransformT<T>::fromMatrix(const Matrix4<T>& other) {
59 TransformT<T> result;
60 result.translation = other.template topRightCorner<3, 1>();
61 // Calculate the scale by taking the norm of the first column, assuming uniform scaling
62 const auto& scaledR = other.template topLeftCorner<3, 3>();
63 result.scale = scaledR.col(0).norm();
64 MT_CHECK(result.scale >= Eps<T>(), "Scale is too small: {}", result.scale);
65 MT_CHECK(result.scale < T(1) / Eps<T>(), "Inverse scale is too small: {}", result.scale);
66 result.rotation = scaledR / result.scale;
67 return result;
68}

84template <typename T>
85TransformT<T> TransformT<T>::inverse() const {
86 // (translate(t) * rotation(R) * scale(s)).inv() =
87 // scale(s).inv() * rotation(R).inv() * translate(t).inv()
88 // = scale(1/s) * rotation(invR) * translate(-t)
89 // = translate(-invR*s*t) * rotation(invR) * scale(1/s)
90 const Quaternion<T> invRot = rotation.inverse();
91 const T invScale = T(1) / scale;
92 return TransformT<T>(-invScale * (invRot * translation), invRot, invScale);
93}

95template <typename T>
96TransformT<T> blendTransforms(
97 std::span<const TransformT<T>> transforms,
98 std::span<const T> weights) {
99 assert(transforms.size() == weights.size());
100 if (transforms.empty()) {
101 return TransformT<T>();
102 }
103
104 // Find average rotation by means described in
105 // https://stackoverflow.com/questions/12374087/average-of-multiple-quaternions
106 // http://www.acsu.buffalo.edu/~johnc/ave_quat07.pdf
107 //
108 // i.e. Stack quaternion coeffs in Q, compute M = Q^T x Q, and yield the eigenvector corresponding
109 // to the largest eigenvalue as the average rotation
110 Matrix4<T> QtQ = Matrix4<T>::Zero();
111
112 const auto n = transforms.size();
113 for (size_t i = 0; i < n; ++i) {
114 const auto& q = transforms[i].rotation;
115 QtQ += weights[i] * (q.coeffs() * q.coeffs().transpose());
116 }
117
118 const Eigen::SelfAdjointEigenSolver<Matrix4<T>> eigDecomp(QtQ);
119 assert(
120 eigDecomp.eigenvalues()[2] <=
121 eigDecomp.eigenvalues()[3]); // Eigen is supposed to sort smallest to largest.
122 Vector4<T> avgcoeffs = eigDecomp.eigenvectors().col(3);
123 assert(std::abs(avgcoeffs.norm() - 1) < 1e-4); // should be guaranteed by the eigensolver.
124 Quaternion<T> rot(avgcoeffs);
125
126 Vector3<T> pos = Vector3<T>::Zero();
127 T weightSum = 0;
128 T scale = 0;
129 for (size_t i = 0; i < n; ++i) {
130 pos += weights[i] * transforms[i].translation;
131 scale += weights[i] * transforms[i].scale;
132 weightSum += weights[i];
133 }
134
135 if (weightSum != 0) {
136 pos /= weightSum;
137 scale /= weightSum;
138 }
139
140 return TransformT<T>{pos, rot, scale};
141}

143template <typename T>
144TransformT<T> slerp(const TransformT<T>& t1, const TransformT<T>& t2, T weight) {
145 return TransformT<T>(
146 t1.translation + weight * (t2.translation - t1.translation),
147 t1.rotation.slerp(weight, t2.rotation),
148 t1.scale + weight * (t2.scale - t1.scale));
149}
