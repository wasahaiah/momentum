53template <typename T>
54Eigen::Vector3<T> SkinnedLocatorErrorFunctionT<T>::getLocatorRestPosition(
55 const ModelParametersT<T>& modelParams,
56 int locatorIndex) const {
57 MT_CHECK(locatorIndex >= 0 && locatorIndex < static_cast<int>(character_.skinnedLocators.size()));
58 const auto& locator = character_.skinnedLocators[locatorIndex];
59
60 Vector3<T> result = locator.position.template cast<T>();
61 int locatorParameterIndex = -1;
62 if (locatorIndex < character_.parameterTransform.skinnedLocatorParameters.size()) {
63 locatorParameterIndex = character_.parameterTransform.skinnedLocatorParameters[locatorIndex];
64 }
65
66 if (locatorParameterIndex >= 0) {
67 result += modelParams.v.template segment<3>(locatorParameterIndex).template cast<T>();
68 }
69
70 return result;
71}

73template <typename T>
74Eigen::Vector3<T> SkinnedLocatorErrorFunctionT<T>::calculateSkinnedLocatorPosition(
75 const SkeletonStateT<T>& state,
76 int locatorIndex,
77 const Eigen::Vector3<T>& locatorRestPos) const {
78 MT_CHECK(locatorIndex >= 0 && locatorIndex < static_cast<int>(character_.skinnedLocators.size()));
79 const auto& locator = character_.skinnedLocators[locatorIndex];
80
81 Eigen::Vector3<T> worldPos = Eigen::Vector3<T>::Zero();
82 T weightSum = 0;
83 for (int k = 0; k < locator.skinWeights.size(); ++k) {
84 const auto& weight = locator.skinWeights[k];
85 const auto boneIndex = locator.parents[k];
86 const auto& jointState = state.jointState[boneIndex];
87
88 worldPos += weight *
89 (jointState.transform *
90 (character_.inverseBindPose[boneIndex].template cast<T>() * locatorRestPos))
91 .template cast<T>();
92 weightSum += weight;
93 }
94
95 return worldPos / weightSum;
96}

98template <typename T>
99double SkinnedLocatorErrorFunctionT<T>::getError(
100 const ModelParametersT<T>& modelParameters,
101 const SkeletonStateT<T>& state,
102 const MeshStateT<T>& /* meshState */) {
103 MT_PROFILE_FUNCTION();
104
105 // loop over all constraints and calculate the error
106 double error = 0.0;
107
108 for (size_t i = 0; i < constraints_.size(); ++i) {
109 const SkinnedLocatorConstraintT<T>& constr = constraints_[i];
110
111 const Eigen::Vector3<T> locatorRestPos =
112 getLocatorRestPosition(modelParameters, constr.locatorIndex);
113 const Eigen::Vector3<T> locatorWorldPos =
114 calculateSkinnedLocatorPosition(state, constr.locatorIndex, locatorRestPos);
115 const Eigen::Vector3<T> diff = locatorWorldPos - constr.targetPosition;
116 error += constr.weight * diff.squaredNorm();
117 }
118
119 // return error
120 return error * this->weight_;
121}

123template <typename T>
124void SkinnedLocatorErrorFunctionT<T>::calculateDWorldPos(
125 const SkeletonStateT<T>& state,
126 const SkinnedLocatorConstraintT<T>& constr,
127 const Eigen::Vector3<T>& d_restPos,
128 Eigen::Vector3<T>& d_worldPos) const {
129 const auto& locator = character_.skinnedLocators[constr.locatorIndex];
130
131 for (int k = 0; k < locator.skinWeights.size(); ++k) {
132 const auto& weight = locator.skinWeights[k];
133 const auto boneIndex = locator.parents[k];
134 const auto& jointState = state.jointState[boneIndex];
135
136 if (weight > 0) {
137 // Use the full transformation matrix to be consistent with calculateSkinnedLocatorPosition
138 d_worldPos += weight *
139 ((jointState.transform * character_.inverseBindPose[boneIndex].template cast<T>())
140 .linear() *
141 d_restPos);
142 }
143 }
144}

146template <typename T>
147double SkinnedLocatorErrorFunctionT<T>::calculatePositionGradient(
148 const ModelParametersT<T>& modelParameters,
149 const SkeletonStateT<T>& state,
150 const SkinnedLocatorConstraintT<T>& constr,
151 Eigen::Ref<Eigen::VectorX<T>> gradient) const {
152 const Eigen::Vector3<T> locatorRestPos =
153 getLocatorRestPosition(modelParameters, constr.locatorIndex);
154 const Eigen::Vector3<T> locatorWorldPos =
155 calculateSkinnedLocatorPosition(state, constr.locatorIndex, locatorRestPos);
156 const Eigen::Vector3<T> diff = locatorWorldPos - constr.targetPosition;
157
158 // calculate the difference between target and position and error
159 const T wgt = constr.weight * 2.0f * this->weight_;
160
161 // Handle derivatives wrt skinnedLocatorParameters if this locator is parameterized
162 int locatorParameterIndex = getSkinnedLocatorParameterIndex(constr.locatorIndex);
163 if (locatorParameterIndex >= 0) {
164 // For each coordinate (x, y, z), calculate the derivative
165 for (int d = 0; d < 3; ++d) {
166 // Create a unit vector in the direction of the coordinate
167 Eigen::Vector3<T> d_restPos = Eigen::Vector3<T>::Zero();
168 d_restPos[d] = 1.0;
169
170 // Calculate the derivative of the world position with respect to this coordinate
171 Eigen::Vector3<T> d_worldPos = Eigen::Vector3<T>::Zero();
172 calculateDWorldPos(state, constr, d_restPos, d_worldPos);
173
174 // Add contribution to the gradient
175 gradient(locatorParameterIndex + d) += diff.dot(d_worldPos) * wgt;
176 }
177 }
178
179 SkinningWeightIteratorT<T> skinningIter(
180 this->character_, character_.skinnedLocators[constr.locatorIndex], locatorRestPos, state);
181
182 // Handle derivatives wrt jointParameters
183 while (!skinningIter.finished()) {
184 const auto [jointIndex, boneWeight, pos] = skinningIter.next();
185
186 // check for valid index
187 MT_CHECK(jointIndex < this->skeleton_.joints.size());
188
189 const auto& jointState = state.jointState[jointIndex];
190 const size_t paramIndex = jointIndex * kParametersPerJoint;
191 const Eigen::Vector3<T> posd = pos - jointState.translation();
192
193 // calculate derivatives based on active joints
194 for (size_t d = 0; d < 3; d++) {
195 if (this->activeJointParams_[paramIndex + d]) {
196 // Gradient wrt translation:
197 gradient_jointParams_to_modelParams(
198 boneWeight * diff.dot(jointState.getTranslationDerivative(d)) * wgt,
199 paramIndex + d,
200 this->parameterTransform_,
201 gradient);
202 }
203 if (this->activeJointParams_[paramIndex + 3 + d]) {
204 // Gradient wrt rotation:
205 gradient_jointParams_to_modelParams(
206 boneWeight * diff.dot(jointState.getRotationDerivative(d, posd)) * wgt,
207 paramIndex + 3 + d,
208 this->parameterTransform_,
209 gradient);
210 }
211 }
212 if (this->activeJointParams_[paramIndex + 6]) {
213 // Gradient wrt scale:
214 gradient_jointParams_to_modelParams(
215 boneWeight * diff.dot(jointState.getScaleDerivative(posd)) * wgt,
216 paramIndex + 6,
217 this->parameterTransform_,
218 gradient);
219 }
220 }
221
222 return constr.weight * diff.squaredNorm();
223}

225template <typename T>
226double SkinnedLocatorErrorFunctionT<T>::calculatePositionJacobian(
227 const ModelParametersT<T>& modelParameters,
228 const SkeletonStateT<T>& state,
229 const SkinnedLocatorConstraintT<T>& constr,
230 Ref<Eigen::MatrixX<T>> jac,
231 Ref<Eigen::VectorX<T>> res) const {
232 const Eigen::Vector3<T> locatorRestPos =
233 getLocatorRestPosition(modelParameters, constr.locatorIndex);
234 const Eigen::Vector3<T> locatorWorldPos =
235 calculateSkinnedLocatorPosition(state, constr.locatorIndex, locatorRestPos);
236 const Eigen::Vector3<T> diff = locatorWorldPos - constr.targetPosition;
237
238 const T wgt = std::sqrt(constr.weight * this->weight_);
239
240 // Handle derivatives wrt skinnedLocatorParameters if this locator is parameterized
241 int locatorParameterIndex = getSkinnedLocatorParameterIndex(constr.locatorIndex);
242 if (locatorParameterIndex >= 0) {
243 // For each coordinate (x, y, z), calculate the derivative
244 for (int d = 0; d < 3; ++d) {
245 // Create a unit vector in the direction of the coordinate
246 Eigen::Vector3<T> d_restPos = Eigen::Vector3<T>::Zero();
247 d_restPos[d] = 1.0;
248
249 // Calculate the derivative of the world position with respect to this coordinate
250 Eigen::Vector3<T> d_worldPos = Eigen::Vector3<T>::Zero();
251 calculateDWorldPos(state, constr, d_restPos, d_worldPos);
252
253 // Add contribution to the jacobian
254 jac.col(locatorParameterIndex + d).template segment<3>(0) += d_worldPos * wgt;
255 }
256 }
257
258 SkinningWeightIteratorT<T> skinningIter(
259 this->character_, character_.skinnedLocators[constr.locatorIndex], locatorRestPos, state);
260
261 // Handle derivatives wrt jointParameters
262 while (!skinningIter.finished()) {
263 const auto [jointIndex, boneWeight, pos] = skinningIter.next();
264
265 // check for valid index
266 MT_CHECK(jointIndex < this->skeleton_.joints.size());
267
268 const auto& jointState = state.jointState[jointIndex];
269 const size_t paramIndex = jointIndex * kParametersPerJoint;
270 const Eigen::Vector3<T> posd = pos - jointState.translation();
271
272 // calculate derivatives based on active joints
273 for (size_t d = 0; d < 3; d++) {
274 if (this->activeJointParams_[paramIndex + d]) {
275 jacobian_jointParams_to_modelParams<T>(
276 boneWeight * wgt * jointState.getTranslationDerivative(d),
277 paramIndex + d,
278 this->parameterTransform_,
279 jac);
280 }
281 if (this->activeJointParams_[paramIndex + 3 + d]) {
282 jacobian_jointParams_to_modelParams<T>(
283 boneWeight * wgt * jointState.getRotationDerivative(d, posd),
284 paramIndex + d + 3,
285 this->parameterTransform_,
286 jac);
287 }
288 }
289 if (this->activeJointParams_[paramIndex + 6]) {
290 jacobian_jointParams_to_modelParams<T>(
291 boneWeight * wgt * jointState.getScaleDerivative(posd),
292 paramIndex + 6,
293 this->parameterTransform_,
294 jac);
295 }
296 }
297
298 res = diff * wgt;
299
300 return wgt * wgt * diff.squaredNorm();
301}

303template <typename T>
304double SkinnedLocatorErrorFunctionT<T>::getGradient(
305 const ModelParametersT<T>& modelParameters,
306 const SkeletonStateT<T>& state,
307 const MeshStateT<T>& /* meshState */,
308 Eigen::Ref<Eigen::VectorX<T>> gradient) {
309 double error = 0;
310
311 // Process each constraint
312 for (const auto& constraint : constraints_) {
313 error += calculatePositionGradient(modelParameters, state, constraint, gradient);
314 }
315
316 return this->weight_ * error;
317}

319template <typename T>
320double SkinnedLocatorErrorFunctionT<T>::getJacobian(
321 const ModelParametersT<T>& modelParameters,
322 const SkeletonStateT<T>& state,
323 const MeshStateT<T>& /* meshState */,
324 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
325 Eigen::Ref<Eigen::VectorX<T>> residual,
326 int& usedRows) {
327 MT_CHECK(
328 jacobian.cols() == static_cast<Eigen::Index>(this->parameterTransform_.transform.cols()));
329 MT_CHECK(jacobian.rows() >= (Eigen::Index)(3 * constraints_.size()));
330 MT_CHECK(residual.rows() >= (Eigen::Index)(3 * constraints_.size()));
331
332 double error = 0;
333
334 for (size_t iCons = 0; iCons < constraints_.size(); ++iCons) {
335 const auto& constraint = constraints_[iCons];
336 error += calculatePositionJacobian(
337 modelParameters,
338 state,
339 constraint,
340 jacobian.block(3 * iCons, 0, 3, modelParameters.size()),
341 residual.middleRows(3 * iCons, 3));
342 usedRows += 3;
343 }
344
345 return error;
346}
