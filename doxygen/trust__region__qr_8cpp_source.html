50template <typename T>
51void TrustRegionQRT<T>::doIteration() {
52 MT_PROFILE_FUNCTION();
53
54 const auto sf = static_cast<SkeletonSolverFunctionT<T>*>(this->solverFunction_);
55
56 const auto skeleton = sf->getSkeleton();
57 const auto parameterTransform = sf->getParameterTransform();
58
59 {
60 MT_PROFILE_EVENT("JtJR - update state");
61 skeletonState_.set(parameterTransform->apply(this->parameters_), *skeleton);
62 }
63
64 if (sf->needsMeshState()) {
65 MT_PROFILE_EVENT("JtJR - update mesh state");
66 meshState_.update(this->parameters_, skeletonState_, sf->getCharacter());
67 }
68
69 const auto nFullParams = gsl::narrow_cast<Eigen::Index>(this->parameters_.size());
70
71 std::vector<Eigen::Index> enabledParameters;
72 enabledParameters.reserve(nFullParams);
73 for (Eigen::Index iFullParam = 0; iFullParam < nFullParams; ++iFullParam) {
74 if (this->activeParameters_.test(iFullParam)) {
75 enabledParameters.push_back(iFullParam);
76 }
77 }
78
79 const Eigen::Index nSubsetParams = gsl::narrow_cast<Eigen::Index>(enabledParameters.size());
80
81 // Add a tiny lambda just to make sure we don't divide by zero when back-substituting into R,
82 // this isn't intended to regularize the problem at all (searching for the actual lambda parameter
83 // that respects the trust region radius happens below).
84 T lambda = 1e-10;
85 this->qrSolver_.reset(nSubsetParams, lambda);
86
87 // Compute the QR factorization of the Jacobian:
88 double error_orig = 0.0;
89 for (auto errorFunction : sf->getErrorFunctions()) {
90 if (errorFunction->getWeight() <= 0) {
91 continue;
92 }
93
94 auto rows = errorFunction->getJacobianSize();
95
96 jacobian_.resizeAndSetZero(rows, nFullParams);
97 residual_.resizeAndSetZero(rows);
98
99 int usedRows = 0;
100 error_orig += errorFunction->getJacobian(
101 this->parameters_, skeletonState_, meshState_, jacobian_.mat(), residual_.mat(), usedRows);
102 if (usedRows == 0) {
103 continue;
104 }
105
106 this->qrSolver_.addMutating(
107 ColumnIndexedMatrix<Eigen::MatrixX<T>>(
108 jacobian_.mat().topRows(usedRows), enabledParameters),
109 residual_.mat().topRows(usedRows));
110 }
111
112 this->error_ = error_orig;
113
114 // Because the error function is computed r^t*r, the gradient is
115 // 2*J^T*r.
116 this->gradientSub_ = T(2) * qrSolver_.At_times_b();
117
118 // Save the R matrix before we start adding lambdas:
119 this->Rmatrix_ = this->qrSolver_.R();
120
121 // The Hessian is approximated by 2*J^T*J.
122 // In the QR solver, we compute J = Q*R.
123 // 2*J^T*J = (Q*R)^T * (Q*R)
124 // = R^T * Q^T * Q * R
125 // = 2 * R^T * R
126 // To compute v^T * H * v, we can then compute
127 // 2 * ||R * v||^2
128
129 // This is the local approximation to the function used in the trust region solve:
130 auto evalQuadraticModel = [&](Eigen::Ref<const Eigen::VectorX<T>> p) -> T {
131 // MT_LOGI("In evalQuadraticModel(); orig error: {}", this->error_);
132 T result = this->error_;
133 result -= this->gradientSub_.dot(p);
134 // Normally in textbooks you'd see this multiplied by 1/2 for the proper Taylor series
135 // expansion but remember that our Hessian is actually 2*J^T*J:
136 result += (this->Rmatrix_.template triangularView<Eigen::Upper>() * p).squaredNorm();
137 return result;
138 };
139
140 auto subsetToFullVector = [&](Eigen::Ref<const Eigen::VectorX<T>> v_subset) -> Eigen::VectorX<T> {
141 Eigen::VectorX<T> result = Eigen::VectorX<T>::Zero(nFullParams);
142 for (Eigen::Index iSubsetParam = 0; iSubsetParam < nSubsetParams; ++iSubsetParam) {
143 const Eigen::Index iFullParam = enabledParameters[iSubsetParam];
144 result[iFullParam] = v_subset[iSubsetParam];
145 }
146 return result;
147 };
148
149 // TODO what is a good value for nu?
150 const T nu = 0;
151
152 for (size_t iTrustStep = 0; iTrustStep < 10; ++iTrustStep) {
153 Eigen::VectorX<T> searchDir_sub = this->qrSolver_.result();
154
155 // If even our most optimistic guess of how much we can decrease the error
156 // suggests that it's not going to change by a measurable amount (FLT_EPS times its current
157 // value), then stop trying to iterate. If we don't do that we run into issues below where the
158 // rho value is just capturing noise and our trust region will drop to 0.
159 if (searchDir_sub.dot(gradientSub_) < FLT_EPSILON * (T(1.0) + this->error_)) {
160 break;
161 }
162
163 // We use a standard Newton approach for quickly converging to the lambda that limits
164 // the step size to the trust region. This is described in section 4.3 of Nocedal/Wright,
165 // "Iterative solution of the subproblem."
166 //
167 // We will only allow the value of lambda to _increase_, which corresponds to shrinking the
168 // trust region. That is, we start with a trust region that is too large, and then we
169 // iteratively shrink it until it's close enough to the desired delta. This is because (1) in
170 // practice, this is what seems to happen anyways, given that we start from 0 and (2) it's a lot
171 // easier to add to lambda than subtract from lambda; the former can always be done by just
172 // adding some additional rows to the Jacobian while the latter requires "backing out" to a
173 // previous version of the QR solver. This is okay because it's always more acceptable (within
174 // reason) to take a too-small step than a too-large one:
175 for (size_t iIter = 0; iIter < 3; ++iIter) {
176 if (searchDir_sub.norm() < T(1.05) * this->curTrustRegionRadius_) {
177 break;
178 }
179
180 const Eigen::MatrixX<T>& Rmat = this->qrSolver_.R();
181 // This is equation 4.44 in Nocedal/Wright. Note that our error function is actually
182 // r^T*r which means gradientSub_ is 2*J^T*r, but the Hessian approximated inside the QR
183 // solver is J^T*J. Since we're applying the lambda within the QR solver also, we'll stay
184 // consistent with the J^T*J/J^T*r version of the world and multiply our gradientSub_ by 0.5
185 // on its way in.
186 Eigen::VectorX<T> p_l = Rmat.template triangularView<Eigen::Upper>().solve(
187 Rmat.template triangularView<Eigen::Upper>().transpose().solve(-T(0.5) * gradientSub_));
188 Eigen::VectorX<T> q_l = Rmat.template triangularView<Eigen::Upper>().transpose().solve(p_l);
189
190 const T p_l_norm2 = p_l.squaredNorm();
191 const T q_l_norm2 = q_l.squaredNorm();
192
193 if (q_l_norm2 < FLT_EPSILON) {
194 break;
195 }
196
197 const T p_l_norm = std::sqrt(p_l_norm2);
198 const T deltaLambda = (p_l_norm2 / q_l_norm2) *
199 ((p_l_norm - this->curTrustRegionRadius_) / (this->curTrustRegionRadius_));
200
201 // We can only easily _increase_ the lambda value, so stop if the change is negative:
202 if (deltaLambda <= 0) {
203 break;
204 }
205
206 {
207 // We want to find a y such that:
208 // lambda + y^2 = (lambda + deltaLambda)
209 // y = sqrt((lambda + deltaLambda) - lambda)
210 const T lambdaNew = lambda + deltaLambda;
211 const T y = std::sqrt(lambdaNew - lambda);
212
213 // Add the lambda terms:
214 // TODO make this faster than O(n^2):
215 this->lambdaDiag_.resize(nSubsetParams, nSubsetParams);
216 this->lambdaDiag_.setZero();
217 this->lambdaDiag_.diagonal().setConstant(y);
218
219 this->lambdaZero_.resize(nSubsetParams);
220 this->lambdaZero_.setZero();
221
222 this->qrSolver_.addMutating(this->lambdaDiag_, this->lambdaZero_);
223
224 lambda = lambdaNew;
225 }
226
227 searchDir_sub = this->qrSolver_.result();
228
229 MT_LOGI_IF(
230 this->verbose_,
231 " Iter: {}; Search dir cur norm: {}; delta: {}; lambda: {}",
232 iIter,
233 searchDir_sub.norm(),
234 this->curTrustRegionRadius_,
235 lambda);
236 }
237
238 const Eigen::VectorX<T> parameters_orig = this->parameters_;
239 this->solverFunction_->updateParameters(this->parameters_, subsetToFullVector(searchDir_sub));
240 const double error_new = this->solverFunction_->getError(this->parameters_);
241
242 // This is the decrease of the function relative to what we _expected_ the function to
243 // decrease by. It is eq. 4.4 in Nocedal and Wright.
244 const T quadraticModelEval = evalQuadraticModel(searchDir_sub);
245 const T rho = (this->error_ - error_new) / (this->error_ - quadraticModelEval);
246 MT_LOGI_IF(
247 this->verbose_,
248 "Error orig: {}; error new: {}; quadratic model: {}; rho: {}",
249 this->error_,
250 error_new,
251 quadraticModelEval,
252 rho);
253
254 if (rho < T(0.25)) {
255 // Model is no good, decrease the radius:
256 this->curTrustRegionRadius_ = T(0.25) * this->curTrustRegionRadius_;
257 } else if (rho > T(0.75) && lambda > 0) {
258 // Model seems good and we're pushing up against the trust region boundary, so increase it:
259 this->curTrustRegionRadius_ =
260 std::min(T(2) * this->curTrustRegionRadius_, this->maxTrustRegionRadius_);
261 }
262
263 if (rho > nu) {
264 break;
265 } else {
266 // Reject the step, reduce the trust region size and try again:
267 this->parameters_ = parameters_orig;
268 }
269 }
270}
