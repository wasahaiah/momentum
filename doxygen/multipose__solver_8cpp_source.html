51template <typename T>
52void MultiposeSolverT<T>::doIteration() {
53 MT_PROFILE_FUNCTION();
54
55 MultiposeSolverFunctionT<T>* fn =
56 dynamic_cast<MultiposeSolverFunctionT<T>*>(this->solverFunction_);
57
58 const std::vector<Eigen::Index> genericParameters(
59 fn->genericParameters_.begin(), fn->genericParameters_.end());
60 const std::vector<Eigen::Index> universalParameters(
61 fn->universalParameters_.begin(), fn->universalParameters_.end());
62
63 fn->setFrameParametersFromJoinedParameterVector(this->parameters_);
64
65 OnlineBlockHouseholderQR<T> qrSolver(fn->universalParameters_.size(), std::sqrt(regularization_));
66 const auto nFullParameters = fn->parameterTransform_->numAllModelParameters();
67
68 this->error_ = 0;
69 for (size_t f = 0; f < fn->getNumFrames(); ++f) {
70 const auto& frameParameters = fn->frameParameters_[f];
71 auto& skelState = fn->states_[f];
72
73 skelState.set(fn->parameterTransform_->apply(frameParameters), *fn->skeleton_);
74
75 // TODO: Update mesh state for functions that need it once we have character access
76 // meshState_->update(frameParameters, skelState, character);
77
78 for (auto&& solvable : fn->errorFunctions_[f]) {
79 if (solvable->getWeight() <= 0.0f) {
80 continue;
81 }
82
83 const size_t n = solvable->getJacobianSize();
84 const size_t jacobianSize = n + 8 - (n % 8);
85
86 jacobianBlock_.resize(jacobianSize, nFullParameters);
87 jacobianBlock_.setZero();
88
89 residualBlock_.resize(jacobianSize);
90 residualBlock_.setZero();
91 int rows = 0;
92 this->error_ += solvable->getJacobian(
93 frameParameters, skelState, *meshState_, jacobianBlock_, residualBlock_, rows);
94
95 qrSolver.addMutating(
96 f,
97 ColumnIndexedMatrix<Eigen::MatrixX<T>>(jacobianBlock_, genericParameters),
98 ColumnIndexedMatrix<Eigen::MatrixX<T>>(jacobianBlock_, universalParameters),
99 residualBlock_);
100 }
101 }
102
103 // Now, extract the delta:
104 // TODO line search?
105 const Eigen::VectorX<T> delta = qrSolver.x_dense();
106 this->solverFunction_->updateParameters(this->parameters_, delta);
107}
