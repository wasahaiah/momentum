29std::vector<std::vector<PositionData>> createConstraintData(
30 const std::span<const std::vector<Marker>> markerData,
31 const LocatorList& locators) {
32 std::vector<std::vector<PositionData>> results(markerData.size());
33
34 // map locator name to its index
35 std::map<std::string, size_t> locatorLookup;
36 for (size_t i = 0; i < locators.size(); i++) {
37 locatorLookup[locators[i].name] = i;
38 }
39
40 // create a list of position constraints per frame
41 for (size_t iFrame = 0; iFrame < markerData.size(); ++iFrame) {
42 const auto& markerList = markerData[iFrame];
43 for (const auto& jMarker : markerList) {
44 if (jMarker.occluded) {
45 continue;
46 }
47 auto query = locatorLookup.find(jMarker.name);
48 if (query == locatorLookup.end()) {
49 continue;
50 }
51 size_t locatorIdx = query->second;
52
53 results.at(iFrame).emplace_back(PositionData(
54 locators.at(locatorIdx).offset,
55 jMarker.pos.cast<float>(),
56 locators.at(locatorIdx).parent,
57 locators.at(locatorIdx).weight));
58 }
59 }
60
61 return results;
62}

99std::pair<Eigen::Vector<uint32_t, kMaxSkinJoints>, Eigen::Vector<float, kMaxSkinJoints>>
100averageTriangleSkinWeights(
101 const Character& character,
102 int triangleIndex,
103 const Eigen::Vector3f& barycentric) {
104 Eigen::VectorXf skinWeights = Eigen::VectorXf::Zero(character.skeleton.joints.size());
105
106 const auto& mesh = *character.mesh;
107 const auto& skin = *character.skinWeights;
108
109 // get the triangle
110 const auto& triangle = mesh.faces[triangleIndex];
111 for (int kTriVert = 0; kTriVert < 3; ++kTriVert) {
112 for (int kSkinWeight = 0; kSkinWeight < skin.index.cols(); ++kSkinWeight) {
113 const auto& skinIndex = skin.index(triangle(kTriVert), kSkinWeight);
114 const auto& skinWeight = skin.weight(triangle(kTriVert), kSkinWeight);
115 skinWeights[skinIndex] += skinWeight * barycentric[kTriVert];
116 }
117 }
118
119 std::vector<std::pair<float, uint32_t>> sortedWeights;
120 for (Eigen::Index i = 0; i < skinWeights.size(); ++i) {
121 if (skinWeights[i] > 0.0f) {
122 sortedWeights.emplace_back(skinWeights[i], static_cast<uint32_t>(i));
123 }
124 }
125 std::sort(sortedWeights.begin(), sortedWeights.end(), std::greater<>());
126
127 Eigen::Vector<float, kMaxSkinJoints> resultWeights = Eigen::Vector<float, kMaxSkinJoints>::Zero();
128 Eigen::Vector<uint32_t, kMaxSkinJoints> resultIndices =
129 Eigen::Vector<uint32_t, kMaxSkinJoints>::Zero();
130 for (size_t i = 0; i < kMaxSkinJoints; ++i) {
131 if (i < sortedWeights.size()) {
132 resultWeights[i] = sortedWeights[i].first;
133 resultIndices[i] = sortedWeights[i].second;
134 }
135 }
136
137 resultWeights /= resultWeights.sum();
138
139 return {resultIndices, resultWeights};
140}

153ClosestPointOnMeshResult closestPointOnMeshMatchingParent(
154 const momentum::Mesh& mesh,
155 const momentum::SkinWeights& skin,
156 const Eigen::Vector3f& p_world,
157 uint32_t parentIdx,
158 float cutoffWeight = 0.1) {
159 ClosestPointOnMeshResult result;
160 for (size_t iTri = 0; iTri < mesh.faces.size(); ++iTri) {
161 const auto& f = mesh.faces[iTri];
162
163 float skinWeight = 0;
164 for (int kTriVert = 0; kTriVert < 3; ++kTriVert) {
165 for (int kSkinWeight = 0; kSkinWeight < skin.index.cols(); ++kSkinWeight) {
166 if (skin.index(f(kTriVert), kSkinWeight) == parentIdx) {
167 skinWeight += skin.weight(f(kTriVert), kSkinWeight);
168 }
169 }
170 }
171 skinWeight /= 3.0f;
172
173 if (skinWeight < cutoffWeight) {
174 continue;
175 }
176
177 // Note: the return value from the does _not_ have anything to do with
178 // whether the projection is valid, it just indicates whether the projection is
179 // inside the triangle or not, so we can ignore it and just use the projected point directly.
180 Eigen::Vector3f p_tri;
181 Eigen::Vector3f bary;
182 axel::projectOnTriangle(
183 p_world, mesh.vertices[f(0)], mesh.vertices[f(1)], mesh.vertices[f(2)], p_tri, &bary);
184 const float dist = (p_tri - p_world).norm();
185 if (!result.valid || dist < result.distance) {
186 result = {true, iTri, bary, dist};
187 }
188 }
189
190 return result;
191}

193momentum::Character locatorsToSkinnedLocators(
194 const momentum::Character& sourceCharacter,
195 float maxDistance) {
196 if (!sourceCharacter.mesh || !sourceCharacter.skinWeights) {
197 return sourceCharacter;
198 }
199
200 const JointParameters restJointParams =
201 JointParameters::Zero(sourceCharacter.parameterTransform.numJointParameters());
202 const SkeletonState restState(restJointParams, sourceCharacter.skeleton);
203
204 SkinnedLocatorList skinnedLocators;
205 LocatorList locators;
206 for (size_t i = 0; i < sourceCharacter.locators.size(); ++i) {
207 const auto& locator = sourceCharacter.locators[i];
208 const auto& offset = locator.offset;
209 const auto& parent = locator.parent;
210 const Eigen::Vector3f p_world = restState.jointState[parent].transform * offset;
211
212 // Find the closest point on the mesh to the locator.
213 const auto closestPointResult = closestPointOnMeshMatchingParent(
214 *sourceCharacter.mesh,
215 *sourceCharacter.skinWeights,
216 p_world,
217 gsl::narrow_cast<uint32_t>(parent));
218 if (!closestPointResult.valid || closestPointResult.distance > maxDistance) {
219 locators.push_back(locator);
220 continue;
221 }
222
223 SkinnedLocator skinnedLocator;
224 skinnedLocator.name = locator.name;
225 skinnedLocator.position = p_world;
226 skinnedLocator.weight = locator.weight;
227
228 // Set the locator's parent to the closest face.
229 std::tie(skinnedLocator.parents, skinnedLocator.skinWeights) = averageTriangleSkinWeights(
230 sourceCharacter, closestPointResult.triangleIdx, closestPointResult.baryCoords);
231 skinnedLocators.push_back(skinnedLocator);
232 }
233
234 // Strip out regular locators and replace with skinned locators:
235 return {
236 sourceCharacter.skeleton,
237 sourceCharacter.parameterTransform,
238 sourceCharacter.parameterLimits,
239 locators,
240 sourceCharacter.mesh.get(),
241 sourceCharacter.skinWeights.get(),
242 sourceCharacter.collision.get(),
243 sourceCharacter.poseShapes.get(),
244 sourceCharacter.blendShape,
245 sourceCharacter.faceExpressionBlendShape,
246 sourceCharacter.name,
247 sourceCharacter.inverseBindPose,
248 skinnedLocators};
249}

251momentum::Character skinnedLocatorsToLocators(const momentum::Character& sourceCharacter) {
252 if (sourceCharacter.skinnedLocators.empty()) {
253 MT_LOGI("No skinned locators to convert");
254 return sourceCharacter;
255 }
256
257 const JointParameters restJointParams =
258 JointParameters::Zero(sourceCharacter.parameterTransform.numJointParameters());
259 const SkeletonState restState(restJointParams, sourceCharacter.skeleton);
260
261 LocatorList locators = sourceCharacter.locators;
262
263 for (const auto& skinnedLocator : sourceCharacter.skinnedLocators) {
264 // Find the bone with the highest skin weight
265 int maxWeightIdx = 0;
266 float maxWeight = skinnedLocator.skinWeights[0];
267
268 for (int i = 1; i < kMaxSkinJoints; ++i) {
269 if (skinnedLocator.skinWeights[i] > maxWeight) {
270 maxWeight = skinnedLocator.skinWeights[i];
271 maxWeightIdx = i;
272 }
273 }
274
275 // Get the parent bone index
276 const uint32_t parentBoneIdx = skinnedLocator.parents[maxWeightIdx];
277
278 // Transform the position from rest pose space to local bone space
279 const Eigen::Vector3f offset =
280 restState.jointState[parentBoneIdx].transform.inverse() * skinnedLocator.position;
281
282 // Create a regular locator
283 Locator locator;
284 locator.name = skinnedLocator.name;
285 locator.parent = parentBoneIdx;
286 locator.offset = offset;
287 locator.weight = skinnedLocator.weight;
288 locator.locked = Vector3i::Zero();
289 locator.limitOrigin = offset;
290 locator.limitWeight = Vector3f::Zero();
291
292 MT_LOGT(
293 "Converting skinned locator {} to regular locator attached to bone {} ({})",
294 skinnedLocator.name,
295 sourceCharacter.skeleton.joints.at(parentBoneIdx).name,
296 parentBoneIdx);
297
298 locators.push_back(locator);
299 }
300
301 // Return character with skinned locators converted to regular locators
302 return {
303 sourceCharacter.skeleton,
304 sourceCharacter.parameterTransform,
305 sourceCharacter.parameterLimits,
306 locators,
307 sourceCharacter.mesh.get(),
308 sourceCharacter.skinWeights.get(),
309 sourceCharacter.collision.get(),
310 sourceCharacter.poseShapes.get(),
311 sourceCharacter.blendShape,
312 sourceCharacter.faceExpressionBlendShape,
313 sourceCharacter.name,
314 sourceCharacter.inverseBindPose,
315 {}};
316}

318std::vector<momentum::SkinnedLocatorTriangleConstraintT<float>> createSkinnedLocatorMeshConstraints(
319 const momentum::Character& character,
320 float targetDepth) {
321 if (!character.mesh || !character.skinWeights) {
322 return {};
323 }
324
325 // skip the triangle tree if we don't need it:
326 if (character.skinnedLocators.empty()) {
327 return {};
328 }
329
330 const auto& mesh = *character.mesh;
331
332 std::vector<momentum::SkinnedLocatorTriangleConstraintT<float>> result;
333 result.reserve(character.skinnedLocators.size());
334 for (size_t i = 0; i < character.skinnedLocators.size(); ++i) {
335 const auto& locator = character.skinnedLocators[i];
336 const Eigen::Vector3f p_world = locator.position;
337
338 // Find the closest point on the mesh to the locator.
339 const auto closestPointResult = closestPointOnMeshMatchingParent(
340 *character.mesh, *character.skinWeights, p_world, locator.parents[0], targetDepth);
341 if (!closestPointResult.valid) {
342 continue;
343 }
344
345 momentum::SkinnedLocatorTriangleConstraintT<float> constr;
346 constr.locatorIndex = i;
347 constr.tgtTriangleIndices = mesh.faces[closestPointResult.triangleIdx];
348 constr.tgtTriangleBaryCoords = closestPointResult.baryCoords;
349 constr.depth = targetDepth;
350 result.push_back(constr);
351 }
352
353 return result;
354}

356Character createLocatorCharacter(const Character& sourceCharacter, const std::string& prefix) {
357 // create a copy
358 Character character = sourceCharacter;
359
360 auto& newSkel = character.skeleton;
361 auto& newTransform = character.parameterTransform;
362 auto& newLocators = character.locators;
363 auto& newLimits = character.parameterLimits;
364 auto& newInvBindPose = character.inverseBindPose;
365
366 ParameterSet locatorSet;
367
368 // create a new additional joint for each locator
369 std::vector<Eigen::Triplet<float>> triplets;
370 for (auto& newLocator : newLocators) {
371 // ignore locator if it is fixed
372 if (newLocator.locked.all()) {
373 continue;
374 }
375
376 // create a new joint for the locator
377 Joint joint;
378 joint.name = std::string(prefix + newLocator.name);
379 joint.parent = newLocator.parent;
380 joint.translationOffset = newLocator.offset;
381
382 // insert joint
383 const size_t id = newSkel.joints.size();
384 newSkel.joints.push_back(joint);
385 newInvBindPose.push_back(Affine3f::Identity());
386
387 // create parameter for the added joint
388 static const std::array<std::string, 3> tNames{"_tx", "_ty", "_tz"};
389 for (size_t j = 0; j < 3; j++) {
390 if (newLocator.locked[j] != 0) {
391 continue;
392 }
393 const std::string jname = joint.name + tNames[j];
394 triplets.emplace_back(
395 static_cast<int>(id) * kParametersPerJoint + static_cast<int>(j),
396 static_cast<int>(newTransform.name.size()),
397 1.0f);
398 MT_CHECK(
399 newTransform.name.size() < kMaxModelParams,
400 "Number of model parameters reached the {} limit.",
401 kMaxModelParams);
402 locatorSet.set(newTransform.name.size());
403 newTransform.name.push_back(jname);
404
405 // check if we need joint limit
406 if (newLocator.limitWeight[j] > 0.0f) {
407 ParameterLimit p;
408 p.type = MinMaxJoint;
409 p.data.minMaxJoint.jointIndex = id;
410 p.data.minMaxJoint.jointParameter = j;
411 const float referencePosition = newLocator.limitOrigin[j] - newLocator.offset[j];
412 p.data.minMaxJoint.limits = Vector2f(referencePosition, referencePosition);
413 p.weight = newLocator.limitWeight[j];
414 newLimits.push_back(p);
415 }
416 }
417 newTransform.offsets.conservativeResize(newTransform.offsets.size() + kParametersPerJoint);
418 newTransform.activeJointParams.conservativeResize(
419 newTransform.activeJointParams.size() + kParametersPerJoint);
420 for (size_t j = newTransform.offsets.size() - gsl::narrow<int>(kParametersPerJoint);
421 j < newTransform.offsets.size();
422 ++j) {
423 newTransform.offsets(j) = 0.0;
424 }
425
426 for (size_t j = newTransform.activeJointParams.size() - gsl::narrow<int>(kParametersPerJoint);
427 j < newTransform.activeJointParams.size();
428 ++j) {
429 newTransform.activeJointParams(j) = true;
430 }
431
432 // reattach locator to new joint
433 newLocator.parent = id;
434 newLocator.offset.setZero();
435 }
436 newTransform.parameterSets["locators"] = locatorSet;
437
438 // update parameter transform
439 const int newRows = static_cast<int>(newSkel.joints.size()) * kParametersPerJoint;
440 const int newCols = static_cast<int>(newTransform.name.size());
441 newTransform.transform.conservativeResize(newRows, newCols);
442 SparseRowMatrixf additionalTransforms(newRows, newCols);
443 additionalTransforms.setFromTriplets(triplets.begin(), triplets.end());
444 newTransform.transform += additionalTransforms;
445
446 // return the new character
447 return character;
448}

450LocatorList extractLocatorsFromCharacter(
451 const Character& locatorCharacter,
452 const CharacterParameters& calibParams) {
453 const SkeletonState state(
454 locatorCharacter.parameterTransform.apply(calibParams), locatorCharacter.skeleton);
455 const LocatorList& locators = locatorCharacter.locators;
456 const auto& skeleton = locatorCharacter.skeleton;
457
458 LocatorList result = locators;
459
460 // Check if locators is empty
461 if (locators.empty()) {
462 return result;
463 }
464
465 // revert each locator to the original attachment and add an offset
466 for (size_t i = 0; i < locators.size(); i++) {
467 // only map locators back that are not fixed
468 if (locators[i].locked.all()) {
469 continue;
470 }
471
472 // joint id
473 const size_t jointId = locators[i].parent;
474
475 // Check bounds for jointState access
476 if (jointId >= state.jointState.size()) {
477 MT_LOGW(
478 "Joint ID {} is out of bounds for jointState (size: {})",
479 jointId,
480 state.jointState.size());
481 continue;
482 }
483
484 // get global locator position
485 const Vector3f pos = state.jointState[jointId].transform * locators[i].offset;
486
487 // change attachment to original joint
488 result[i].parent = skeleton.joints[jointId].parent;
489
490 // Check bounds for parent joint access
491 if (result[i].parent >= state.jointState.size()) {
492 MT_LOGW(
493 "Parent joint ID {} is out of bounds for jointState (size: {})",
494 result[i].parent,
495 state.jointState.size());
496 continue;
497 }
498
499 // calculate new offset
500 const Vector3f offset = state.jointState[result[i].parent].transform.inverse() * pos;
501
502 // change offset to current state
503 result[i].offset = offset;
504 }
505
506 // return
507 return result;
508}

640 markers.emplace_back(Marker{loc.name, pos, false});
641 }
642 }
