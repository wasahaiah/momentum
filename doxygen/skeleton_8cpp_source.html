43template <typename T>
44std::vector<size_t> SkeletonT<T>::getChildrenJoints(const size_t jointId, const bool recursive)
45 const {
46 MT_THROW_IF_T(
47 jointId >= joints.size(),
48 std::out_of_range,
49 "Out of bounds getChildrenJoints query. Requested index: {}. Number of joints: {}",
50 jointId,
51 joints.size());
52
53 std::vector<size_t> childrenJoints;
54 std::vector<int> jointDistance(joints.size(), -1);
55 jointDistance[jointId] = 0;
56
57 // traversal assuming parentJoint < childJoint
58 for (size_t jointIter = jointId + 1; jointIter < joints.size(); ++jointIter) {
59 const auto jointParent = joints[jointIter].parent;
60 const int distParent = (jointParent == kInvalidIndex) ? -1 : jointDistance[jointParent];
61 if ((recursive && distParent >= 0) || (!recursive && distParent == 0)) {
62 jointDistance[jointIter] = distParent + 1;
63 childrenJoints.push_back(jointIter);
64 }
65 }
66
67 return childrenJoints;
68}

70template <typename T>
71bool SkeletonT<T>::isAncestor(size_t jointId, size_t ancestorJointId) const {
72 MT_CHECK(jointId < joints.size(), "{} vs {}", jointId, joints.size());
73 MT_CHECK(ancestorJointId < joints.size(), "{} vs {}", ancestorJointId, joints.size());
74 while (jointId != kInvalidIndex) {
75 if (jointId == ancestorJointId) {
76 return true;
77 }
78 jointId = joints[jointId].parent;
79 }
80
81 return false;
82}

84template <typename T>
85size_t SkeletonT<T>::commonAncestor(size_t joint1, size_t joint2) const {
86 MT_CHECK(joint1 == kInvalidIndex || joint1 < joints.size());
87 MT_CHECK(joint2 == kInvalidIndex || joint2 < joints.size());
88
89 while (joint1 != kInvalidIndex && joint2 != kInvalidIndex && joint1 != joint2) {
90 if (joint1 < joint2) {
91 joint2 = joints[joint2].parent;
92 } else {
93 joint1 = joints[joint1].parent;
94 }
95 }
96
97 return joint1;
98}
