56template <typename T>
57double VertexVertexDistanceErrorFunctionT<T>::getError(
58 const ModelParametersT<T>& modelParameters,
59 const SkeletonStateT<T>& state,
60 const MeshStateT<T>& meshState) {
61 MT_PROFILE_FUNCTION();
62
63 MT_CHECK_NOTNULL(meshState.posedMesh_);
64
65 double error = 0.0;
66
67 for (const auto& constraint : constraints_) {
68 const auto& pos1 = meshState.posedMesh_->vertices[constraint.vertexIndex1];
69 const auto& pos2 = meshState.posedMesh_->vertices[constraint.vertexIndex2];
70
71 const T actualDistance = (pos1 - pos2).norm();
72 const T distanceDiff = actualDistance - constraint.targetDistance;
73
74 error += constraint.weight * distanceDiff * distanceDiff;
75 }
76
77 return error * this->weight_;
78}

100template <typename T>
101double VertexVertexDistanceErrorFunctionT<T>::getJacobian(
102 const ModelParametersT<T>& modelParameters,
103 const SkeletonStateT<T>& state,
104 const MeshStateT<T>& meshState,
105 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
106 Eigen::Ref<Eigen::VectorX<T>> residual,
107 int& usedRows) {
108 MT_PROFILE_FUNCTION();
109
110 MT_CHECK(
111 jacobian.cols() == static_cast<Eigen::Index>(this->parameterTransform_.transform.cols()));
112 MT_CHECK(jacobian.rows() >= static_cast<Eigen::Index>(constraints_.size()));
113 MT_CHECK(residual.rows() >= static_cast<Eigen::Index>(constraints_.size()));
114
115 MT_CHECK_NOTNULL(meshState.posedMesh_);
116
117 double error = 0.0;
118
119 for (size_t i = 0; i < constraints_.size(); i++) {
120 T residualValue;
121 error += calculateJacobian(
122 modelParameters,
123 state,
124 meshState,
125 constraints_[i],
126 jacobian.block(i, 0, 1, modelParameters.size()),
127 residualValue);
128 residual(i) = residualValue;
129 }
130
131 usedRows = static_cast<int>(constraints_.size());
132 return error;
133}

140template <typename T>
141double VertexVertexDistanceErrorFunctionT<T>::calculateJacobian(
142 const ModelParametersT<T>& modelParameters,
143 const SkeletonStateT<T>& state,
144 const MeshStateT<T>& meshState,
145 const VertexVertexDistanceConstraintT<T>& constraint,
146 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
147 T& residual) const {
148 MT_PROFILE_FUNCTION();
149
150 const auto& pos1 = meshState.posedMesh_->vertices[constraint.vertexIndex1];
151 const auto& pos2 = meshState.posedMesh_->vertices[constraint.vertexIndex2];
152
153 const Eigen::Vector3<T> diff = pos1 - pos2;
154 const T actualDistance = diff.norm();
155
156 // Handle degenerate case where vertices are at the same position
157 if (actualDistance == T(0)) {
158 residual = T(0);
159 return T(0); // No meaningful jacobian when distance is zero
160 }
161
162 const T distanceDiff = actualDistance - constraint.targetDistance;
163 const Eigen::Vector3<T> distanceGradient = diff / actualDistance; // normalized difference vector
164
165 // Weight for the jacobian: sqrt(weight * this->weight_)
166 const T wgt = std::sqrt(constraint.weight * this->weight_);
167
168 // Set residual: wgt * distanceDiff
169 residual = wgt * distanceDiff;
170
171 // Calculate jacobian contribution from vertex1 (positive contribution)
172 calculateVertexJacobian(
173 modelParameters, state, meshState, constraint.vertexIndex1, wgt * distanceGradient, jacobian);
174
175 // Calculate jacobian contribution from vertex2 (negative contribution)
176 calculateVertexJacobian(
177 modelParameters,
178 state,
179 meshState,
180 constraint.vertexIndex2,
181 -wgt * distanceGradient,
182 jacobian);
183
184 return wgt * wgt * distanceDiff * distanceDiff;
185}

187template <typename T>
188void VertexVertexDistanceErrorFunctionT<T>::calculateVertexJacobian(
189 const ModelParametersT<T>& /*modelParameters*/,
190 const SkeletonStateT<T>& state,
191 const MeshStateT<T>& meshState,
192 int vertexIndex,
193 const Eigen::Vector3<T>& jacobianDirection,
194 Eigen::Ref<Eigen::MatrixX<T>> jacobian) const {
195 MT_PROFILE_FUNCTION();
196
197 SkinningWeightIteratorT<T> skinningIter(
198 this->character_, *meshState.restMesh_, state, vertexIndex);
199
200 // Handle derivatives wrt joint parameters
201 while (!skinningIter.finished()) {
202 size_t jointIndex = 0;
203 T boneWeight;
204 Eigen::Vector3<T> pos;
205 std::tie(jointIndex, boneWeight, pos) = skinningIter.next();
206
207 // Check for valid index
208 MT_CHECK(jointIndex < this->skeleton_.joints.size());
209
210 const auto& jointState = state.jointState[jointIndex];
211 const size_t paramIndex = jointIndex * kParametersPerJoint;
212 const Eigen::Vector3<T> posd = pos - jointState.translation();
213
214 // Calculate derivatives based on active joints
215 for (size_t d = 0; d < 3; d++) {
216 if (this->activeJointParams_[paramIndex + d]) {
217 // Jacobian wrt translation:
218 jacobian_jointParams_to_modelParams<T>(
219 boneWeight * jacobianDirection.dot(jointState.getTranslationDerivative(d)),
220 paramIndex + d,
221 this->parameterTransform_,
222 jacobian);
223 }
224 if (this->activeJointParams_[paramIndex + 3 + d]) {
225 // Jacobian wrt rotation:
226 jacobian_jointParams_to_modelParams<T>(
227 boneWeight * jacobianDirection.dot(jointState.getRotationDerivative(d, posd)),
228 paramIndex + 3 + d,
229 this->parameterTransform_,
230 jacobian);
231 }
232 }
233 if (this->activeJointParams_[paramIndex + 6]) {
234 // Jacobian wrt scale:
235 jacobian_jointParams_to_modelParams<T>(
236 boneWeight * jacobianDirection.dot(jointState.getScaleDerivative(posd)),
237 paramIndex + 6,
238 this->parameterTransform_,
239 jacobian);
240 }
241 }
242
243 // Handle derivatives wrt blend shape parameters
244 if (this->character_.blendShape) {
245 for (Eigen::Index iBlendShape = 0;
246 iBlendShape < this->parameterTransform_.blendShapeParameters.size();
247 ++iBlendShape) {
248 const auto paramIdx = this->parameterTransform_.blendShapeParameters[iBlendShape];
249 if (paramIdx < 0) {
250 continue;
251 }
252
253 const Eigen::Vector3<T> d_restPos =
254 this->character_.blendShape->getShapeVectors()
255 .template block<3, 1>(3 * vertexIndex, iBlendShape, 3, 1)
256 .template cast<T>();
257 Eigen::Vector3<T> d_worldPos = Eigen::Vector3<T>::Zero();
258 calculateDWorldPos(state, vertexIndex, d_restPos, d_worldPos);
259
260 jacobian(0, paramIdx) += jacobianDirection.dot(d_worldPos);
261 }
262 }
263
264 // Handle derivatives wrt face expression blend shape parameters
265 if (this->character_.faceExpressionBlendShape) {
266 for (Eigen::Index iBlendShape = 0;
267 iBlendShape < this->parameterTransform_.faceExpressionParameters.size();
268 ++iBlendShape) {
269 const auto paramIdx = this->parameterTransform_.faceExpressionParameters[iBlendShape];
270 if (paramIdx < 0) {
271 continue;
272 }
273
274 const Eigen::Vector3<T> d_restPos =
275 this->character_.faceExpressionBlendShape->getShapeVectors()
276 .template block<3, 1>(3 * vertexIndex, iBlendShape, 3, 1)
277 .template cast<T>();
278 Eigen::Vector3<T> d_worldPos = Eigen::Vector3<T>::Zero();
279 calculateDWorldPos(state, vertexIndex, d_restPos, d_worldPos);
280
281 jacobian(0, paramIdx) += jacobianDirection.dot(d_worldPos);
282 }
283 }
284}

286template <typename T>
287double VertexVertexDistanceErrorFunctionT<T>::calculateGradient(
288 const ModelParametersT<T>& modelParameters,
289 const SkeletonStateT<T>& state,
290 const MeshStateT<T>& meshState,
291 const VertexVertexDistanceConstraintT<T>& constraint,
292 Eigen::Ref<Eigen::VectorX<T>> gradient) const {
293 MT_PROFILE_FUNCTION();
294
295 const auto& pos1 = meshState.posedMesh_->vertices[constraint.vertexIndex1];
296 const auto& pos2 = meshState.posedMesh_->vertices[constraint.vertexIndex2];
297
298 const Eigen::Vector3<T> diff = pos1 - pos2;
299 const T actualDistance = diff.norm();
300
301 // Handle degenerate case where vertices are at the same position
302 if (actualDistance == T(0)) {
303 return T(0); // No meaningful gradient when distance is zero
304 }
305
306 const T distanceDiff = actualDistance - constraint.targetDistance;
307 const Eigen::Vector3<T> distanceGradient = diff / actualDistance; // normalized difference vector
308
309 // Weight for the gradient: 2 * weight * distanceDiff * this->weight_
310 const T wgt = T(2) * constraint.weight * distanceDiff * this->weight_;
311
312 // Calculate gradient contribution from vertex1 (positive contribution)
313 calculateVertexGradient(
314 modelParameters, state, meshState, constraint.vertexIndex1, wgt * distanceGradient, gradient);
315
316 // Calculate gradient contribution from vertex2 (negative contribution)
317 calculateVertexGradient(
318 modelParameters,
319 state,
320 meshState,
321 constraint.vertexIndex2,
322 -wgt * distanceGradient,
323 gradient);
324
325 return constraint.weight * distanceDiff * distanceDiff;
326}

427template <typename T>
428void VertexVertexDistanceErrorFunctionT<T>::calculateDWorldPos(
429 const SkeletonStateT<T>& state,
430 int vertexIndex,
431 const Eigen::Vector3<T>& d_restPos,
432 Eigen::Vector3<T>& d_worldPos) const {
433 const auto& skinWeights = *character_.skinWeights;
434
435 for (uint32_t i = 0; i < kMaxSkinJoints; ++i) {
436 const auto w = skinWeights.weight(vertexIndex, i);
437 const auto parentBone = skinWeights.index(vertexIndex, i);
438 if (w > 0) {
439 d_worldPos += w *
440 (state.jointState[parentBone].transform.toLinear() *
441 (character_.inverseBindPose[parentBone].linear().template cast<T>() * d_restPos));
442 }
443 }
