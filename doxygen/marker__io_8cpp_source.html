21size_t markerCount(const std::vector<std::vector<Marker>>& frames) {
22 size_t maxCountMarkers = 0;
23 for (const auto& fr : frames) {
24 const size_t markerCount =
25 std::count_if(fr.begin(), fr.end(), [](const Marker& marker) { return !marker.occluded; });
26 maxCountMarkers = std::max(markerCount, maxCountMarkers);
27 }
28 return maxCountMarkers;
29}

32std::vector<MarkerSequence> loadMarkers(
33 const std::string& filename,
34 UpVector up,
35 std::span<const std::string> validMarkerNames) {
36 const std::string ext = filesystem::path(filename).extension().string();
37 try {
38 if (ext == ".c3d") {
39 return loadC3d(filename, up, validMarkerNames);
40 } else if (ext == ".trc") {
41 return {loadTrc(filename, up)};
42 } else if (ext == ".glb") {
43 return {loadMarkerSequence(filename)};
44 } else if (ext == ".fbx") {
45 return {loadFbxMarkerSequence(filename)};
46 } else {
47 MT_LOGE("{} Unknown marker file type {}", __func__, filename);
48 return {};
49 }
50 } catch (...) {
51 MT_LOGE("{}: Unknown error in {}", __func__, filename);
52 return {};
53 }

69int findMainSubjectIndex(std::span<const MarkerSequence> markerSequences) {
70 if (markerSequences.empty()) {
71 return -1;
72 }
73
74 // We define the main subject as one with the most markers. Ideally, the main subject is the one
75 // with a proper name, because unlabeled markers are usually the largest number, except when there
76 // is only one actor.
77 // XXX Note that the logic could still break, eg. when a person is unnamed but there are named
78 // objects.
79 //
80 // We will go through the sequence to find the max number of visible markers. This is quite slow
81 // but more robust.
82 const size_t numActors = markerSequences.size();
83 std::vector<int> maxMarkers(numActors, 0);
84 // count markers for each actor
85 for (size_t iActor = 0; iActor < numActors; ++iActor) {
86 maxMarkers.at(iActor) = markerCount(markerSequences[iActor].frames);
87 }
88
89 // Special case when there's only one actor.
90 // Return the actor even if it has no markers (empty sequence is valid).
91 if (numActors == 1) {
92 return 0;
93 }
94
95 // find the max marker count with a name
96 size_t maxCount = 0;
97 size_t actorID = 0; // default to be the first actor
98 for (size_t iActor = 0; iActor < numActors; ++iActor) {
99 const size_t count = maxMarkers.at(iActor);
100 auto actorName = markerSequences[iActor].name;
101 std::transform(actorName.begin(), actorName.end(), actorName.begin(), [](unsigned char c) {
102 return std::tolower(c);
103 });
104 if (count > maxCount && !actorName.empty() && actorName != "unlabeled") {
105 maxCount = count;
106 actorID = iActor;
107 }
108 }
109
110 if (maxCount > 0) {
111 return actorID;
112 } else {
113 return -1;
114 }
115}
