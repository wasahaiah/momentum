68template <typename T>
69void GaussNewtonSolverT<T>::doIterationDense() {
70 MT_PROFILE_FUNCTION();
71
72 if (useBlockJtJ_) {
73 // get JtJ and JtR pre-computed
74 MT_PROFILE_EVENT("Get JtJ and JtR");
75 this->error_ = this->solverFunction_->getJtJR(this->parameters_, hessianApprox_, JtR_);
76 } else {
77 MT_PROFILE_EVENT("Get Jacobian and compute JtJ and JtR");
78 // Get the jacobian and compute JtJ and JtR here
79 size_t actualRows = 0;
80 this->error_ =
81 this->solverFunction_->getJacobian(this->parameters_, jacobian_, residual_, actualRows);
82
83 if (hessianApprox_.rows() != this->actualParameters_) {
84 hessianApprox_.resize(this->actualParameters_, this->actualParameters_);
85 }
86
87 const auto JBlock = jacobian_.topLeftCorner(actualRows, this->actualParameters_);
88
89 hessianApprox_.template triangularView<Eigen::Lower>() = JBlock.transpose() * JBlock;
90
91 JtR_.noalias() = JBlock.transpose() * residual_.head(actualRows);
92 }
93
94 // calculate the step direction according to the gauss newton update
95 Eigen::VectorX<T> delta = Eigen::VectorX<T>::Zero(this->numParameters_);
96 {
97 MT_PROFILE_EVENT("Dense gauss newton step");
98
99 // delta = (Jt*J)^-1*Jt*r ...
100 // - add some regularization to make sure the system is never unstable and explodes in weird
101 // ways.
102 hessianApprox_.diagonal().array() += alpha_;
103
104 // - llt solve
105 delta.head(this->actualParameters_) = llt_.compute(hessianApprox_).solve(JtR_);
106 }
107
108 updateParameters(delta);
109
110 {
111 MT_PROFILE_EVENT("Store history");
112 if (this->storeHistory) {
113 auto& jtjHist = this->iterationHistory_["jtj"];
114 if (jtjHist.rows() !=
115 gsl::narrow_cast<Eigen::Index>(this->actualParameters_ * this->getMaxIterations()) ||
116 jtjHist.cols() != gsl::narrow_cast<Eigen::Index>(this->actualParameters_)) {
117 jtjHist.resize(this->actualParameters_ * this->getMaxIterations(), this->actualParameters_);
118 }
119 jtjHist.block(
120 this->iteration_ * this->actualParameters_,
121 0,
122 this->actualParameters_,
123 this->actualParameters_) = hessianApprox_;
124 }
125 }
126}

128template <typename T>
129void GaussNewtonSolverT<T>::doIterationSparse() {
130 MT_PROFILE_FUNCTION();
131
132 Eigen::VectorX<T> delta;
133
134 if (!directSparseJtJ_) { // make a dense matrix and sparsify later
135 if (useBlockJtJ_) {
136 // get JtJ and JtR pre-computed
137 {
138 MT_PROFILE_EVENT("Get sparse JtJ and JtR");
139 this->error_ = this->solverFunction_->getJtJR(this->parameters_, hessianApprox_, JtR_);
140 }
141
142 {
143 MT_PROFILE_EVENT("Get sparse JtJ and JtR");
144 // sparsify the system
145 JtJ_ = hessianApprox_.sparseView();
146 }
147 } else {
148 MT_PROFILE_EVENT("Sparse gauss newton step");
149
150 // get the jacobian and residual
151 size_t size = 0;
152 {
153 MT_PROFILE_EVENT("get Jacobian");
154 this->error_ =
155 this->solverFunction_->getJacobian(this->parameters_, jacobian_, residual_, size);
156 }
157
158 // sparsify the system
159 const Eigen::SparseMatrix<T> sjac =
160 jacobian_.topLeftCorner(size, this->actualParameters_).sparseView();
161 JtR_.noalias() = sjac.transpose() * residual_.head(size);
162
163 JtJ_ = sjac.transpose() * sjac;
164 }
165 } else { // directly sparse JtJ
166 MT_PROFILE_EVENT("Get sparse JtJ and JtR");
167 this->error_ = this->solverFunction_->getJtJR_Sparse(this->parameters_, JtJ_, JtR_);
168 }
169
170 if (D_.innerSize() != this->actualParameters_) {
171 D_.resize(this->actualParameters_, this->actualParameters_);
172 D_.setIdentity();
173 }
174 JtJ_ += D_ * alpha_;
175
176 // Symbolic decomposition, only needed if the params pattern changed
177 if (this->newParameterPattern_) {
178 MT_PROFILE_EVENT("Sparse analyze");
179 lltSolver_.analyzePattern(JtJ_);
180 this->newParameterPattern_ = !directSparseJtJ_; // works fine for sparse matrix with explicit 0s
181 }
182
183 // Numerical update with the new coefficients
184 {
185 MT_PROFILE_EVENT("Sparse factorization");
186 lltSolver_.factorize(JtJ_);
187 }
188
189 // Solve, compute the gauss-newton step
190 {
191 MT_PROFILE_EVENT("Sparse solve");
192 delta.setZero(this->numParameters_);
193 delta.head(this->actualParameters_) = lltSolver_.solve(JtR_);
194 }
195
196 updateParameters(delta);
197
198 {
199 MT_PROFILE_EVENT("Store history");
200 if (this->storeHistory) {
201 this->iterationHistory_["solver_err"].setZero(1, 1);
202 if (lltSolver_.info() != Eigen::Success) {
203 this->iterationHistory_["solver_err"](0, 0) = 1.0;
204 }
205
206 this->iterationHistory_["jtr_norm"].resize(1, 1);
207 this->iterationHistory_["jtr_norm"](0, 0) = JtR_.norm();
208 }
209 }
210}

212template <typename T>
213void GaussNewtonSolverT<T>::updateParameters(Eigen::VectorX<T>& delta) {
214 if (!doLineSearch_) {
215 MT_PROFILE_EVENT("Update params");
216 this->solverFunction_->updateParameters(this->parameters_, delta);
217 return;
218 }
219
220 MT_PROFILE_EVENT("Line search");
221
222 static constexpr T kC1 = 1e-3;
223 static constexpr T kTau = 0.5;
224 static constexpr size_t kMaxLineSearchSteps = 10;
225
226 const T scaledError = kC1 * this->error_;
227 const Eigen::VectorX<T> parametersOrig = this->parameters_;
228 T lineSearchScale = 1.0;
229
230 for (size_t i = 0; i < kMaxLineSearchSteps && std::isnormal(lineSearchScale); ++i) {
231 this->parameters_ = parametersOrig;
232
233 this->solverFunction_->updateParameters(this->parameters_, lineSearchScale * delta);
234
235 const double errorNew = this->solverFunction_->getError(this->parameters_);
236 if ((this->error_ - errorNew) >= lineSearchScale * scaledError) {
237 break;
238 }
239
240 // Reduce step size:
241 lineSearchScale *= kTau;
242 }
243}
