18bool PoseConstraint::operator==(const PoseConstraint& poseConstraint) const {
19 // Compare the parameterIdValue as sets
20 std::map<size_t, float> paramIdToValue1;
21 std::copy(
22 parameterIdValue.begin(),
23 parameterIdValue.end(),
24 std::inserter(paramIdToValue1, paramIdToValue1.begin()));
25
26 std::map<size_t, float> paramIdToValue2;
27 std::copy(
28 poseConstraint.parameterIdValue.begin(),
29 poseConstraint.parameterIdValue.end(),
30 std::inserter(paramIdToValue2, paramIdToValue2.begin()));
31
32 if (paramIdToValue1.size() != paramIdToValue2.size()) {
33 return false;
34 }
35
36 auto pred = [](const auto& l, const auto& r) {
37 return ((l.first == r.first) && isApprox(l.second, r.second));
38 };
39 return std::equal(paramIdToValue1.begin(), paramIdToValue1.end(), paramIdToValue2.begin(), pred);
40}

77 ParameterTransformT<T> result;
78 const size_t nJoints = jointNames.size();
79 const size_t nJointParameters = nJoints * kParametersPerJoint;
80 result.offsets.setZero(nJointParameters);
81 result.activeJointParams.setConstant(nJointParameters, true);
82 result.transform.resize(nJointParameters, nJointParameters);
83 result.transform.setIdentity();
84 result.name.resize(nJointParameters);
85 for (size_t i = 0; i < nJoints; i++) {
86 for (size_t j = 0; j < kParametersPerJoint; ++j) {

88 fmt::format("{}_{}", jointNames[i], kJointParameterNames[j]);
89 }
90 }
91 return result;

108template <typename T>
109JointParametersT<T> ParameterTransformT<T>::apply(const ModelParametersT<T>& parameters) const {
110 MT_PROFILE_FUNCTION();
111 MT_CHECK(
112 parameters.size() == transform.cols(),
113 "Size mismatch: The size of the parameters vector '{}' does not match the number of columns in the transform matrix '{}'.",
114 parameters.size(),
115 transform.cols());
116 MT_CHECK(
117 offsets.size() == transform.rows(),
118 "Size mismatch: The size of the offsets vector '{}' does not match the number of rows in the transform matrix '{}'.",
119 offsets.size(),
120 transform.rows());
121 return transform * parameters.v + offsets;
122}

141 }

229template <typename T>
230ParameterTransformT<T> mapParameterTransformJoints(
231 const ParameterTransformT<T>& parameterTransform,
232 size_t numTargetJoints,
233 const std::vector<size_t>& jointMapping) {
234 ParameterTransformT<T> mappedTransform;
235
236 // No change in the parameters:
237 mappedTransform.name = parameterTransform.name;
238
239 // resize the offset matrix to the right size
240 mappedTransform.offsets.setZero(numTargetJoints * kParametersPerJoint);
241 mappedTransform.activeJointParams.setConstant(numTargetJoints * kParametersPerJoint, false);
242
243 // map the offset from the anim skel to the new one
244 for (size_t i = 0; i < jointMapping.size(); i++) {
245 if (jointMapping[i] != kInvalidIndex) {
246 for (size_t d = 0; d < kParametersPerJoint; d++) {
247 mappedTransform.offsets[jointMapping[i] * kParametersPerJoint + d] =
248 parameterTransform.offsets(i * kParametersPerJoint + d);
249 }
250 }
251 }
252
253 // map the transform from the anim skel to the new one
254 std::vector<Eigen::Triplet<float>> triplets;
255 for (int k = 0; k < parameterTransform.transform.outerSize(); ++k) {
256 for (typename SparseRowMatrix<T>::InnerIterator it(parameterTransform.transform, k); it; ++it) {
257 // row is joint + type index, col is parameter
258 const int jIndex = static_cast<int>(it.row()) / kParametersPerJoint;
259 const int jOffset = static_cast<int>(it.row()) % kParametersPerJoint;
260
261 const auto mappedJoint = jointMapping[jIndex];
262 if (mappedJoint == kInvalidIndex) {
263 continue;
264 }
265
266 triplets.push_back(
267 Eigen::Triplet<float>(
268 mappedJoint * kParametersPerJoint + jOffset, static_cast<int>(it.col()), it.value()));
269
270 // enable joint channels
271 mappedTransform.activeJointParams[mappedJoint * kParametersPerJoint + jOffset] = true;
272 }
273 }
274
275 // resize the Transform matrix to the correct size
276 mappedTransform.transform.resize(
277 static_cast<int>(numTargetJoints) * kParametersPerJoint,
278 static_cast<int>(mappedTransform.name.size()));
279
280 // create sparse matrix from triplet
281 mappedTransform.transform.setFromTriplets(triplets.begin(), triplets.end());
282
283 // copy over the parameterSets
284 mappedTransform.parameterSets = parameterTransform.parameterSets;
285 mappedTransform.poseConstraints = parameterTransform.poseConstraints;
286 mappedTransform.blendShapeParameters = parameterTransform.blendShapeParameters;
287 mappedTransform.faceExpressionParameters = parameterTransform.faceExpressionParameters;
288 mappedTransform.skinnedLocatorParameters = parameterTransform.skinnedLocatorParameters;
289
290 return mappedTransform;
291}

293template <typename T>
294std::tuple<ParameterTransformT<T>, ParameterLimits> subsetParameterTransform(
295 const ParameterTransformT<T>& paramTransformOld,
296 const ParameterLimits& paramLimitsOld,
297 const ParameterSet& paramSet) {
298 const auto nOldParam = paramTransformOld.numAllModelParameters();
299
300 std::vector<size_t> oldParamToNewParam(nOldParam, kInvalidIndex);
301 std::vector<size_t> newParamToOldParam;
302
303 for (Eigen::Index iOldParam = 0; iOldParam < nOldParam; ++iOldParam) {
304 if (!paramSet.test(iOldParam)) {
305 continue;
306 }
307
308 oldParamToNewParam[iOldParam] = newParamToOldParam.size();
309 newParamToOldParam.push_back(iOldParam);
310 }
311
312 const auto nNewParam = newParamToOldParam.size();
313
314 ParameterTransformT<T> paramTransformNew;
315
316 paramTransformNew.name.reserve(nNewParam);
317 for (const auto iOldParam : newParamToOldParam) {
318 paramTransformNew.name.push_back(paramTransformOld.name[iOldParam]);
319 }
320
321 // Offsets are unchanged because they are the size #joints, not #param.
322 paramTransformNew.offsets = paramTransformOld.offsets;
323
324 paramTransformNew.activeJointParams.setConstant(paramTransformOld.numJointParameters(), false);
325
326 // Remap the transformation.
327 using IndexType = typename SparseRowMatrix<T>::Index;
328 std::vector<Eigen::Triplet<T>> tripletsNew;
329 for (int k = 0; k < paramTransformOld.transform.outerSize(); ++k) {
330 for (typename SparseRowMatrix<T>::InnerIterator it(paramTransformOld.transform, k); it; ++it) {
331 const auto iOldParam = it.col();
332 const auto iNewParam = oldParamToNewParam[iOldParam];
333
334 if (iNewParam == kInvalidIndex) {
335 continue;
336 }
337
338 tripletsNew.emplace_back((int)it.row(), (int)iNewParam, it.value());
339 paramTransformNew.activeJointParams[it.row()] = true;
340 }
341 }
342
343 paramTransformNew.transform.resize(paramTransformOld.transform.rows(), (IndexType)nNewParam);
344 paramTransformNew.transform.setFromTriplets(tripletsNew.begin(), tripletsNew.end());
345
346 for (const auto& [name, constraintsOld] : paramTransformOld.poseConstraints) {
347 PoseConstraint pcNew;
348 for (const auto& [idxOld, weight] : constraintsOld.parameterIdValue) {
349 auto idxNew = oldParamToNewParam[idxOld];
350 if (idxNew != kInvalidIndex) {
351 pcNew.parameterIdValue.emplace_back(idxNew, weight);
352 }
353 }
354
355 paramTransformNew.poseConstraints.insert(std::make_pair(name, pcNew));
356 }
357
358 paramTransformNew.blendShapeParameters =
359 Eigen::VectorXi::Constant(paramTransformOld.blendShapeParameters.size(), -1);
360 for (Eigen::Index iBlendShape = 0; iBlendShape < paramTransformOld.blendShapeParameters.size();
361 ++iBlendShape) {
362 auto paramOld = paramTransformOld.blendShapeParameters(iBlendShape);
363 if (paramOld >= 0) {
364 auto paramNew = oldParamToNewParam[paramOld];
365 if (paramNew != kInvalidIndex) {
366 paramTransformNew.blendShapeParameters(iBlendShape) = paramNew;
367 }
368 }
369 }
370
371 paramTransformNew.faceExpressionParameters =
372 Eigen::VectorXi::Constant(paramTransformOld.faceExpressionParameters.size(), -1);
373 for (Eigen::Index iBlendShape = 0;
374 iBlendShape < paramTransformOld.faceExpressionParameters.size();
375 ++iBlendShape) {
376 auto paramOld = paramTransformOld.faceExpressionParameters(iBlendShape);
377 if (paramOld >= 0) {
378 auto paramNew = oldParamToNewParam[paramOld];
379 if (paramNew != kInvalidIndex) {
380 paramTransformNew.faceExpressionParameters(iBlendShape) = paramNew;
381 }
382 }
383 }
384
385 paramTransformNew.skinnedLocatorParameters =
386 Eigen::VectorXi::Constant(paramTransformOld.skinnedLocatorParameters.size(), -1);
387 for (Eigen::Index iSkinnedLocator = 0;
388 iSkinnedLocator < paramTransformOld.skinnedLocatorParameters.size();
389 ++iSkinnedLocator) {
390 auto paramOld = paramTransformOld.skinnedLocatorParameters(iSkinnedLocator);
391 if (paramOld < 0) {
392 continue;
393 }
394
395 bool anyMapped = false;
396 for (int k = 0; k < 3; ++k) {
397 anyMapped = anyMapped || (oldParamToNewParam[paramOld + k] != kInvalidIndex);
398 }
399
400 if (anyMapped) {
401 auto paramNew = oldParamToNewParam[paramOld];
402 for (int k = 0; k < 3; ++k) {
403 if (oldParamToNewParam[paramOld + k] != paramNew + k) {
404 throw std::runtime_error(
405 fmt::format(
406 "When mapping skinned locator parameters, must retain all 3 parameters for each locator. Locator {} is missing parameter {} ({}).",
407 iSkinnedLocator,
408 paramOld + k,
409 paramTransformOld.name[paramOld + k]));
410 }
411 }
412
413 paramTransformNew.skinnedLocatorParameters(iSkinnedLocator) = gsl::narrow_cast<int>(paramNew);
414 }
415 }
416
417 for (const auto& paramSetOld : paramTransformOld.parameterSets) {
418 ParameterSet paramSetNew;
419 paramSetNew.reset();
420
421 for (size_t iNewParam = 0; iNewParam < nNewParam; ++iNewParam) {
422 const auto iOldParam = newParamToOldParam[iNewParam];
423 paramSetNew.set(iNewParam, paramSetOld.second.test(iOldParam));
424 }
425
426 paramTransformNew.parameterSets.emplace(paramSetOld.first, paramSetNew);
427 }
428
429 ParameterLimits paramLimitsNew;
430 paramLimitsNew.reserve(paramLimitsOld.size());
431 for (const auto& limitOld : paramLimitsOld) {
432 auto limitNew = limitOld;
433 switch (limitOld.type) {
434 case MinMax: {
435 auto& data = limitNew.data.minMax;
436 data.parameterIndex = oldParamToNewParam[data.parameterIndex];
437 if (data.parameterIndex == kInvalidIndex) {
438 continue;
439 }
440 break;
441 }
442
443 case MinMaxJoint: {
444 break;
445 }
446
447 case Linear: {
448 auto& data = limitNew.data.linear;
449 data.referenceIndex = oldParamToNewParam[data.referenceIndex];
450 data.targetIndex = oldParamToNewParam[data.targetIndex];
451 if (data.referenceIndex == kInvalidIndex || data.targetIndex == kInvalidIndex) {
452 continue;
453 }
454 break;
455 }
456
457 case LinearJoint: {
458 break;
459 }
460
461 case HalfPlane: {
462 auto& data = limitNew.data.halfPlane;
463 data.param1 = oldParamToNewParam[data.param1];
464 data.param2 = oldParamToNewParam[data.param2];
465 if (data.param1 == kInvalidIndex || data.param2 == kInvalidIndex) {
466 continue;
467 }
468 break;
469 }
470
471 case Ellipsoid:
472 case MinMaxJointPassive:
473 case LimitTypeCount: {
474 // nothing to do here.
475 break;
476 }
477 }
478
479 paramLimitsNew.push_back(limitNew);
480 }
481
482 return std::make_tuple(paramTransformNew, paramLimitsNew);
483}

539template <typename T>
540bool ParameterTransformT<T>::isApprox(const ParameterTransformT<T>& parameterTransform) const {
541 bool isTransformEqual = false;
542 if (transform.cols() > 0 && transform.rows() > 0 && parameterTransform.transform.cols() > 0 &&
543 parameterTransform.transform.rows() > 0) {
544 isTransformEqual = transform.isApprox(parameterTransform.transform);
545 } else {
546 isTransformEqual = (transform.cols() == parameterTransform.transform.cols()) &&
547 (transform.rows() == parameterTransform.transform.rows());
548 }
549 if (!isTransformEqual) {
550 return false;
551 }
552
553 return (
554 (name == parameterTransform.name) && offsets.isApprox(parameterTransform.offsets) &&
555 activeJointParams.isApprox(parameterTransform.activeJointParams) &&
556 (parameterSets == parameterTransform.parameterSets) &&
557 (poseConstraints == parameterTransform.poseConstraints) &&
558 (blendShapeParameters == parameterTransform.blendShapeParameters) &&
559 (faceExpressionParameters == parameterTransform.faceExpressionParameters) &&
560 (skinnedLocatorParameters == parameterTransform.skinnedLocatorParameters));
561}

563std::tuple<ParameterTransform, ParameterLimits> addBlendShapeParameters(
564 ParameterTransform paramTransform,
565 ParameterLimits paramLimits,
566 Eigen::Index nBlendShapes) {
567 // First, strip out any existing blend shape parameters, to make sure this
568 // operation is idempotent.
569 std::tie(paramTransform, paramLimits) = subsetParameterTransform(
570 paramTransform, paramLimits, ~paramTransform.getBlendShapeParameters());
571
572 // Now add in the additional parameters:
573 paramTransform.blendShapeParameters.setZero(nBlendShapes);
574 for (Eigen::Index iBlend = 0; iBlend < nBlendShapes; ++iBlend) {
575 std::ostringstream oss;
576 oss << "blend_" << iBlend;
577 paramTransform.blendShapeParameters(iBlend) = (int)paramTransform.name.size();
578 paramTransform.name.push_back(oss.str());
579 }
580
581 paramTransform.transform.conservativeResize(
582 paramTransform.transform.rows(), paramTransform.name.size());
583 paramTransform.transform.makeCompressed();
584
585 return {paramTransform, paramLimits};
586}

613std::tuple<ParameterTransform, ParameterLimits> addSkinnedLocatorParameters(
614 ParameterTransform paramTransform,
615 ParameterLimits paramLimits,
616 const std::vector<bool>& activeLocators,
617 const std::vector<std::string>& locatorNames) {
618 // First, strip out any existing blend shape parameters, to make sure this
619 // operation is idempotent.
620 std::tie(paramTransform, paramLimits) = subsetParameterTransform(
621 paramTransform, paramLimits, ~paramTransform.getSkinnedLocatorParameters());
622
623 const auto nLocators = activeLocators.size();
624 MT_CHECK(locatorNames.empty() || locatorNames.size() == nLocators);
625 paramTransform.skinnedLocatorParameters.resize(nLocators);
626 paramTransform.skinnedLocatorParameters.setConstant(-1);
627
628 ParameterSet paramSet;
629
630 // Now add in the additional parameters:
631 for (size_t iLocator = 0; iLocator < activeLocators.size(); ++iLocator) {
632 if (!activeLocators[iLocator]) {
633 continue;
634 }
635
636 paramTransform.skinnedLocatorParameters(iLocator) = (int)paramTransform.name.size();
637
638 auto locatorName =
639 iLocator < locatorNames.size() ? locatorNames[iLocator] : std::to_string(iLocator);
640 std::array<const char*, 3> dirNames{"x", "y", "z"};
641 for (const auto& d : dirNames) {
642 paramSet.set(paramTransform.name.size());
643 std::ostringstream oss;
644 oss << "locator_" << locatorName << "_" << d;
645 paramTransform.name.push_back(oss.str());
646 }
647 }
648
649 paramTransform.parameterSets["skinnedLocators"] = paramSet;
650 paramTransform.transform.conservativeResize(
651 paramTransform.transform.rows(), paramTransform.name.size());
652 paramTransform.transform.makeCompressed();
653
654 return {paramTransform, paramLimits};
