14template <typename T>
15UnionErrorFunctionT<T>::UnionErrorFunctionT(
16 const Skeleton& skel,
17 const ParameterTransform& pt,
18 const std::vector<std::shared_ptr<SkeletonErrorFunctionT<T>>>& errorFunctions)
19 : SkeletonErrorFunctionT<T>(skel, pt), errorFunctions_(errorFunctions) {
20 // Should this be allowed?
21 MT_THROW_IF(errorFunctions_.empty(), "No error functions in union?");
22
23 for (size_t i = 0; i < errorFunctions_.size(); ++i) {
24 auto diff_e =
25 std::dynamic_pointer_cast<FullyDifferentiableSkeletonErrorFunctionT<T>>(errorFunctions_[i]);
26 diffErrorFunctions_.push_back(diff_e);
27
28 if (diff_e) {
29 for (const auto& input : diff_e->inputs()) {
30 inputs_[input].push_back(i);
31 }
32 }
33 }
34
35 {
36 std::vector<std::string> names;
37 for (const auto& e : diffErrorFunctions_) {
38 if (e) {
39 names.push_back(e->name());
40 } else {
41 names.push_back("Unknown");
42 }
43 }
44
45 std::ostringstream name;
46 name << "Union[";
47 if (std::adjacent_find(std::begin(names), std::end(names), std::not_equal_to<>()) ==
48 std::end(names)) {
49 // All the same:
50 name << names.front() << " x " << names.size();
51 } else {
52 bool first = true;
53 for (const auto& n : names) {
54 if (!first) {
55 name << ", ";
56 }
57 name << n;
58 first = false;
59 }
60 }
61 name << "]";
62 name_ = name.str();
63 }
64}

66template <typename T>
67double UnionErrorFunctionT<T>::getError(
68 const ModelParametersT<T>& params,
69 const SkeletonStateT<T>& state,
70 const MeshStateT<T>& meshState) {
71 double error = 0;

92template <typename T>
93double UnionErrorFunctionT<T>::getJacobian(
94 const ModelParametersT<T>& params,
95 const SkeletonStateT<T>& state,
96 const MeshStateT<T>& meshState,
97 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
98 Eigen::Ref<Eigen::VectorX<T>> residual,
99 int& usedRows) {
100 double error = 0;
101 size_t jacobianStart = 0;
102 for (auto& fun : errorFunctions_) {
103 const size_t memberJacobianSize = fun->getJacobianSize();
104
105 error += fun->getJacobian(
106 params,
107 state,
108 meshState,
109 jacobian.block(jacobianStart, 0, memberJacobianSize, jacobian.cols()),
110 residual.segment(jacobianStart, memberJacobianSize),
111 usedRows);
112
113 jacobianStart += memberJacobianSize;
114 }
115
116 jacobian *= std::sqrt(this->weight_);
117 residual *= std::sqrt(this->weight_);
118
119 usedRows = static_cast<int>(jacobian.rows());
120
121 return this->weight_ * error;
122}

188template <typename T>
189Eigen::VectorX<T> UnionErrorFunctionT<T>::d_gradient_d_input_dot(
190 const std::string& name,
191 const ModelParametersT<T>& modelParams,
192 const SkeletonStateT<T>& state,
193 Eigen::Ref<const Eigen::VectorX<T>> inputVec) {
194 auto itr = inputs_.find(name);
195 MT_THROW_IF(
196 itr == inputs_.end(),
197 "Invalid input '{}' for UnionErrorFunction::d_gradient_d_input_dot",
198 name);
199
200 Eigen::Index currentOffset = 0;
201 Eigen::VectorX<T> result = Eigen::VectorX<T>::Zero(getInputSize(name));
202
203 for (const auto& iErrFun : itr->second) {
204 const auto errFun = diffErrorFunctions_[iErrFun];
205 const Eigen::Index sz = errFun->getInputSize(name);
206 result.segment(currentOffset, sz) =
207 errFun->d_gradient_d_input_dot(name, modelParams, state, inputVec);
208 currentOffset += sz;
209 }
210
211 result *= this->weight_;
212
213 return result;
214}
