40template <typename T>
41void CollisionErrorFunctionT<T>::updateCollisionPairs() {
42 // Clear any existing collision pairs
43 excludingPairIds_.clear();
44 jacobianSize_ = 0;
45
46 // Create a zero state for the skeleton
47 const SkeletonStateT<T> state(
48 this->parameterTransform_.zero().template cast<T>(), this->skeleton_);
49
50 // Update the collision state based on the zero state and collision geometry
51 collisionState_.update(state, collisionGeometry_);
52
53 // Get the size of collisionGeometry_
54 const auto n = collisionGeometry_.size();
55
56 // Build BVH
57 std::vector<axel::BoundingBox<T>> aabbs(n);
58 for (size_t i = 0; i < n; ++i) {
59 auto& aabb = aabbs[i];
60 aabb.id = i;
61 updateAabb(
62 aabb, collisionState_.origin[i], collisionState_.direction[i], collisionState_.radius[i]);
63 }
64 bvh_.setBoundingBoxes(aabbs);
65
66 // Exit early as there are no geometry objects to process
67 if (n == 0) {
68 return;
69 }
70
71 // Check if the pair intersects in the initial pose
72 const auto numPotentialPairs = n * (n - 1) / 2;
73 excludingPairIds_.reserve(numPotentialPairs);
74 T distance;
75 Vector2<T> cp;
76 T overlap;
77 for (size_t i = 0; i < n; ++i) {
78 for (size_t j = i + 1; j < n; ++j) {
79 const std::pair<size_t, size_t> pairId = std::make_pair(i, j);
80
81 if (overlaps(
82 collisionState_.origin[i],
83 collisionState_.direction[i],
84 collisionState_.radius[i],
85 collisionState_.delta[i],
86 collisionState_.origin[j],
87 collisionState_.direction[j],
88 collisionState_.radius[j],
89 collisionState_.delta[j],
90 distance,
91 cp,
92 overlap)) {
93 excludingPairIds_.insert(pairId);
94 continue;
95 }
96
97 // Walk down the this->skeleton_ hierarchy for the collision pair until we find a common
98 // ancestor
99 size_t p0 = collisionGeometry_[i].parent;
100 size_t p1 = collisionGeometry_[j].parent;
101 size_t count = 0;
102
103 // Find the common ancestor. We can always just walk up the higher joint index because the
104 // parent always has to have a lower index than the child joint.
105 while (p0 != p1) {
106 if (p0 > p1) {
107 p0 = this->skeleton_.joints[p0].parent;
108 } else {
109 p1 = this->skeleton_.joints[p1].parent;
110 }
111 count++;
112 }
113
114 // Store collision pairs excluding those with the same parent or adjacent joints
115 if (count > 1) {
116 jacobianSize_++;
117 } else {
118 excludingPairIds_.insert(pairId);
119 }
120 }
121 }
122 // TODO: Consider using BVH for faster intersection checks
123}

125template <typename T>
126void CollisionErrorFunctionT<T>::computeBroadPhase(const SkeletonStateT<T>& state) {
127 // Update collision state
128 {
129 MT_PROFILE_EVENT("Collision: updateState");
130 collisionState_.update(state, collisionGeometry_);
131 }
132
133 // Update the AABBs for each geometry object
134 for (auto& aabb : bvh_.getPrimitives()) {
135 const auto& index = aabb.id;
136 updateAabb(
137 aabb,
138 collisionState_.origin[index],
139 collisionState_.direction[index],
140 collisionState_.radius[index]);
141 }
142
143 // Update BVH
144 bvh_.setBoundingBoxes(bvh_.getPrimitives());
145}

147template <typename T>
148std::vector<Vector2i> CollisionErrorFunctionT<T>::getCollisionPairs() const {
149 std::vector<Vector2i> collidingPairs;
150
151 bvh_.traverseOverlappingPairs([&](auto indexA, auto indexB) -> bool {
152 const std::pair<size_t, size_t> pairId = std::make_pair(indexA, indexB);
153
154 // Skip excluded pairs
155 if (excludingPairIds_.find(pairId) != excludingPairIds_.end()) {
156 return true; // continue traversing
157 }
158
159 // Perform narrow-phase collision check
160 T distance;
161 Vector2<T> cp;
162 T overlap;
163 if (!overlaps(
164 collisionState_.origin[indexA],
165 collisionState_.direction[indexA],
166 collisionState_.radius[indexA],
167 collisionState_.delta[indexA],
168 collisionState_.origin[indexB],
169 collisionState_.direction[indexB],
170 collisionState_.radius[indexB],
171 collisionState_.delta[indexB],
172 distance,
173 cp,
174 overlap)) {
175 return true; // continue traversing
176 }
177
178 collidingPairs.emplace_back(indexA, indexB);
179
180 return true; // continue traversal
181 });
182
183 return collidingPairs;
184}

186template <typename T>
187double CollisionErrorFunctionT<T>::getError(
188 const ModelParametersT<T>& /* params */,
189 const SkeletonStateT<T>& state,
190 const MeshStateT<T>& /* meshState */) {
191 if (state.jointState.empty()) {
192 return 0.0;
193 }
194
195 // Check all is valid
196 MT_CHECK(
197 state.jointParameters.size() ==
198 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
199
200 computeBroadPhase(state);
201
202 // Traverse all the overlapping pairs and compute the error
203 double error = 0;
204 bvh_.traverseOverlappingPairs([&](auto indexA, auto indexB) -> bool {
205 const std::pair<size_t, size_t> pairId = std::make_pair(indexA, indexB);
206
207 // Skip excluded pairs
208 if (excludingPairIds_.find(pairId) != excludingPairIds_.end()) {
209 return true; // continue traversing
210 }
211
212 // Perform narrow-phase collision check
213 T distance;
214 Vector2<T> cp;
215 T overlap;
216 if (!overlaps(
217 collisionState_.origin[indexA],
218 collisionState_.direction[indexA],
219 collisionState_.radius[indexA],
220 collisionState_.delta[indexA],
221 collisionState_.origin[indexB],
222 collisionState_.direction[indexB],
223 collisionState_.radius[indexB],
224 collisionState_.delta[indexB],
225 distance,
226 cp,
227 overlap)) {
228 return true; // continue traversing
229 }
230
231 error += sqr(overlap) * kCollisionWeight * this->weight_;
232
233 return true; // continue traversal
234 });
235
236 return error;
237}

239template <typename T>
240double CollisionErrorFunctionT<T>::getGradient(
241 const ModelParametersT<T>& /* params */,
242 const SkeletonStateT<T>& state,
243 const MeshStateT<T>& /* meshState */,
244 Ref<VectorX<T>> gradient) {
245 if (state.jointState.empty()) {
246 return 0.0;
247 }
248
249 // check all is valid
250 MT_CHECK(
251 state.jointParameters.size() ==
252 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
253
254 computeBroadPhase(state);
255
256 // Traverse all the overlapping pairs and compute the error
257 double error = 0;
258 bvh_.traverseOverlappingPairs([&](auto indexA, auto indexB) -> bool {
259 const std::pair<size_t, size_t> pairId = std::make_pair(indexA, indexB);
260
261 // Skip excluded pairs
262 if (excludingPairIds_.find(pairId) != excludingPairIds_.end()) {
263 return true; // continue traversing
264 }
265
266 // Perform narrow-phase collision check
267 T distance;
268 Vector2<T> cp;
269 T overlap;
270 if (!overlaps(
271 collisionState_.origin[indexA],
272 collisionState_.direction[indexA],
273 collisionState_.radius[indexA],
274 collisionState_.delta[indexA],
275 collisionState_.origin[indexB],
276 collisionState_.direction[indexB],
277 collisionState_.radius[indexB],
278 collisionState_.delta[indexB],
279 distance,
280 cp,
281 overlap)) {
282 return true; // continue traversing
283 }
284
285 const auto& jointA = collisionGeometry_[indexA].parent;
286 const auto& jointB = collisionGeometry_[indexB].parent;
287 const size_t commonAncestor = this->skeleton_.commonAncestor(jointA, jointB);
288 error += sqr(overlap) * kCollisionWeight * this->weight_;
289
290 // calculate collision resolve direction. this is what we need to push joint parent i in.
291 // the direction for joint parent j is the inverse
292 const Vector3<T> position_i =
293 (collisionState_.origin[indexA] + collisionState_.direction[indexA] * cp[0]);
294 const Vector3<T> position_j =
295 (collisionState_.origin[indexB] + collisionState_.direction[indexB] * cp[1]);
296 const Vector3<T> direction = position_i - position_j;
297 const T overlapFraction = overlap / distance;
298
299 // calculate weight
300 const T wgt = -T(2) * kCollisionWeight * this->weight_ * overlapFraction;
301
302 // -----------------------------------
303 // process first joint
304 // -----------------------------------
305 size_t jointIndex = collisionGeometry_[indexA].parent;
306 while (jointIndex != kInvalidIndex && jointIndex != commonAncestor) {
307 const auto& jointState = state.jointState[jointIndex];
308 const size_t paramIndex = jointIndex * kParametersPerJoint;
309 const Vector3<T> posd = position_i - jointState.translation();
310
311 // calculate derivatives based on active joints
312 for (size_t d = 0; d < 3; d++) {
313 if (this->activeJointParams_[paramIndex + d]) {
314 // calculate joint gradient
315 const T val = direction.dot(jointState.getTranslationDerivative(d)) * wgt;
316 // explicitly multiply with the parameter transform to generate parameter space
317 // gradients
318 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
319 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
320 ++index) {
321 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
322 val * this->parameterTransform_.transform.valuePtr()[index];
323 }
324 }
325 if (this->activeJointParams_[paramIndex + 3 + d]) {
326 // calculate joint gradient
327 const T val = direction.dot(jointState.getRotationDerivative(d, posd)) * wgt;
328 // explicitly multiply with the parameter transform to generate parameter space
329 // gradients
330 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
331 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
332 ++index) {
333 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
334 val * this->parameterTransform_.transform.valuePtr()[index];
335 }
336 }
337 }
338 if (this->activeJointParams_[paramIndex + 6]) {
339 // calculate joint gradient
340 const T val = direction.dot(jointState.getScaleDerivative(posd)) * wgt;
341 // explicitly multiply with the parameter transform to generate parameter space
342 // gradients
343 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
344 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
345 ++index) {
346 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
347 val * this->parameterTransform_.transform.valuePtr()[index];
348 }
349 }
350
351 // go to the next joint
352 jointIndex = this->skeleton_.joints[jointIndex].parent;
353 }
354
355 // -----------------------------------
356 // process second joint
357 // -----------------------------------
358 jointIndex = collisionGeometry_[indexB].parent;
359 while (jointIndex != kInvalidIndex && jointIndex != commonAncestor) {
360 const auto& jointState = state.jointState[jointIndex];
361 const size_t paramIndex = jointIndex * kParametersPerJoint;
362 const Vector3<T> posd = position_j - jointState.translation();
363
364 // calculate derivatives based on active joints
365 for (size_t d = 0; d < 3; d++) {
366 if (this->activeJointParams_[paramIndex + d]) {
367 // calculate joint gradient
368 const T val = direction.dot(jointState.getTranslationDerivative(d)) * -wgt;
369 // explicitly multiply with the parameter transform to generate parameter space
370 // gradients
371 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
372 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
373 ++index) {
374 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
375 val * this->parameterTransform_.transform.valuePtr()[index];
376 }
377 }
378 if (this->activeJointParams_[paramIndex + 3 + d]) {
379 // calculate joint gradient
380 const T val = direction.dot(jointState.getRotationDerivative(d, posd)) * -wgt;
381 // explicitly multiply with the parameter transform to generate parameter space
382 // gradients
383 const auto maxIndex =
384 this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
385 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
386 index < maxIndex;
387 ++index) {
388 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
389 val * this->parameterTransform_.transform.valuePtr()[index];
390 }
391 }
392 }
393 if (this->activeJointParams_[paramIndex + 6]) {
394 // calculate joint gradient
395 const T val = direction.dot(jointState.getScaleDerivative(posd)) * -wgt;
396 // explicitly multiply with the parameter transform to generate parameter space
397 // gradients
398 const auto maxIndex =
399 this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
400 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
401 index < maxIndex;
402 ++index) {
403 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
404 val * this->parameterTransform_.transform.valuePtr()[index];
405 }
406 }
407
408 // go to the next joint
409 jointIndex = this->skeleton_.joints[jointIndex].parent;
410 }
411
412 return true;
413 });
414
415 return error;
416}

418template <typename T>
419double CollisionErrorFunctionT<T>::getJacobian(
420 const ModelParametersT<T>& /* params */,
421 const SkeletonStateT<T>& state,
422 const MeshStateT<T>& /* meshState */,
423 Ref<MatrixX<T>> jacobian,
424 Ref<VectorX<T>> residual,
425 int& usedRows) {
426 MT_PROFILE_FUNCTION();
427 MT_CHECK(
428 jacobian.rows() >= gsl::narrow<Eigen::Index>(getJacobianSize()),
429 "Jacobian rows mismatch: Actual {}, Expected {}",
430 jacobian.rows(),
431 gsl::narrow<Eigen::Index>(getJacobianSize()));
432
433 MT_CHECK(
434 residual.rows() >= gsl::narrow<Eigen::Index>(getJacobianSize()),
435 "Residual rows mismatch: Actual {}, Expected {}",
436 residual.rows(),
437 gsl::narrow<Eigen::Index>(getJacobianSize()));
438
439 if (state.jointState.empty()) {
440 return 0.0;
441 }
442
443 // check all is valid
444 MT_CHECK(
445 state.jointParameters.size() ==
446 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
447
448 computeBroadPhase(state);
449
450 const T wgt = std::sqrt(kCollisionWeight * this->weight_);
451
452 int pos = 0;
453 double error = 0;
454
455 // loop over all pairs of collision objects and check
456 bvh_.traverseOverlappingPairs([&](auto indexA, auto indexB) -> bool {
457 const std::pair<size_t, size_t> pairId = std::make_pair(indexA, indexB);
458
459 // Skip excluded pairs
460 if (excludingPairIds_.find(pairId) != excludingPairIds_.end()) {
461 return true; // continue traversing
462 }
463
464 // Perform narrow-phase collision check
465 T distance;
466 Vector2<T> cp;
467 T overlap;
468 if (!overlaps(
469 collisionState_.origin[indexA],
470 collisionState_.direction[indexA],
471 collisionState_.radius[indexA],
472 collisionState_.delta[indexA],
473 collisionState_.origin[indexB],
474 collisionState_.direction[indexB],
475 collisionState_.radius[indexB],
476 collisionState_.delta[indexB],
477 distance,
478 cp,
479 overlap)) {
480 return true; // continue traversing
481 }
482
483 const auto& jointA = collisionGeometry_[indexA].parent;
484 const auto& jointB = collisionGeometry_[indexB].parent;
485 const size_t commonAncestor = this->skeleton_.commonAncestor(jointA, jointB);
486 error += sqr(overlap) * kCollisionWeight * this->weight_;
487
488 // calculate collision resolve direction. this is what we need to push joint parent i in.
489 // the direction for joint parent j is the inverse
490 const Vector3<T> position_i =
491 (collisionState_.origin[indexA] + collisionState_.direction[indexA] * cp[0]);
492 const Vector3<T> position_j =
493 (collisionState_.origin[indexB] + collisionState_.direction[indexB] * cp[1]);
494 const Vector3<T> direction = position_i - position_j;
495 const T inverseDistance = T(1) / distance;
496
497 // calculate constant factor
498 const T fac = T(2) * inverseDistance * wgt;
499
500 // get position in jacobian
501 const int row = pos++;
502
503 // -----------------------------------
504 // process first joint
505 // -----------------------------------
506 size_t jointIndex = collisionGeometry_[indexA].parent;
507 while (jointIndex != kInvalidIndex && jointIndex != commonAncestor) {
508 const auto& jointState = state.jointState[jointIndex];
509 const size_t paramIndex = jointIndex * kParametersPerJoint;
510 const Vector3<T> posd = position_i - jointState.translation();
511
512 // calculate derivatives based on active joints
513 for (size_t d = 0; d < 3; d++) {
514 if (this->activeJointParams_[paramIndex + d]) {
515 // calculate joint gradient
516 const T val = direction.dot(jointState.getTranslationDerivative(d)) * -fac;
517 // explicitly multiply with the parameter transform to generate parameter space
518 // gradients
519 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
520 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
521 ++index) {
522 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
523 val * this->parameterTransform_.transform.valuePtr()[index];
524 }
525 }
526 if (this->activeJointParams_[paramIndex + 3 + d]) {
527 // calculate joint gradient
528 const T val = direction.dot(jointState.getRotationDerivative(d, posd)) * -fac;
529 // explicitly multiply with the parameter transform to generate parameter space
530 // gradients
531 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
532 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
533 ++index) {
534 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
535 val * this->parameterTransform_.transform.valuePtr()[index];
536 }
537 }
538 }
539 if (this->activeJointParams_[paramIndex + 6]) {
540 // calculate joint gradient
541 const T val = direction.dot(jointState.getScaleDerivative(posd)) * -fac;
542 // explicitly multiply with the parameter transform to generate parameter space
543 // gradients
544 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
545 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
546 ++index) {
547 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
548 val * this->parameterTransform_.transform.valuePtr()[index];
549 }
550 }
551
552 // go to the next joint
553 jointIndex = this->skeleton_.joints[jointIndex].parent;
554 }
555
556 // -----------------------------------
557 // process second joint
558 // -----------------------------------
559 jointIndex = collisionGeometry_[indexB].parent;
560 while (jointIndex != kInvalidIndex && jointIndex != commonAncestor) {
561 const auto& jointState = state.jointState[jointIndex];
562 const size_t paramIndex = jointIndex * kParametersPerJoint;
563 const Vector3<T> posd = position_j - jointState.translation();
564
565 // calculate derivatives based on active joints
566 for (size_t d = 0; d < 3; d++) {
567 if (this->activeJointParams_[paramIndex + d]) {
568 // calculate joint gradient
569 const T val = direction.dot(jointState.getTranslationDerivative(d)) * fac;
570 // explicitly multiply with the parameter transform to generate parameter space
571 // gradients
572 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
573 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
574 ++index) {
575 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
576 val * this->parameterTransform_.transform.valuePtr()[index];
577 }
578 }
579 if (this->activeJointParams_[paramIndex + 3 + d]) {
580 // calculate joint gradient
581 const T val = direction.dot(jointState.getRotationDerivative(d, posd)) * fac;
582 // explicitly multiply with the parameter transform to generate parameter space
583 // gradients
584 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
585 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
586 ++index) {
587 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
588 val * this->parameterTransform_.transform.valuePtr()[index];
589 }
590 }
591 }
592 if (this->activeJointParams_[paramIndex + 6]) {
593 // calculate joint gradient
594 const T val = direction.dot(jointState.getScaleDerivative(posd)) * fac;
595 // explicitly multiply with the parameter transform to generate parameter space
596 // gradients
597 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
598 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
599 ++index) {
600 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
601 val * this->parameterTransform_.transform.valuePtr()[index];
602 }
603 }
604
605 // go to the next joint
606 jointIndex = this->skeleton_.joints[jointIndex].parent;
607 }
608
609 residual(row) = overlap * wgt;
610
611 return true;
612 });
613
614 usedRows = pos;
615
616 return error;
617}
