19void parameterSetsToJson(const Character& character, nlohmann::json& j) {
20 j = nlohmann::json::object();
21 for (const auto& set : character.parameterTransform.parameterSets) {
22 j[set.first] = nlohmann::json::array();
23 for (size_t i = 0; i < character.parameterTransform.name.size(); i++) {
24 if (set.second.test(i)) {
25 j[set.first].push_back(character.parameterTransform.name[i]);
26 }
27 }
28 }
29}

31ParameterSets parameterSetsFromJson(const Character& character, const nlohmann::json& j) {
32 ParameterSets result;
33
34 for (const auto& parameterSet : j.items()) {
35 auto& set = result[parameterSet.key()];
36 for (const auto& parameter : parameterSet.value().items()) {
37 const auto parameterIndex =
38 character.parameterTransform.getParameterIdByName(parameter.value());
39
40 if (parameterIndex != kInvalidIndex) {
41 set.set(parameterIndex);
42 }
43 }
44 }
45
46 return result;
47}

61PoseConstraints poseConstraintsFromJson(const Character& character, const nlohmann::json& j) {
62 PoseConstraints result;
63
64 for (const auto& constraint : j.items()) {
65 auto& pose = result[constraint.key()];
66 for (const auto& constraintPair : constraint.value().items()) {
67 const auto parameterIndex =
68 character.parameterTransform.getParameterIdByName(constraintPair.key());
69
70 const float value = constraintPair.value();
71 if (parameterIndex != kInvalidIndex) {
72 pose.parameterIdValue.emplace_back(parameterIndex, value);
73 }
74 }
75 }
76
77 return result;

80void parameterTransformToJson(const Character& character, nlohmann::json& j) {
81 j["parameters"] = character.parameterTransform.name;
82
83 for (size_t jt = 0; jt < character.skeleton.joints.size(); jt++) {
84 const auto& jointName = character.skeleton.joints[jt].name;
85 const size_t paramIndex = jt * kParametersPerJoint;
86 for (size_t d = 0; d < kParametersPerJoint; d++) {
87 for (auto index = character.parameterTransform.transform.outerIndexPtr()[paramIndex + d];
88 index < character.parameterTransform.transform.outerIndexPtr()[paramIndex + d + 1];
89 ++index) {
90 const auto& parameterIndex = character.parameterTransform.transform.innerIndexPtr()[index];
91 const auto& parameterValue = character.parameterTransform.transform.valuePtr()[index];
92
93 if (parameterIndex < character.parameterTransform.name.size()) {
94 j["joints"][jointName][kJointParameterNames[d]]
95 [character.parameterTransform.name[parameterIndex]] = parameterValue;
96 }
97 }
98 }
99 }
100 // write blendshape parameters
101 const auto blendShapeParameters = character.parameterTransform.blendShapeParameters;
102 const auto faceExpressionParameters = character.parameterTransform.faceExpressionParameters;
103 auto bs = nlohmann::json::array();
104 for (int i = 0; i < blendShapeParameters.size(); i++) {
105 bs.push_back(blendShapeParameters[i]);
106 }
107 auto fe = nlohmann::json::array();
108 for (int i = 0; i < faceExpressionParameters.size(); i++) {
109 fe.push_back(faceExpressionParameters[i]);
110 }
111 j["blendShapeParameters"] = bs;
112 j["faceExpressionParameters"] = fe;
113}

115ParameterTransform parameterTransformFromJson(const Character& character, const nlohmann::json& j) {
116 ParameterTransform pt;
117 try {
118 pt.activeJointParams =
119 VectorX<bool>::Constant(character.skeleton.joints.size() * kParametersPerJoint, false);
120 pt.offsets = VectorXf::Zero(character.skeleton.joints.size() * kParametersPerJoint);
121
122 // load all the parameter names
123 auto parameterItr = j.find("parameters");
124 MT_THROW_IF(parameterItr == j.end(), "No 'parameters' found in parameter transform.");
125
126 for (const auto& p : *parameterItr) {
127 pt.name.push_back(p.get<std::string>());
128 }
129
130 // triplet list
131 std::vector<Eigen::Triplet<float>> triplets;
132
133 auto jointItr = j.find("joints");
134 MT_THROW_IF(jointItr == j.end(), "No 'joints' found in parameter transform.");
135
136 for (const auto& joint : jointItr->items()) {
137 const std::string& jointName = joint.key();
138
139 // get the right joint to modify
140 const size_t jointIndex = character.skeleton.getJointIdByName(jointName);
141 MT_THROW_IF(jointIndex == kInvalidIndex, "Unknown joint name in expression : {}", jointName);
142
143 for (const auto& jointParameter : joint.value().items()) {
144 const std::string& jointParameterName = jointParameter.key();
145
146 // the first pToken is the name of the joint and it's attribute type
147 size_t attributeIndex = kInvalidIndex;
148 for (size_t jt = 0; jt < kParametersPerJoint; jt++) {
149 if (jointParameterName == kJointParameterNames[jt]) {
150 attributeIndex = jt;
151 break;
152 }
153 }
154
155 // if we didn't find a right name exit with an error
156 MT_THROW_IF(
157 attributeIndex == kInvalidIndex,
158 "Unknown channel name in expression : {}",
159 jointParameterName);
160
161 // enable the attribute in the skeleton if we have a parameter controlling it
162 pt.activeJointParams[jointIndex * kParametersPerJoint + attributeIndex] = 1;
163
164 // split the parameter names
165 for (const auto& parameter : jointParameter.value().items()) {
166 // get weight and name
167 const float weight = parameter.value();
168 const std::string& parameterName = parameter.key();
169
170 auto parameterIndex = pt.getParameterIdByName(parameterName);
171
172 MT_THROW_IF(
173 parameterIndex == kInvalidIndex,
174 "Unknown parameter name in expression : {}",
175 parameterName);
176
177 // add triplet
178 triplets.emplace_back(
179 static_cast<int>(jointIndex) * kParametersPerJoint + static_cast<int>(attributeIndex),
180 static_cast<int>(parameterIndex),
181 weight);
182 }
183 }
184 }
185
186 // resize the Transform matrix to the correct size
187 pt.transform.resize(
188 static_cast<int>(character.skeleton.joints.size()) * kParametersPerJoint,
189 static_cast<int>(pt.name.size()));
190
191 // finish parameter setup by creating sparse matrix
192 pt.transform.setFromTriplets(triplets.begin(), triplets.end());
193 } catch (...) {
194 // create an identity parameter transform
195 pt = ParameterTransform::identity(character.skeleton.getJointNames());
196 }
197
198 // read blendshape parameters
199 auto blendShapeParameters = j.find("blendShapeParameters");
200 if (blendShapeParameters != j.end()) {
201 pt.blendShapeParameters.resize(blendShapeParameters->size());
202 for (int i = 0; i < blendShapeParameters->size(); i++) {
203 pt.blendShapeParameters[i] = (*blendShapeParameters)[i].get<int>();
204 }
205 } else {
206 pt.blendShapeParameters.resize(0);
207 }
208 auto faceExpressionParameters = j.find("faceExpressionParameters");
209 if (faceExpressionParameters != j.end()) {
210 pt.faceExpressionParameters.resize(faceExpressionParameters->size());
211 for (int i = 0; i < faceExpressionParameters->size(); i++) {
212 pt.faceExpressionParameters[i] = (*faceExpressionParameters)[i].get<int>();
213 }
214 } else {
215 pt.faceExpressionParameters.resize(0);
216 }
217
218 return pt;
219}

221void parameterLimitsToJson(const Character& character, nlohmann::json& j) {
222 j = nlohmann::json::array();
223 for (const auto& lim : character.parameterLimits) {
224 j.push_back(nlohmann::json::object());
225 auto& li = j.back();
226 li["weight"] = lim.weight;
227
228 switch (lim.type) {
229 case MinMax:
230 li["type"] = "minmax";
231 li["parameter"] = character.parameterTransform.name[lim.data.minMax.parameterIndex];
232 li["limits"] = lim.data.minMax.limits;
233 break;
234 case MinMaxJoint:
235 li["type"] = "minmax_joint";
236 if (!character.skeleton.joints.empty() &&
237 lim.data.minMaxJoint.jointIndex < character.skeleton.joints.size()) {
238 li["jointIndex"] = character.skeleton.joints[lim.data.minMaxJoint.jointIndex].name;
239 }
240 li["jointParameter"] = kJointParameterNames[lim.data.minMaxJoint.jointParameter];
241 li["limits"] = lim.data.minMaxJoint.limits;
242 break;
243 case MinMaxJointPassive:
244 li["type"] = "minmax_joint_passive";
245 if (!character.skeleton.joints.empty() &&
246 lim.data.minMaxJoint.jointIndex < character.skeleton.joints.size()) {
247 li["jointIndex"] = character.skeleton.joints[lim.data.minMaxJoint.jointIndex].name;
248 }
249 li["jointParameter"] = kJointParameterNames[lim.data.minMaxJoint.jointParameter];
250 li["limits"] = lim.data.minMaxJoint.limits;
251 break;
252 case Linear:
253 li["type"] = "linear";
254 li["referenceParameter"] =
255 character.parameterTransform.name[lim.data.linear.referenceIndex];
256 li["targetParameter"] = character.parameterTransform.name[lim.data.linear.targetIndex];
257 li["scale"] = lim.data.linear.scale;
258 li["offset"] = lim.data.linear.offset;
259 if (lim.data.linear.rangeMin != -std::numeric_limits<float>::max()) {
260 li["rangeMin"] = lim.data.linear.rangeMin;
261 }
262 if (lim.data.linear.rangeMax != std::numeric_limits<float>::max()) {
263 li["rangeMax"] = lim.data.linear.rangeMax;
264 }
265 break;
266 case LinearJoint:
267 li["type"] = "linear_joint";
268 if (!character.skeleton.joints.empty() &&
269 lim.data.linearJoint.referenceJointIndex < character.skeleton.joints.size()) {
270 li["referenceJoint"] =
271 character.skeleton.joints[lim.data.linearJoint.referenceJointIndex].name;
272 }
273 li["referenceJointParameter"] = lim.data.linearJoint.referenceJointParameter;
274 if (!character.skeleton.joints.empty() &&
275 lim.data.linearJoint.targetJointIndex < character.skeleton.joints.size()) {
276 li["targetJoint"] = character.skeleton.joints[lim.data.linearJoint.targetJointIndex].name;
277 }
278 li["targetJointParameter"] = lim.data.linearJoint.targetJointParameter;
279 li["scale"] = lim.data.linearJoint.scale;
280 li["offset"] = lim.data.linearJoint.offset;
281 if (lim.data.linearJoint.rangeMin != -std::numeric_limits<float>::max()) {
282 li["rangeMin"] = lim.data.linearJoint.rangeMin;
283 }
284 if (lim.data.linearJoint.rangeMax != std::numeric_limits<float>::max()) {
285 li["rangeMax"] = lim.data.linearJoint.rangeMax;
286 }
287 break;
288 case HalfPlane:
289 li["type"] = "half_plane";
290 li["param1"] = character.parameterTransform.name[lim.data.halfPlane.param1];
291 li["param2"] = character.parameterTransform.name[lim.data.halfPlane.param2];
292 li["normal"] = lim.data.halfPlane.normal;
293 li["offset"] = lim.data.halfPlane.offset;
294 break;
295 case Ellipsoid: {
296 li["type"] = "ellipsoid";
297 if (!character.skeleton.joints.empty() &&
298 lim.data.ellipsoid.parent < character.skeleton.joints.size()) {
299 li["parent"] = character.skeleton.joints[lim.data.ellipsoid.parent].name;
300 }
301 if (!character.skeleton.joints.empty() &&
302 lim.data.ellipsoid.ellipsoidParent < character.skeleton.joints.size()) {
303 li["ellipsoidParent"] =
304 character.skeleton.joints[lim.data.ellipsoid.ellipsoidParent].name;
305 }
306 li["offset"] = lim.data.ellipsoid.offset * toM();
307 auto eli = lim.data.ellipsoid.ellipsoid;
308 eli.translation() *= toM();
309 toJson(eli.matrix(), li["ellipsoid"]);
310 break;
311 }
312 default: {
313 MT_LOGE(
314 "Unknown parameter limit type '{}' from character name '{}'",
315 toString(lim.type),
316 character.name);
317 break;
318 }
319 }
320 }

323ParameterLimits parameterLimitsFromJson(const Character& character, const nlohmann::json& j) {
324 ParameterLimits result;
325
326 for (const auto& element : j) {
327 const std::string type = element.value("type", "");
328 ParameterLimit l;
329 l.weight = element.value("weight", 0.0f);
330 if (type == "minmax") {
331 l.type = MinMax;
332 l.data.minMax.parameterIndex =
333 character.parameterTransform.getParameterIdByName(element.value("parameter", ""));
334 l.data.minMax.limits = fromJson<Vector2f>(element["limits"]);
335 } else if (type == "minmax_joint") {
336 l.type = MinMaxJoint;
337 l.data.minMaxJoint.jointIndex =
338 character.skeleton.getJointIdByName(element.value("jointIndex", ""));
339 const std::string attribute = element.value("jointParameter", "");
340 size_t attributeIndex = kInvalidIndex;
341 for (size_t t = 0; t < kParametersPerJoint; t++) {
342 if (attribute == kJointParameterNames[t]) {
343 attributeIndex = t;
344 break;
345 }
346 }
347 l.data.minMaxJoint.jointParameter = attributeIndex;
348 l.data.minMaxJoint.limits = fromJson<Vector2f>(element["limits"]);
349 } else if (type == "minmax_joint_passive") {
350 l.type = MinMaxJointPassive;
351 l.data.minMaxJoint.jointIndex =
352 character.skeleton.getJointIdByName(element.value("jointIndex", ""));
353 const std::string attribute = element.value("jointParameter", "");
354 size_t attributeIndex = kInvalidIndex;
355 for (size_t t = 0; t < kParametersPerJoint; t++) {
356 if (attribute == kJointParameterNames[t]) {
357 attributeIndex = t;
358 break;
359 }
360 }
361 l.data.minMaxJoint.jointParameter = attributeIndex;
362 l.data.minMaxJoint.limits = fromJson<Vector2f>(element["limits"]);
363 } else if (type == "linear") {
364 l.type = Linear;
365 l.data.linear.referenceIndex = character.parameterTransform.getParameterIdByName(
366 element.value("referenceParameter", ""));
367 l.data.linear.targetIndex =
368 character.parameterTransform.getParameterIdByName(element.value("targetParameter", ""));
369 l.data.linear.scale = element["scale"];
370 l.data.linear.offset = element["offset"];
371 l.data.linear.rangeMin = element.value("rangeMin", -std::numeric_limits<float>::max());
372 l.data.linear.rangeMax = element.value("rangeMax", std::numeric_limits<float>::max());
373 } else if (type == "linear_joint") {
374 l.type = LinearJoint;
375 l.data.linearJoint.referenceJointIndex =
376 character.skeleton.getJointIdByName(element.value("referenceJoint", ""));
377 l.data.linearJoint.targetJointIndex =
378 character.skeleton.getJointIdByName(element.value("targetJoint", ""));
379 l.data.linearJoint.referenceJointParameter = element["referenceJointParameter"].get<size_t>();
380 l.data.linearJoint.targetJointParameter = element["targetJointParameter"].get<size_t>();
381 l.data.linearJoint.scale = element["scale"];
382 l.data.linearJoint.offset = element["offset"];
383 l.data.linearJoint.rangeMin = element.value("rangeMin", -std::numeric_limits<float>::max());
384 l.data.linearJoint.rangeMax = element.value("rangeMax", std::numeric_limits<float>::max());
385 } else if (type == "half_plane") {
386 l.type = HalfPlane;
387 l.data.halfPlane.param1 =
388 character.parameterTransform.getParameterIdByName(element.value("param1", ""));
389 l.data.halfPlane.param2 =
390 character.parameterTransform.getParameterIdByName(element.value("param2", ""));
391 l.data.halfPlane.normal = fromJson<Vector2f>(element["normal"]);
392 l.data.halfPlane.offset = element["offset"];
393 } else if (type == "ellipsoid") {
394 l.type = Ellipsoid;
395 l.data.ellipsoid.parent = character.skeleton.getJointIdByName(element.value("parent", ""));
396 l.data.ellipsoid.ellipsoidParent =
397 character.skeleton.getJointIdByName(element.value("ellipsoidParent", ""));
398 l.data.ellipsoid.offset = fromJson<Vector3f>(element["offset"]);
399 l.data.ellipsoid.offset /= toM();
400 l.data.ellipsoid.ellipsoid.matrix() = fromJson<Matrix4f>(element["ellipsoid"]);
401 l.data.ellipsoid.ellipsoid.translation() /= toM();
402 l.data.ellipsoid.ellipsoidInv = l.data.ellipsoid.ellipsoid.inverse();
403 } else if (type == "elipsoid") {
404 // TODO: Remove once all the model files are migrated to ellipsoid
405 MT_LOGW_ONCE(
406 "Deprecated parameter limit type: {} (typo). Please use 'ellipsoid' instead.", type);
407 l.type = Ellipsoid;
408 l.data.ellipsoid.parent = character.skeleton.getJointIdByName(element.value("parent", ""));
409 l.data.ellipsoid.ellipsoidParent =
410 character.skeleton.getJointIdByName(element.value("elipsoidParent", ""));
411 l.data.ellipsoid.offset = fromJson<Vector3f>(element["offset"]);
412 l.data.ellipsoid.offset /= toM();
413 l.data.ellipsoid.ellipsoid.matrix() = fromJson<Matrix4f>(element["elipsoid"]);
414 l.data.ellipsoid.ellipsoid.translation() /= toM();
415 l.data.ellipsoid.ellipsoidInv = l.data.ellipsoid.ellipsoid.inverse();
416 } else {
417 MT_THROW("Unknown parameter limit type '{}' from character name '{}'.", type, character.name);
418 }
419 result.push_back(l);
420 }
421
422 return result;
423}
