40template <typename T>
41void CollisionErrorFunctionStatelessT<T>::updateCollisionPairs() {
42 // clear collisions
43 collisionPairs.clear();
44
45 // create zero state
46 const SkeletonStateT<T> state(
47 this->parameterTransform_.zero().template cast<T>(), this->skeleton_);
48 collisionState.update(state, collisionGeometry);
49
50 // go over all possible pairs
51 for (size_t i = 0; i < collisionGeometry.size(); ++i) {
52 for (size_t j = i + 1; j < collisionGeometry.size(); ++j) {
53 // check if pair naturally intersects
54 T distance;
55 Vector2<T> cp;
56 T overlap;
57 if (overlaps(
58 collisionState.origin[i],
59 collisionState.direction[i],
60 collisionState.radius[i],
61 collisionState.delta[i],
62 collisionState.origin[j],
63 collisionState.direction[j],
64 collisionState.radius[j],
65 collisionState.delta[j],
66 distance,
67 cp,
68 overlap)) {
69 continue;
70 }
71
72 // no, it doesn't. check
73
74 // walk down the this->skeleton_ hierarchy for the collision pair until we find a common
75 // ancestor
76 size_t p0 = collisionGeometry[i].parent;
77 size_t p1 = collisionGeometry[j].parent;
78 size_t count = 0;
79
80 // Find the common ancestor. We can always just walk up the higher joint index because the
81 // parent always has to have a lower index than the child joint.
82 while (p0 != p1) {
83 if (p0 > p1) {
84 p0 = this->skeleton_.joints[p0].parent;
85 } else {
86 p1 = this->skeleton_.joints[p1].parent;
87 }
88 count++;
89 }
90
91 const Vector3i collisionPair(
92 gsl::narrow_cast<int>(i), gsl::narrow_cast<int>(j), gsl::narrow_cast<int>(p0));
93
94 // Store collision pairs excluding those with the same parent or adjacent joints
95 if (count > 1) {
96 collisionPairs.push_back(collisionPair);
97 }
98 }
99 }
100
101 collisionActive.setConstant(collisionPairs.size(), false);
102}

104template <typename T>
105std::vector<Vector2i> CollisionErrorFunctionStatelessT<T>::getCollisionPairs() const {
106 std::vector<Vector2i> result;
107
108 for (size_t i = 0; i < collisionPairs.size(); i++) {
109 if (collisionActive[i]) {
110 result.emplace_back(collisionPairs[i].template head<2>());
111 }
112 }
113
114 return result;

117template <typename T>
118double CollisionErrorFunctionStatelessT<T>::getError(
119 const ModelParametersT<T>& /* params */,
120 const SkeletonStateT<T>& state,
121 const MeshStateT<T>& /* meshState */) {
122 if (state.jointState.empty()) {
123 return 0.0;
124 }
125
126 // check all is valid
127 MT_CHECK(
128 state.jointParameters.size() ==
129 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
130
131 // update collision state
132 collisionState.update(state, collisionGeometry);
133 collisionActive.setConstant(false);
134
135 // loop over all pairs of collision objects and check
136 double error = 0.0;
137
138 for (size_t i = 0; i < collisionPairs.size(); i++) {
139 const auto& pr = collisionPairs[i];
140 T distance;
141 Vector2<T> cp;
142 T overlap;
143 if (!overlaps(
144 collisionState.origin[pr[0]],
145 collisionState.direction[pr[0]],
146 collisionState.radius[pr[0]],
147 collisionState.delta[pr[0]],
148 collisionState.origin[pr[1]],
149 collisionState.direction[pr[1]],
150 collisionState.radius[pr[1]],
151 collisionState.delta[pr[1]],
152 distance,
153 cp,
154 overlap)) {
155 continue;
156 }
157
158 collisionActive[i] = true;
159 error += sqr(overlap) * kCollisionWeight * this->weight_;
160 }
161
162 // return error
163 return error;
164}

166template <typename T>
167double CollisionErrorFunctionStatelessT<T>::getGradient(
168 const ModelParametersT<T>& /* params */,
169 const SkeletonStateT<T>& state,
170 const MeshStateT<T>& /* meshState */,
171 Ref<VectorX<T>> gradient) {
172 if (state.jointState.empty()) {
173 return 0.0;
174 }
175
176 // check all is valid
177 MT_CHECK(
178 state.jointParameters.size() ==
179 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
180
181 // update collision state
182 collisionState.update(state, collisionGeometry);
183 collisionActive.setConstant(false);
184
185 double error = 0;
186 for (size_t i = 0; i < collisionPairs.size(); ++i) {
187 const auto& pr = collisionPairs[i];
188 T distance;
189 Vector2<T> cp;
190 T overlap;
191 if (!overlaps(
192 collisionState.origin[pr[0]],
193 collisionState.direction[pr[0]],
194 collisionState.radius[pr[0]],
195 collisionState.delta[pr[0]],
196 collisionState.origin[pr[1]],
197 collisionState.direction[pr[1]],
198 collisionState.radius[pr[1]],
199 collisionState.delta[pr[1]],
200 distance,
201 cp,
202 overlap)) {
203 continue;
204 }
205
206 collisionActive[i] = true;
207 error += sqr(overlap) * kCollisionWeight * this->weight_;
208
209 // calculate collision resolve direction. this is what we need to push joint parent i in.
210 // the direction for joint parent j is the inverse
211 const Vector3<T> position_i =
212 (collisionState.origin[pr[0]] + collisionState.direction[pr[0]] * cp[0]);
213 const Vector3<T> position_j =
214 (collisionState.origin[pr[1]] + collisionState.direction[pr[1]] * cp[1]);
215 const Vector3<T> direction = position_i - position_j;
216 const T overlapFraction = overlap / distance;
217
218 // calculate weight
219 const T wgt = -T(2) * kCollisionWeight * this->weight_ * overlapFraction;
220
221 // -----------------------------------
222 // process first joint
223 // -----------------------------------
224 size_t jointIndex = collisionGeometry[pr[0]].parent;
225 while (jointIndex != kInvalidIndex && jointIndex != gsl::narrow_cast<size_t>(pr[2])) {
226 const auto& jointState = state.jointState[jointIndex];
227 const size_t paramIndex = jointIndex * kParametersPerJoint;
228 const Vector3<T> posd = position_i - jointState.translation();
229
230 // calculate derivatives based on active joints
231 for (size_t d = 0; d < 3; d++) {
232 if (this->activeJointParams_[paramIndex + d]) {
233 // calculate joint gradient
234 const T val = direction.dot(jointState.getTranslationDerivative(d)) * wgt;
235 // explicitly multiply with the parameter transform to generate parameter space
236 // gradients
237 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
238 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
239 ++index) {
240 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
241 val * this->parameterTransform_.transform.valuePtr()[index];
242 }
243 }
244 if (this->activeJointParams_[paramIndex + 3 + d]) {
245 // calculate joint gradient
246 const T val = direction.dot(jointState.getRotationDerivative(d, posd)) * wgt;
247 // explicitly multiply with the parameter transform to generate parameter space
248 // gradients
249 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
250 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
251 ++index) {
252 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
253 val * this->parameterTransform_.transform.valuePtr()[index];
254 }
255 }
256 }
257 if (this->activeJointParams_[paramIndex + 6]) {
258 // calculate joint gradient
259 const T val = direction.dot(jointState.getScaleDerivative(posd)) * wgt;
260 // explicitly multiply with the parameter transform to generate parameter space
261 // gradients
262 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
263 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
264 ++index) {
265 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
266 val * this->parameterTransform_.transform.valuePtr()[index];
267 }
268 }
269
270 // go to the next joint
271 jointIndex = this->skeleton_.joints[jointIndex].parent;
272 }
273
274 // -----------------------------------
275 // process second joint
276 // -----------------------------------
277 jointIndex = collisionGeometry[pr[1]].parent;
278 while (jointIndex != kInvalidIndex && jointIndex != gsl::narrow_cast<size_t>(pr[2])) {
279 const auto& jointState = state.jointState[jointIndex];
280 const size_t paramIndex = jointIndex * kParametersPerJoint;
281 const Vector3<T> posd = position_j - jointState.translation();
282
283 // calculate derivatives based on active joints
284 for (size_t d = 0; d < 3; d++) {
285 if (this->activeJointParams_[paramIndex + d]) {
286 // calculate joint gradient
287 const T val = direction.dot(jointState.getTranslationDerivative(d)) * -wgt;
288 // explicitly multiply with the parameter transform to generate parameter space
289 // gradients
290 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
291 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
292 ++index) {
293 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
294 val * this->parameterTransform_.transform.valuePtr()[index];
295 }
296 }
297 if (this->activeJointParams_[paramIndex + 3 + d]) {
298 // calculate joint gradient
299 const T val = direction.dot(jointState.getRotationDerivative(d, posd)) * -wgt;
300 // explicitly multiply with the parameter transform to generate parameter space
301 // gradients
302 const auto maxIndex =
303 this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
304 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
305 index < maxIndex;
306 ++index) {
307 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
308 val * this->parameterTransform_.transform.valuePtr()[index];
309 }
310 }
311 }
312 if (this->activeJointParams_[paramIndex + 6]) {
313 // calculate joint gradient
314 const T val = direction.dot(jointState.getScaleDerivative(posd)) * -wgt;
315 // explicitly multiply with the parameter transform to generate parameter space
316 // gradients
317 const auto maxIndex =
318 this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
319 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
320 index < maxIndex;
321 ++index) {
322 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
323 val * this->parameterTransform_.transform.valuePtr()[index];
324 }
325 }
326
327 // go to the next joint
328 jointIndex = this->skeleton_.joints[jointIndex].parent;
329 }
330 }
331
332 // return error
333 return error;
334}

336template <typename T>
337double CollisionErrorFunctionStatelessT<T>::getJacobian(
338 const ModelParametersT<T>& /* params */,
339 const SkeletonStateT<T>& state,
340 const MeshStateT<T>& /* meshState */,
341 Ref<MatrixX<T>> jacobian,
342 Ref<VectorX<T>> residual,
343 int& usedRows) {
344 MT_PROFILE_FUNCTION();
345 MT_CHECK(
346 jacobian.rows() >= gsl::narrow<Eigen::Index>(collisionPairs.size()),
347 "Jacobian rows mismatch: Actual {}, Expected {}",
348 jacobian.rows(),
349 gsl::narrow<Eigen::Index>(collisionPairs.size()));
350
351 MT_CHECK(
352 residual.rows() >= gsl::narrow<Eigen::Index>(collisionPairs.size()),
353 "Residual rows mismatch: Actual {}, Expected {}",
354 residual.rows(),
355 gsl::narrow<Eigen::Index>(collisionPairs.size()));
356
357 if (state.jointState.empty()) {
358 return 0.0;
359 }
360
361 // check all is valid
362 MT_CHECK(
363 state.jointParameters.size() ==
364 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
365
366 // update collision state
367 {
368 MT_PROFILE_EVENT("Collision: updateState");
369
370 collisionState.update(state, collisionGeometry);
371 collisionActive.setConstant(false);
372 }
373
374 const T wgt = std::sqrt(kCollisionWeight * this->weight_);
375
376 int pos = 0;
377 double error = 0;
378
379 // loop over all pairs of collision objects and check
380 for (size_t i = 0; i < collisionPairs.size(); ++i) {
381 const auto& pr = collisionPairs[i];
382 T distance;
383 Vector2<T> cp;
384 T overlap;
385 if (!overlaps(
386 collisionState.origin[pr[0]],
387 collisionState.direction[pr[0]],
388 collisionState.radius[pr[0]],
389 collisionState.delta[pr[0]],
390 collisionState.origin[pr[1]],
391 collisionState.direction[pr[1]],
392 collisionState.radius[pr[1]],
393 collisionState.delta[pr[1]],
394 distance,
395 cp,
396 overlap)) {
397 continue;
398 }
399
400 collisionActive[i] = true;
401 error += sqr(overlap) * kCollisionWeight * this->weight_;
402
403 // calculate collision resolve direction. this is what we need to push joint parent i in.
404 // the direction for joint parent j is the inverse
405 const Vector3<T> position_i =
406 (collisionState.origin[pr[0]] + collisionState.direction[pr[0]] * cp[0]);
407 const Vector3<T> position_j =
408 (collisionState.origin[pr[1]] + collisionState.direction[pr[1]] * cp[1]);
409 const Vector3<T> direction = position_i - position_j;
410 const T inverseDistance = T(1) / distance;
411
412 // calculate constant factor
413 const T fac = T(2) * inverseDistance * wgt;
414
415 // get position in jacobian
416 const int row = pos++;
417
418 // -----------------------------------
419 // process first joint
420 // -----------------------------------
421 size_t jointIndex = collisionGeometry[pr[0]].parent;
422 while (jointIndex != kInvalidIndex && jointIndex != gsl::narrow_cast<size_t>(pr[2])) {
423 const auto& jointState = state.jointState[jointIndex];
424 const size_t paramIndex = jointIndex * kParametersPerJoint;
425 const Vector3<T> posd = position_i - jointState.translation();
426
427 // calculate derivatives based on active joints
428 for (size_t d = 0; d < 3; d++) {
429 if (this->activeJointParams_[paramIndex + d]) {
430 // calculate joint gradient
431 const T val = direction.dot(jointState.getTranslationDerivative(d)) * -fac;
432 // explicitly multiply with the parameter transform to generate parameter space
433 // gradients
434 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
435 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
436 ++index) {
437 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
438 val * this->parameterTransform_.transform.valuePtr()[index];
439 }
440 }
441 if (this->activeJointParams_[paramIndex + 3 + d]) {
442 // calculate joint gradient
443 const T val = direction.dot(jointState.getRotationDerivative(d, posd)) * -fac;
444 // explicitly multiply with the parameter transform to generate parameter space
445 // gradients
446 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
447 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
448 ++index) {
449 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
450 val * this->parameterTransform_.transform.valuePtr()[index];
451 }
452 }
453 }
454 if (this->activeJointParams_[paramIndex + 6]) {
455 // calculate joint gradient
456 const T val = direction.dot(jointState.getScaleDerivative(posd)) * -fac;
457 // explicitly multiply with the parameter transform to generate parameter space
458 // gradients
459 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
460 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
461 ++index) {
462 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
463 val * this->parameterTransform_.transform.valuePtr()[index];
464 }
465 }
466
467 // go to the next joint
468 jointIndex = this->skeleton_.joints[jointIndex].parent;
469 }
470
471 // -----------------------------------
472 // process second joint
473 // -----------------------------------
474 jointIndex = collisionGeometry[pr[1]].parent;
475 while (jointIndex != kInvalidIndex && jointIndex != gsl::narrow_cast<size_t>(pr[2])) {
476 const auto& jointState = state.jointState[jointIndex];
477 const size_t paramIndex = jointIndex * kParametersPerJoint;
478 const Vector3<T> posd = position_j - jointState.translation();
479
480 // calculate derivatives based on active joints
481 for (size_t d = 0; d < 3; d++) {
482 if (this->activeJointParams_[paramIndex + d]) {
483 // calculate joint gradient
484 const T val = direction.dot(jointState.getTranslationDerivative(d)) * fac;
485 // explicitly multiply with the parameter transform to generate parameter space

487 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
488 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];

490 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
491 val * this->parameterTransform_.transform.valuePtr()[index];

493 }

495 // calculate joint gradient

497 // explicitly multiply with the parameter transform to generate parameter space
498 // gradients
499 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
500 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
501 ++index) {
502 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
503 val * this->parameterTransform_.transform.valuePtr()[index];
504 }
505 }
506 }
507 if (this->activeJointParams_[paramIndex + 6]) {
508 // calculate joint gradient
509 const T val = direction.dot(jointState.getScaleDerivative(posd)) * fac;
510 // explicitly multiply with the parameter transform to generate parameter space
511 // gradients
512 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
