49template <typename T>
50nlohmann::json transformToJson(const TransformT<T>& transform) {
51 nlohmann::json j;
52
53 // Extract translation
54 j["Translation"] = eigenToJsonArray(transform.translation);
55
56 // Extract rotation as quaternion
57 j["Rotation"] = eigenToJsonArray(transform.rotation);
58
59 // Extract scale
60 j["Scale"] = transform.scale;
61
62 return j;
63}

81Eigen::Quaternion<T> jsonArrayToEigenQuaternion(const nlohmann::json& j) {
82 MT_THROW_IF(j.size() != 4, "Expected array of size 4 for Quaternion");
83 // Eigen constructor takes (w, x, y, z); json storage is (x, y, z, w)

88template <typename T>
89TransformT<T> jsonToTransform(const nlohmann::json& j) {
90 TransformT<T> transform; // Identity by default
91
92 if (j.contains("Translation")) {
93 transform.translation = jsonArrayToEigenVector3<T>(j["Translation"]);
94 }
95
96 if (j.contains("Rotation")) {
97 transform.rotation = jsonArrayToEigenQuaternion<T>(j["Rotation"]);
98 }
99
100 if (j.contains("Scale")) {
101 transform.scale = j["Scale"].get<T>();
102 }
103
104 return transform;
105}

109 const nlohmann::json& json,
110 const std::initializer_list<const char*>& names) {

120Skeleton legacySkeletonToMomentum(const nlohmann::json& legacySkeleton) {
121 MT_THROW_IF(!legacySkeleton.contains("Bones"), "Legacy skeleton JSON missing 'Bones' field");
122
123 const auto& bones = legacySkeleton["Bones"];
124 JointList joints;
125 joints.reserve(bones.size());
126
127 for (const auto& boneJson : bones) {
128 Joint joint;
129
130 // Extract joint name
131 joint.name = boneJson["Name"].get<std::string>();
132
133 // Extract parent index
134 joint.parent = boneJson["Parent"].get<size_t>();
135
136 // Extract pre-rotation quaternion
137 if (boneJson.contains("PreRotation")) {
138 joint.preRotation = jsonArrayToEigenQuaternion<float>(boneJson["PreRotation"]);
139 } else {
140 joint.preRotation = Eigen::Quaternionf::Identity();
141 }
142
143 // Extract translation offset
144 if (boneJson.contains("TranslationOffset")) {
145 joint.translationOffset = jsonArrayToEigenVector3<float>(boneJson["TranslationOffset"]);
146 } else {
147 joint.translationOffset = Eigen::Vector3f::Zero();
148 }
149
150 joints.push_back(joint);
151 }
152
153 return Skeleton(joints);
154}

156nlohmann::json momentumSkeletonToLegacy(const Skeleton& skeleton) {
157 nlohmann::json legacySkeleton;
158 nlohmann::json bones = nlohmann::json::array();
159
160 for (const auto& joint : skeleton.joints) {
161 nlohmann::json boneJson;
162
163 boneJson["Name"] = joint.name;
164 boneJson["Parent"] = joint.parent;
165 boneJson["PreRotation"] = eigenToJsonArray(joint.preRotation);
166 boneJson["TranslationOffset"] = eigenToJsonArray(joint.translationOffset);
167
168 // Add default values for legacy compatibility
169 boneJson["RestState"] = nlohmann::json{
170 {"Rot", nlohmann::json::array({0.0, 0.0, 0.0})},
171 {"Trans", nlohmann::json::array({0.0, 0.0, 0.0})},
172 {"Scale", 0.0}};
173
174 // Determine joint type
175 if (joint.parent == kInvalidIndex) {
176 boneJson["JointType"] = "Root";
177 } else {
178 boneJson["JointType"] = "Limb";
179 }
180
181 boneJson["RotationOrder"] = "XYZ";
182
183 bones.push_back(boneJson);
184 }
185
186 legacySkeleton["Bones"] = bones;
187 return legacySkeleton;

190std::pair<Mesh, SkinWeights> legacySkinnedModelToMomentum(
191 const nlohmann::json& legacySkinnedModel) {
192 Mesh mesh;
193 SkinWeights skinWeights;
194
195 // Extract vertices - support multiple naming variants
196 auto verticesItr = findFieldNames(legacySkinnedModel, {"RestPositions", "vertices"});
197 if (verticesItr != legacySkinnedModel.end()) {
198 const auto& vertices = *verticesItr;
199 mesh.vertices.reserve(vertices.size());
200 for (const auto& vertex : vertices) {
201 mesh.vertices.push_back(jsonArrayToEigenVector3<float>(vertex));
202 }
203 }
204
205 // Extract normals - support multiple naming variants
206 auto normalsItr = findFieldNames(legacySkinnedModel, {"RestVertexNormals", "normals"});
207 if (normalsItr != legacySkinnedModel.end()) {
208 const auto& normals = *normalsItr;
209 mesh.normals.reserve(normals.size());
210 for (const auto& normal : normals) {
211 mesh.normals.push_back(jsonArrayToEigenVector3<float>(normal));
212 }
213 }
214
215 // Extract faces - support multiple naming variants and triangulate if needed
216 auto facesItr = findFieldNames(legacySkinnedModel, {"Faces", "faces"});
217 if (facesItr != legacySkinnedModel.end()) {
218 const auto& facesObj = *facesItr;
219 // Expecting "Indices", optional "TextureIndices", and "Offsets"
220 MT_THROW_IF(!facesObj.contains("Indices"), "Faces object missing 'Indices' field");
221 MT_THROW_IF(!facesObj.contains("Offsets"), "Faces object missing 'Offsets' field");
222
223 const auto& indices = facesObj["Indices"];
224 const auto& offsets = facesObj["Offsets"];
225 // Optional texture indices
226 const auto* texIndicesPtr =
227 facesObj.contains("TextureIndices") ? &facesObj["TextureIndices"] : nullptr;
228
229 size_t numFaces = offsets.size() - 1;
230 mesh.faces.reserve(numFaces * 2); // Conservative estimate for triangulation
231
232 for (size_t faceIdx = 0; faceIdx < numFaces; ++faceIdx) {
233 size_t start = offsets[faceIdx].get<size_t>();
234 size_t end = offsets[faceIdx + 1].get<size_t>();
235 size_t faceSize = end - start;
236
237 // Triangulate face (ngon) using fan triangulation
238 if (faceSize < 3) {
239 MT_THROW_IF(true, "Face must have at least 3 vertices");
240 } else {
241 int v0 = indices[start].get<int>();
242 for (size_t i = 1; i < faceSize - 1; ++i) {
243 int v1 = indices[start + i].get<int>();
244 int v2 = indices[start + i + 1].get<int>();
245 mesh.faces.emplace_back(v0, v1, v2);
246 }
247 if (texIndicesPtr && texIndicesPtr->is_array()) {
248 int t0 = (*texIndicesPtr)[start].get<int>();
249 for (size_t i = 1; i < faceSize - 1; ++i) {
250 int t1 = (*texIndicesPtr)[start + i].get<int>();
251 int t2 = (*texIndicesPtr)[start + i + 1].get<int>();
252 mesh.texcoord_faces.emplace_back(t0, t1, t2);
253 }
254 }
255 }
256 }
257
258 // Shrink to actual size after triangulation
259 mesh.faces.shrink_to_fit();
260 mesh.texcoord_faces.shrink_to_fit();
261 }
262
263 // Extract texture coordinates - support multiple naming variants
264 auto texcoordsItr = findFieldNames(legacySkinnedModel, {"TextureCoordinates", "texcoords"});
265 if (texcoordsItr != legacySkinnedModel.end()) {
266 const auto& texcoords = *texcoordsItr;
267 mesh.texcoords.reserve(texcoords.size());
268 for (const auto& texcoord : texcoords) {
269 mesh.texcoords.push_back(jsonArrayToEigenVector2<float>(texcoord));
270 }
271 }
272
273 // Try momentum model rig format with separate SkinningWeights and SkinningOffsets
274 auto skinningWeightsItr = findFieldNames(legacySkinnedModel, {"SkinningWeights"});
275 auto skinningOffsetsItr = findFieldNames(legacySkinnedModel, {"SkinningOffsets"});
276
277 if (skinningWeightsItr != legacySkinnedModel.end() &&
278 skinningOffsetsItr != legacySkinnedModel.end()) {
279 const auto& skinningWeights = *skinningWeightsItr;
280 const auto& skinningOffsets = *skinningOffsetsItr;
281
282 if (skinningWeights.is_array() && skinningOffsets.is_array()) {
283 const size_t numVertices = skinningOffsets.size() - 1;
284 skinWeights.index = IndexMatrix::Zero(numVertices, kMaxSkinJoints);
285 skinWeights.weight = WeightMatrix::Zero(numVertices, kMaxSkinJoints);
286
287 for (size_t i = 0; i < numVertices; ++i) {
288 size_t start = skinningOffsets[i].get<size_t>();
289 size_t end = skinningOffsets[i + 1].get<size_t>();
290 size_t numInfluences = std::min(end - start, static_cast<size_t>(kMaxSkinJoints));
291
292 for (size_t j = 0; j < numInfluences; ++j) {
293 const auto& pair = skinningWeights[start + j];
294 // Each pair is [joint index, skinning weight]
295 skinWeights.index(i, j) = pair[0].get<uint32_t>();
296 skinWeights.weight(i, j) = pair[1].get<float>();
297 }
298 }
299 }
300 }
301
302 return std::make_pair(std::move(mesh), std::move(skinWeights));
303}

305nlohmann::json momentumSkinnedModelToLegacy(const Mesh& mesh, const SkinWeights& skinWeights) {
306 nlohmann::json legacySkinnedModel;
307
308 // Convert vertices
309 nlohmann::json restPositions = nlohmann::json::array();
310 for (const auto& vertex : mesh.vertices) {
311 restPositions.push_back(eigenToJsonArray(vertex));
312 }
313 legacySkinnedModel["RestPositions"] = restPositions;
314
315 // Convert normals
316 if (!mesh.normals.empty()) {
317 nlohmann::json restVertexNormals = nlohmann::json::array();
318 for (const auto& normal : mesh.normals) {
319 restVertexNormals.push_back(eigenToJsonArray(normal));
320 }
321 legacySkinnedModel["RestVertexNormals"] = restVertexNormals;
322 }
323
324 // Convert faces (legacy expects an object with Indices, Offsets, and optionally TextureIndices)
325 nlohmann::json facesObj;
326 nlohmann::json indices = nlohmann::json::array();
327 nlohmann::json offsets = nlohmann::json::array();
328 nlohmann::json textureIndices = nlohmann::json::array();
329
330 // To reconstruct offsets, we need to group triangles into faces.
331 // But since we only have triangles, we treat each triangle as a face.
332 size_t numFaces = mesh.faces.size();
333 offsets.push_back(0);
334 for (size_t i = 0; i < numFaces; ++i) {
335 const auto& face = mesh.faces[i];
336 indices.push_back(face.x());
337 indices.push_back(face.y());
338 indices.push_back(face.z());
339 offsets.push_back((i + 1) * 3);
340 // If texcoord_faces exist, add them as TextureIndices
341 if (!mesh.texcoord_faces.empty()) {
342 const auto& tface = mesh.texcoord_faces[i];
343 textureIndices.push_back(tface.x());
344 textureIndices.push_back(tface.y());
345 textureIndices.push_back(tface.z());
346 }
347 }
348 facesObj["Indices"] = indices;
349 facesObj["Offsets"] = offsets;
350 if (!textureIndices.empty()) {
351 facesObj["TextureIndices"] = textureIndices;
352 }
353 legacySkinnedModel["Faces"] = facesObj;
354
355 // Convert texture coordinates
356 if (!mesh.texcoords.empty()) {
357 nlohmann::json textureCoordinates = nlohmann::json::array();
358 for (const auto& texcoord : mesh.texcoords) {
359 textureCoordinates.push_back(eigenToJsonArray(texcoord));
360 }
361 legacySkinnedModel["TextureCoordinates"] = textureCoordinates;
362 }
363
364 // Convert skin weights to momentum model format: SkinningWeights and SkinningOffsets
365 nlohmann::json skinningWeights = nlohmann::json::array();
366 nlohmann::json skinningOffsets = nlohmann::json::array();
367 size_t runningIndex = 0;
368 skinningOffsets.push_back(runningIndex);
369 for (int i = 0; i < skinWeights.index.rows(); ++i) {
370 int influences = 0;
371 for (int j = 0; j < skinWeights.index.cols(); ++j) {
372 if (skinWeights.weight(i, j) > 0.0f) {
373 nlohmann::json pair = nlohmann::json::array();
374 pair.push_back(skinWeights.index(i, j));
375 pair.push_back(skinWeights.weight(i, j));
376 skinningWeights.push_back(pair);
377 ++influences;
378 }
379 }
380 runningIndex += influences;
381 skinningOffsets.push_back(runningIndex);
382 }
383 legacySkinnedModel["SkinningWeights"] = skinningWeights;
384 legacySkinnedModel["SkinningOffsets"] = skinningOffsets;
385
386 return legacySkinnedModel;
387}

389CollisionGeometry legacyCollisionToMomentum(const nlohmann::json& legacyCollision) {
390 CollisionGeometry collision;
391
392 if (legacyCollision.is_array()) {
393 collision.reserve(legacyCollision.size());
394
395 for (const auto& capsuleJson : legacyCollision) {
396 TaperedCapsule capsule;
397
398 if (capsuleJson.contains("transformation")) {
399 capsule.transformation = jsonToTransform<float>(capsuleJson["transformation"]);
400 }
401
402 if (capsuleJson.contains("radius")) {
403 capsule.radius = jsonArrayToEigenVector2<float>(capsuleJson["radius"]);
404 }
405
406 if (capsuleJson.contains("parent")) {
407 capsule.parent = capsuleJson["parent"].get<size_t>();
408 }
409
410 if (capsuleJson.contains("length")) {
411 capsule.length = capsuleJson["length"].get<float>();
412 }
413
414 collision.push_back(capsule);
415 }
416 }
417
418 return collision;
419}

438LocatorList legacyLocatorsToMomentum(const nlohmann::json& legacyLocators) {
439 LocatorList locators;
440
441 if (legacyLocators.is_array()) {
442 locators.reserve(legacyLocators.size());
443
444 for (const auto& locatorJson : legacyLocators) {
445 Locator locator;
446
447 if (locatorJson.contains("name")) {
448 locator.name = locatorJson["name"].get<std::string>();
449 }
450
451 if (locatorJson.contains("parent")) {
452 locator.parent = locatorJson["parent"].get<size_t>();
453 }
454
455 if (locatorJson.contains("offset")) {
456 locator.offset = jsonArrayToEigenVector3<float>(locatorJson["offset"]);
457 } else if (
458 locatorJson.contains("offsetX") && locatorJson.contains("offsetY") &&
459 locatorJson.contains("offsetZ")) {
460 // Legacy format with separate X, Y, Z fields
461 locator.offset = Eigen::Vector3f(
462 locatorJson["offsetX"].get<float>(),
463 locatorJson["offsetY"].get<float>(),
464 locatorJson["offsetZ"].get<float>());
465 }
466
467 // Set default values for other fields
468 locator.locked = Eigen::Vector3i::Zero();
469 locator.weight = 1.0f;
470 locator.limitOrigin = locator.offset;
471 locator.limitWeight = Eigen::Vector3f::Zero();
472
473 locators.push_back(locator);
474 }
475 }
476
477 return locators;
478}

515Character loadCharacterFromLegacyJsonString(const std::string& jsonString) {
516 nlohmann::json j = nlohmann::json::parse(jsonString);
517
518 // Extract skeleton - support multiple naming variants for backward compatibility
519 auto skeletonItr = findFieldNames(j, {"Skeleton", "BodySkeleton", "skeleton"});
520 MT_THROW_IF(skeletonItr == j.end(), "Legacy JSON missing skeleton field");
521 Skeleton skeleton = legacySkeletonToMomentum(*skeletonItr);
522
523 // Create default parameter transform
524 ParameterTransform parameterTransform =
525 ParameterTransform::empty(skeleton.joints.size() * kParametersPerJoint);
526
527 // Extract mesh and skin weights if present - support multiple naming variants
528 std::unique_ptr<Mesh> mesh;
529 std::unique_ptr<SkinWeights> skinWeights;
530
531 auto skinnedModelItr = findFieldNames(j, {"SkinnedModel", "BodySkinnedModel", "skinnedmodel"});
532 if (skinnedModelItr != j.end()) {
533 auto [meshData, skinWeightsData] = legacySkinnedModelToMomentum(*skinnedModelItr);
534 mesh = std::make_unique<Mesh>(std::move(meshData));
535 skinWeights = std::make_unique<SkinWeights>(std::move(skinWeightsData));
536 }
537
538 // Extract collision geometry if present - support multiple naming variants
539 std::unique_ptr<CollisionGeometry> collision;
540 auto collisionItr = findFieldNames(j, {"Collision", "collision"});
541 if (collisionItr != j.end()) {
542 CollisionGeometry collisionData = legacyCollisionToMomentum(*collisionItr);
543 if (!collisionData.empty()) {
544 collision = std::make_unique<CollisionGeometry>(std::move(collisionData));
545 }
546 }
547
548 // Extract locators if present - support multiple naming variants
549 LocatorList locators;
550 auto locatorsItr = findFieldNames(j, {"Locators", "locators"});
551 if (locatorsItr != j.end()) {
552 locators = legacyLocatorsToMomentum(*locatorsItr);
553 }
554
555 // Create and return the character
556 return {
557 skeleton,
558 parameterTransform,
559 ParameterLimits(), // Default parameter limits
560 locators,
561 mesh.get(),
562 skinWeights.get(),
563 collision.get()};
564}

580nlohmann::json characterToLegacyJson(const Character& character) {
581 nlohmann::json j;
582
583 // Convert skeleton - use capitalized naming to match BasicBodyProfile
584 j["Skeleton"] = momentumSkeletonToLegacy(character.skeleton);
585
586 // Convert mesh and skin weights if present - use capitalized naming to match BasicBodyProfile
587 if (character.mesh && character.skinWeights) {
588 j["SkinnedModel"] = momentumSkinnedModelToLegacy(*character.mesh, *character.skinWeights);
589 }
590
591 // Convert collision geometry if present
592 if (character.collision && !character.collision->empty()) {
593 j["Collision"] = momentumCollisionToLegacy(*character.collision);
594 }
595
596 // Convert locators if present
597 if (!character.locators.empty()) {
598 j["Locators"] = momentumLocatorsToLegacy(character.locators);
599 }
600
601 return j;
602}
