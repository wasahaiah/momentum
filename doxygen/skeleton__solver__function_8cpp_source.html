24template <typename T>
25SkeletonSolverFunctionT<T>::SkeletonSolverFunctionT(
26 const Character& character,
27 const ParameterTransformT<T>& parameterTransform,
28 std::span<const std::shared_ptr<SkeletonErrorFunctionT<T>>> errorFunctions)
29 : character_(character), parameterTransform_(parameterTransform), needsMeshState_(false) {
30 this->numParameters_ = parameterTransform_.numAllModelParameters();
31 this->actualParameters_ = this->numParameters_;
32 state_ = std::make_unique<SkeletonStateT<T>>(parameterTransform_.zero(), character_.skeleton);
33 activeJointParams_ = parameterTransform_.activeJointParams;
34 meshState_ = std::make_unique<MeshStateT<T>>();
35
36 for (auto& errf : errorFunctions) {
37 addErrorFunction(std::move(errf));
38 }
39}

44template <typename T>
45void SkeletonSolverFunctionT<T>::setEnabledParameters(const ParameterSet& ps) {
46 // find the last enabled parameter
47 this->actualParameters_ = 0;
48 for (size_t i = 0; i < this->numParameters_; i++) {
49 if (ps.test(i)) {
50 this->actualParameters_ = gsl::narrow_cast<int>(i + 1);
51 }
52 }
53 // set the enabled joints based on the parameter set
54 activeJointParams_ = parameterTransform_.computeActiveJointParams(ps);
55
56 // give data to helper functions
57 for (auto&& solvable : errorFunctions_) {
58 solvable->setActiveJoints(activeJointParams_);
59 solvable->setEnabledParameters(ps);
60 }
61}

63template <typename T>
64double SkeletonSolverFunctionT<T>::getError(const Eigen::VectorX<T>& parameters) {
65 // update the state according to the transformed parameters
66 state_->set(parameterTransform_.apply(parameters), character_.skeleton, false);
67
68 // Update mesh state if needed
69 if (needsMeshState()) {
70 updateMeshState(parameters, *state_);
71 }
72
73 double error = 0.0;
74
75 // sum up error for all solvables
76 for (auto&& solvable : errorFunctions_) {
77 if (solvable->getWeight() > 0.0f) {
78 error += (double)solvable->getError(parameters, *state_, *meshState_);
79 }
80 }
81
82 return (float)error;
83}

113template <typename T>
114std::pair<size_t, std::vector<size_t>> getDimensions(
115 const std::vector<std::shared_ptr<SkeletonErrorFunctionT<T>>>& error_func) {
116 MT_PROFILE_FUNCTION();
117 std::vector<size_t> offset(error_func.size());
118 size_t jacobianSize = 0;
119 for (size_t i = 0; i < error_func.size(); i++) {
120 const auto& solvable = error_func[i];
121
122 // ! offset values are only defined when the cost function is active
123 if (solvable->getWeight() > 0.0f) {
124 const auto jSize = solvable->getJacobianSize();
125 offset[i] = jSize;
126 jacobianSize += jSize;
127 }
128 }
129 jacobianSize += 8 - (jacobianSize % 8);
130 return std::make_pair(jacobianSize, std::move(offset));
131}

133template <typename T>
134double SkeletonSolverFunctionT<T>::getJacobian(
135 const Eigen::VectorX<T>& parameters,
136 Eigen::MatrixX<T>& jacobian,
137 Eigen::VectorX<T>& residual,
138 size_t& actualRows) {
139 MT_PROFILE_FUNCTION();
140
141 // update the state according to the transformed parameters
142 {
143 MT_PROFILE_EVENT("Set state");
144 state_->set(parameterTransform_.apply(parameters), character_.skeleton);
145 }
146
147 // Update mesh state if needed
148 if (needsMeshState()) {
149 updateMeshState(parameters, *state_);
150 }
151
152 double error = 0.0;
153
154 // calculate the jacobian size
155 const auto dimensions = getDimensions(errorFunctions_);
156 const auto& jacobianSize = dimensions.first;
157 const auto& offset = dimensions.second;
158
159 if (jacobianSize > static_cast<size_t>(jacobian.rows()) || parameters.size() != jacobian.cols()) {
160 MT_PROFILE_EVENT("ResizeJacobian");
161 jacobian.resize(jacobianSize, parameters.size());
162 residual.resize(jacobianSize);
163 }
164 {
165 MT_PROFILE_EVENT("InitializeJacobian");
166 jacobian.setZero();
167 residual.setZero();
168 }
169 actualRows = jacobianSize;
170
171 // add values to the jacobian
172 size_t position = 0;
173
174 MT_PROFILE_EVENT("Collect all jacobians");
175
176 for (size_t i = 0; i < errorFunctions_.size(); i++) {
177 int rows = 0;
178 const auto& solvable = errorFunctions_[i];
179 if (solvable->getWeight() > 0.0f) {
180 const auto& n = offset[i];
181 error += solvable->getJacobian(
182 parameters,
183 *state_,
184 *meshState_,
185 jacobian.block(position, 0, n, parameters.size()),
186 residual.middleRows(position, n),
187 rows);
188 position += n;
189 }
190 }
191
192 return error;
193}

196template <typename T>
197double SkeletonSolverFunctionT<T>::getJtJR(
198 const Eigen::VectorX<T>& parameters,
199 Eigen::MatrixX<T>& JtJ,
200 Eigen::VectorX<T>& JtR) {
201 MT_PROFILE_FUNCTION();
202
203 // update the state according to the transformed parameters
204 {
205 MT_PROFILE_EVENT("JtJR - update state");
206 state_->set(parameterTransform_.apply(parameters), character_.skeleton);
207 }
208
209 // Update mesh state if needed
210 if (needsMeshState()) {
211 updateMeshState(parameters, *state_);
212 }
213
214 // calculate the jacobian size
215 const auto dimensions = getDimensions(errorFunctions_);
216 const auto& jacobianSize = dimensions.first;
217 const auto n_parameters = parameters.size();
218
219 if (dimensions.first > static_cast<size_t>(tJacobian_.rows()) ||
220 parameters.size() != tJacobian_.cols()) {
221 MT_PROFILE_EVENT("ResizetJacobian");
222 tJacobian_.resize(jacobianSize, n_parameters);
223 tResidual_.resize(jacobianSize);
224 }
225
226 if (JtJ.cols() != gsl::narrow_cast<Eigen::Index>(this->actualParameters_)) {
227 JtJ.resize(this->actualParameters_, this->actualParameters_);
228 JtR.resize(this->actualParameters_);
229 }
230
231 {
232 MT_PROFILE_EVENT("JtJR - set to Zero");
233
234 tJacobian_.topRows(jacobianSize).setZero();
235 tResidual_.head(jacobianSize).setZero();
236 JtJ.setZero();
237 JtR.setZero();
238 }
239
240 // block compute JtJ and JtR on the fly
241 size_t position = 0;
242 double error = 0.0;
243 for (size_t i = 0; i < errorFunctions_.size(); i++) {
244 const auto& solvable = errorFunctions_[i];
245
246 if (solvable->getWeight() > 0.0f) {
247 int rows = 0;
248 const auto& n = dimensions.second[i];
249
250 // This is timed through the corresponding solvable call
251 error += solvable->getJacobian(
252 parameters,
253 *state_,
254 *meshState_,
255 tJacobian_.block(position, 0, n, n_parameters),
256 tResidual_.segment(position, n),
257 rows);
258 MT_CHECK(rows <= gsl::narrow_cast<int>(n));
259
260 // Update JtJ
261 if (rows > 0) {
262 MT_PROFILE_EVENT("Partial JtJ JtR");
263
264 // ! In truth, on the the "this->actualParameters_" leftmost block will be used
265 // We take advantage of this here and skip the other computations
266 const auto JtBlock =
267 tJacobian_.block(position, 0, rows, this->actualParameters_).transpose();
268
269 // Efficiently update JtJ (JtJ = J^T * J) using selfadjointView with rankUpdate,
270 // replacing triangularView to leverage symmetry and improve performance
271 JtJ.template selfadjointView<Eigen::Lower>().rankUpdate(JtBlock);
272
273 // Update JtR
274 JtR.noalias() += JtBlock * tResidual_.segment(position, rows);
275 }
276 position += n;
277 }
278 }
279
280 return error;
281}

284template <typename T>
285double SkeletonSolverFunctionT<T>::getSolverDerivatives(
286 const Eigen::VectorX<T>& parameters,
287 Eigen::MatrixX<T>& hess,
288 Eigen::VectorX<T>& grad) {
289 MT_PROFILE_FUNCTION();
290
291 // update the state according to the transformed parameters
292 {
293 MT_PROFILE_EVENT("UpdateState");
294 state_->set(parameterTransform_.apply(parameters), character_.skeleton);
295 }
296
297 // Update mesh state if needed
298 if (needsMeshState()) {
299 updateMeshState(parameters, *state_);
300 }
301
302 if (hess.cols() != gsl::narrow_cast<Eigen::Index>(this->actualParameters_)) {
303 hess.resize(this->actualParameters_, this->actualParameters_);
304 grad.resize(this->actualParameters_);
305 }
306
307 hess.setZero();
308 grad.setZero();
309
310 double error = 0.0;
311 for (size_t i = 0; i < errorFunctions_.size(); i++) {
312 const auto& solvable = errorFunctions_[i];
313 if (solvable->getWeight() > 0.0f) {
314 error += solvable->getSolverDerivatives(
315 parameters, *state_, *meshState_, this->actualParameters_, hess, grad);
316 }
317 }
318
319 return error;
320}
